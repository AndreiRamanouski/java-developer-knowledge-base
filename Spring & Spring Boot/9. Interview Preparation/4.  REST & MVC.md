## REST & MVC Questions

### Q1: DispatcherServlet flow

**Short Answer**: DispatcherServlet is Spring MVC's front controller that receives all HTTP requests, delegates to handlers, resolves views, and returns responses. It follows a well-defined processing chain involving handler mapping, handler adapters, view resolvers, and interceptors.

**Complete Request Flow**:

```
1. Client Request
        ↓
2. DispatcherServlet (Front Controller)
        ↓
3. HandlerMapping (Find handler)
        ↓
4. HandlerInterceptor.preHandle()
        ↓
5. HandlerAdapter (Execute handler)
        ↓
6. Controller Method
        ↓
7. HandlerInterceptor.postHandle()
        ↓
8. ViewResolver (if view name returned)
        ↓
9. View Rendering
        ↓
10. HandlerInterceptor.afterCompletion()
        ↓
11. Response to Client
```

**Detailed Step-by-Step Flow**:

```java
// Step 1: Client makes HTTP request
// GET /api/users/123

// Step 2: DispatcherServlet receives request
public class DispatcherServlet extends FrameworkServlet {
    
    @Override
    protected void doService(HttpServletRequest request, 
                            HttpServletResponse response) {
        
        // Step 3: Find handler for this request
        HandlerExecutionChain chain = getHandler(request);
        // Returns: UserController.getUser() + interceptors
        
        if (chain == null) {
            // No handler found - 404
            noHandlerFound(request, response);
            return;
        }
        
        // Step 4: Apply pre-handle interceptors
        if (!chain.applyPreHandle(request, response)) {
            return; // Interceptor stopped processing
        }
        
        // Step 5: Get appropriate adapter for handler
        HandlerAdapter adapter = getHandlerAdapter(chain.getHandler());
        
        // Step 6: Execute handler (controller method)
        ModelAndView mv = adapter.handle(request, response, chain.getHandler());
        
        // Step 7: Apply post-handle interceptors
        chain.applyPostHandle(request, response, mv);
        
        // Step 8: Process result (render view or return JSON)
        processDispatchResult(request, response, chain, mv, null);
        
        // Step 9: Apply after-completion interceptors
        chain.triggerAfterCompletion(request, response, null);
    }
}
```

**2. HandlerExecutionChain**:

```java
public class HandlerExecutionChain {
    private final Object handler; // Controller method
    private HandlerInterceptor[] interceptors;
    
    boolean applyPreHandle(HttpServletRequest request, 
                          HttpServletResponse response) {
        for (HandlerInterceptor interceptor : interceptors) {
            if (!interceptor.preHandle(request, response, handler)) {
                return false; // Stop processing
            }
        }
        return true;
    }
    
    void applyPostHandle(HttpServletRequest request,
                        HttpServletResponse response,
                        ModelAndView mv) {
        for (int i = interceptors.length - 1; i >= 0; i--) {
            interceptors[i].postHandle(request, response, handler, mv);
        }
    }
    
    void triggerAfterCompletion(HttpServletRequest request,
                               HttpServletResponse response,
                               Exception ex) {
        for (int i = interceptors.length - 1; i >= 0; i--) {
            interceptors[i].afterCompletion(request, response, handler, ex);
        }
    }
}
```

**3. HandlerInterceptor**:

```java
public interface HandlerInterceptor {
    
    // Before controller execution
    default boolean preHandle(HttpServletRequest request,
                             HttpServletResponse response,
                             Object handler) {
        return true; // Continue processing
    }
    
    // After controller execution, before view rendering
    default void postHandle(HttpServletRequest request,
                           HttpServletResponse response,
                           Object handler,
                           ModelAndView modelAndView) {
    }
    
    // After view rendering (always called, even on exceptions)
    default void afterCompletion(HttpServletRequest request,
                                HttpServletResponse response,
                                Object handler,
                                Exception ex) {
    }
}

// Real-world example: Authentication interceptor
@Component
public class AuthenticationInterceptor implements HandlerInterceptor {
    
    @Autowired
    private TokenService tokenService;
    
    @Override
    public boolean preHandle(HttpServletRequest request,
                            HttpServletResponse response,
                            Object handler) {
        String token = request.getHeader("Authorization");
        
        if (token == null || !tokenService.isValid(token)) {
            response.setStatus(HttpServletResponse.SC_UNAUTHORIZED);
            return false; // Stop processing
        }
        
        // Set user context
        User user = tokenService.getUserFromToken(token);
        request.setAttribute("currentUser", user);
        
        return true; // Continue to controller
    }
    
    @Override
    public void postHandle(HttpServletRequest request,
                          HttpServletResponse response,
                          Object handler,
                          ModelAndView modelAndView) {
        // Add common model attributes
        if (modelAndView != null) {
            User user = (User) request.getAttribute("currentUser");
            modelAndView.addObject("currentUser", user);
        }
    }
    
    @Override
    public void afterCompletion(HttpServletRequest request,
                               HttpServletResponse response,
                               Object handler,
                               Exception ex) {
        // Cleanup (e.g., close resources, log request)
        long duration = System.currentTimeMillis() - 
                       (Long) request.getAttribute("startTime");
        log.info("Request processed in {}ms", duration);
    }
}

// Register interceptor
@Configuration
public class WebMvcConfig implements WebMvcConfigurer {
    
    @Autowired
    private AuthenticationInterceptor authInterceptor;
    
    @Override
    public void addInterceptors(InterceptorRegistry registry) {
        registry.addInterceptor(authInterceptor)
                .addPathPatterns("/api/**")
                .excludePathPatterns("/api/auth/login", "/api/public/**");
    }
}
```

**4. HandlerAdapter**:

```java
public interface HandlerAdapter {
    boolean supports(Object handler);
    ModelAndView handle(HttpServletRequest request,
                       HttpServletResponse response,
                       Object handler);
}

// RequestMappingHandlerAdapter - handles @RequestMapping methods
public class RequestMappingHandlerAdapter extends AbstractHandlerMethodAdapter {
    
    @Override
    protected ModelAndView handleInternal(HttpServletRequest request,
                                         HttpServletResponse response,
                                         HandlerMethod handlerMethod) {
        // 1. Resolve method arguments
        Object[] args = resolveArguments(handlerMethod, request, response);
        // Example: @PathVariable, @RequestBody, @RequestParam
        
        // 2. Invoke controller method
        Object returnValue = handlerMethod.getMethod().invoke(
            handlerMethod.getBean(), 
            args
        );
        
        // 3. Handle return value
        return handleReturnValue(returnValue, handlerMethod, request, response);
    }
    
    private Object[] resolveArguments(HandlerMethod method,
                                     HttpServletRequest request,
                                     HttpServletResponse response) {
        MethodParameter[] parameters = method.getMethodParameters();
        Object[] args = new Object[parameters.length];
        
        for (int i = 0; i < parameters.length; i++) {
            MethodParameter param = parameters[i];
            
            if (param.hasParameterAnnotation(PathVariable.class)) {
                // Extract from URL path
                args[i] = extractPathVariable(param, request);
                
            } else if (param.hasParameterAnnotation(RequestBody.class)) {
                // Parse JSON from request body
                args[i] = parseRequestBody(param, request);
                
            } else if (param.hasParameterAnnotation(RequestParam.class)) {
                // Extract from query parameters
                args[i] = extractRequestParam(param, request);
                
            } else if (param.getParameterType() == HttpServletRequest.class) {
                args[i] = request;
                
            } else if (param.getParameterType() == HttpServletResponse.class) {
                args[i] = response;
            }
            // ... many more argument resolvers
        }
        
        return args;
    }
}
```

**Exception Handling in Flow**:

```java
public class DispatcherServlet extends FrameworkServlet {
    
    protected void doService(HttpServletRequest request,
                            HttpServletResponse response) {
        HandlerExecutionChain chain = null;
        Exception exception = null;
        
        try {
            // Find handler
            chain = getHandler(request);
            
            // Pre-handle interceptors
            if (!chain.applyPreHandle(request, response)) {
                return;
            }
            
            // Execute controller
            ModelAndView mv = adapter.handle(request, response, handler);
            
            // Post-handle interceptors
            chain.applyPostHandle(request, response, mv);
            
            // Render view
            processDispatchResult(request, response, chain, mv, null);
            
        } catch (Exception ex) {
            exception = ex;
            
            // Try to handle exception
            ModelAndView exceptionMv = processHandlerException(
                request, response, handler, ex
            );
            
            if (exceptionMv != null) {
                // Exception handled
                processDispatchResult(request, response, chain, exceptionMv, ex);
            } else {
                // Rethrow if not handled
                throw ex;
            }
            
        } finally {
            // Always call afterCompletion (even on exception)
            if (chain != null) {
                chain.triggerAfterCompletion(request, response, exception);
            }
        }
    }
}
```

### Q2: Difference between @RestController and @Controller

**Short Answer**:

- `@Controller` returns view names (HTML pages) and needs `@ResponseBody` for JSON/XML responses
- `@RestController` = `@Controller` + `@ResponseBody`, always returns data (JSON/XML), not views

**Detailed Comparison**:

**1. @Controller (Traditional MVC)**:

```java
@Controller
@RequestMapping("/web/users")
public class UserWebController {
    
    @Autowired
    private UserService userService;
    
    // Returns view name (HTML page)
    @GetMapping("/{id}")
    public String getUser(@PathVariable Long id, Model model) {
        User user = userService.findById(id);
        model.addAttribute("user", user);
        return "userDetail"; // View name → /WEB-INF/views/userDetail.jsp
    }
    
    // Returns view name (redirect)
    @PostMapping
    public String createUser(@ModelAttribute UserForm form) {
        userService.createUser(form);
        return "redirect:/web/users"; // Redirect to user list
    }
    
    // Returns JSON (needs @ResponseBody)
    @GetMapping("/{id}/json")
    @ResponseBody // Required for JSON response!
    public User getUserAsJson(@PathVariable Long id) {
        return userService.findById(id);
    }
    
    // Returns view with data
    @GetMapping
    public ModelAndView listUsers() {
        List<User> users = userService.findAll();
        ModelAndView mav = new ModelAndView("userList");
        mav.addObject("users", users);
        return mav;
    }
}
```

**2. @RestController (RESTful API)**:


**ResponseEntity with @RestController**:

```java
@RestController
@RequestMapping("/api/users")
public class UserController {
    
    // Simple return
    @GetMapping("/{id}")
    public User getUser(@PathVariable Long id) {
        return userService.findById(id);
        // Default: 200 OK with JSON body
    }
    
    // ResponseEntity for full control
    @GetMapping("/{id}/full")
    public ResponseEntity<User> getUserFull(@PathVariable Long id) {
        User user = userService.findById(id);
        
        if (user == null) {
            return ResponseEntity.notFound().build(); // 404
        }
        
        return ResponseEntity.ok()
            .header("X-User-Id", user.getId().toString())
            .eTag(user.getVersion().toString())
            .lastModified(user.getUpdatedAt())
            .body(user); // 200 with custom headers
    }
    
    // Different status codes
    @PostMapping
    public ResponseEntity<User> createUser(@RequestBody UserRequest request) {
        User user = userService.createUser(request);
        
        URI location = ServletUriComponentsBuilder
            .fromCurrentRequest()
            .path("/{id}")
            .buildAndExpand(user.getId())
            .toUri();
        
        return ResponseEntity
            .created(location) // 201 Created
            .body(user);
    }
    
    @DeleteMapping("/{id}")
    public ResponseEntity<Void> deleteUser(@PathVariable Long id) {
        userService.deleteUser(id);
        return ResponseEntity.noContent().build(); // 204 No Content
    }
    
    @PutMapping("/{id}")
    public ResponseEntity<User> updateUser(@PathVariable Long id,
                                          @RequestBody UserRequest request) {
        if (!userService.exists(id)) {
            return ResponseEntity.notFound().build(); // 404
        }
        
        User updated = userService.updateUser(id, request);
        return ResponseEntity.ok(updated); // 200
    }
}
```

**Choosing Between @Controller and @RestController**:

|Use Case|Annotation|Return Type|Response|
|---|---|---|---|
|Rendering HTML pages|@Controller|String (view name)|HTML|
|Returning JSON/XML data|@RestController|Object/List/etc|JSON/XML|
|Single JSON endpoint in MVC app|@Controller + @ResponseBody|Object|JSON|
|Mixed (HTML + JSON)|@Controller|String or @ResponseBody Object|HTML or JSON|
|RESTful API|@RestController|Object + ResponseEntity|JSON/XML|
|File download|Both (with ResponseEntity)|ResponseEntity<byte[]>|Binary|

### Q3: Exception handling strategies

**Short Answer**: Spring provides multiple exception handling strategies: `@ExceptionHandler` (method-level), `@ControllerAdvice` (global), `ResponseEntityExceptionHandler` (framework exceptions), and custom error responses. Choose based on scope (controller vs global) and use case.

**Strategy 1: @ExceptionHandler (Controller-level)**:

```java
@RestControllerAdvice
public class GlobalExceptionHandler extends ResponseEntityExceptionHandler {
    
    // Override to handle Spring MVC exceptions
    
    // 400 - Method argument not valid (validation)
    @Override
    protected ResponseEntity<Object> handleMethodArgumentNotValid(
            MethodArgumentNotValidException ex,
            HttpHeaders headers,
            HttpStatusCode status,
            WebRequest request) {
        
        Map<String, String> errors = new HashMap<>();
        ex.getBindingResult().getFieldErrors().forEach(error -> 
            errors.put(error.getField(), error.getDefaultMessage())
        );
        
        ErrorResponse errorResponse = ErrorResponse.builder()
            .status(HttpStatus.BAD_REQUEST.value())
            .error("Validation Failed")
            .message("Input validation failed")
            .path(request.getDescription(false))
            .timestamp(LocalDateTime.now())
            .validationErrors(errors)
            .build();
        
        return ResponseEntity.badRequest().body(errorResponse);
    }
    
    // 400 - Missing servlet request parameter
    @Override
    protected ResponseEntity<Object> handleMissingServletRequestParameter(
            MissingServletRequestParameterException ex,
            HttpHeaders headers,
            HttpStatusCode status,
            WebRequest request) {
        
        String message = String.format("Required parameter '%s' is missing", 
                                      ex.getParameterName());
        
        ErrorResponse error = ErrorResponse.of(
            HttpStatus.BAD_REQUEST,
            message,
            request.getDescription(false)
        );
        
        return ResponseEntity.badRequest().body(error);
    }
    
    // 415 - Unsupported media type
    @Override
    protected ResponseEntity<Object> handleHttpMediaTypeNotSupported(
            HttpMediaTypeNotSupportedException ex,
            HttpHeaders headers,
            HttpStatusCode status,
            WebRequest request) {
        
        String supported = ex.getSupportedMediaTypes().stream()
            .map(MediaType::toString)
            .collect(Collectors.joining(", "));
        
        String message = String.format(
            "Media type '%s' not supported. Supported types: %s",
            ex.getContentType(),
            supported
        );
        
        ErrorResponse error = ErrorResponse.of(
            HttpStatus.UNSUPPORTED_MEDIA_TYPE,
            message,
            request.getDescription(false)
        );
        
        return ResponseEntity
            .status(HttpStatus.UNSUPPORTED_MEDIA_TYPE)
            .body(error);
    }
    
    // 405 - Method not allowed
    @Override
    protected ResponseEntity<Object> handleHttpRequestMethodNotSupported(
            HttpRequestMethodNotSupportedException ex,
            HttpHeaders headers,
            HttpStatusCode status,
            WebRequest request) {
        
        String supported = ex.getSupportedHttpMethods().stream()
            .map(HttpMethod::name)
            .collect(Collectors.joining(", "));
        
        String message = String.format(
            "Method '%s' not supported. Supported methods: %s",
            ex.getMethod(),
            supported
        );
        
        ErrorResponse error = ErrorResponse.of(
            HttpStatus.METHOD_NOT_ALLOWED,
            message,
            request.getDescription(false)
        );
        
        return ResponseEntity
            .status(HttpStatus.METHOD_NOT_ALLOWED)
            .header(HttpHeaders.ALLOW, supported)
            .body(error);
    }
```

### Q4: Idempotency in REST APIs

**Short Answer**: Idempotency means multiple identical requests have the same effect as a single request. GET, PUT, DELETE are naturally idempotent. POST is not idempotent and requires special handling (idempotency keys) for safe retries.

**HTTP Methods and Idempotency**:

|Method|Idempotent|Safe|Example|
|---|---|---|---|
|GET|✅ Yes|✅ Yes|Get user by ID|
|PUT|✅ Yes|❌ No|Update user with full data|
|DELETE|✅ Yes|❌ No|Delete user|
|POST|❌ No|❌ No|Create user|
|PATCH|⚠️ Depends|❌ No|Partial update|
|HEAD|✅ Yes|✅ Yes|Get headers only|
|OPTIONS|✅ Yes|✅ Yes|Get allowed methods|

**Idempotent Methods**:

**GET (Naturally Idempotent)**:

```java
@RestController
@RequestMapping("/api/users")
public class UserController {
    
    @GetMapping("/{id}")
    public ResponseEntity<User> getUser(@PathVariable Long id) {
        // Calling this 1 time or 100 times has same effect
        // Always returns the same user (or 404 if not found)
        // No side effects - read-only operation
        
        return userService.findById(id)
            .map(ResponseEntity::ok)
            .orElse(ResponseEntity.notFound().build());
    }
}

// Request 1: GET /api/users/123 → Returns User(id=123)
// Request 2: GET /api/users/123 → Returns User(id=123)
// Request 3: GET /api/users/123 → Returns User(id=123)
// All requests return identical result, no state change
```

**PUT (Naturally Idempotent)**:

```java
@RestController
@RequestMapping("/api/users")

            @PathVariable Long id,

// Request 3: PUT /api/users/123 with {name:"John", email:"john@example.com"}
```

**DELETE (Naturally Idempotent)**:

```java
@RestController
@RequestMapping("/api/users")
public class UserController {
    
    @DeleteMapping("/{id}")
    public ResponseEntity<Void> deleteUser(@PathVariable Long id) {
        // First call: Deletes user → 204 No Content
        // Subsequent calls: User already deleted → 404 Not Found or 204
        // End result: User is deleted (same state)
        
        if (!userService.exists(id)) {
            return ResponseEntity.notFound().build();
        }
        
        userService.deleteById(id);
        return ResponseEntity.noContent().build();
    }
}

// Request 1: DELETE /api/users/123 → 204 No Content (deleted)
// Request 2: DELETE /api/users/123 → 404 Not Found (already deleted)
// Request 3: DELETE /api/users/123 → 404 Not Found (still deleted)
// Final state: User doesn't exist (idempotent)
```

**Making POST Idempotent - Strategy 1: Idempotency Key**:

```java
@RestController
@RequestMapping("/api/orders")
public class OrderController {
    
    @Autowired
    private IdempotencyService idempotencyService;
    
    @Autowired
    private OrderService orderService;
    
    @PostMapping
    public ResponseEntity<Order> createOrder(
            @RequestHeader("Idempotency-Key") String idempotencyKey,
            @RequestBody CreateOrderRequest request) {
        
        // Check if request was already processed
        Optional<Order> existing = idempotencyService.getResult(idempotencyKey);
        if (existing.isPresent()) {
            // Return cached result from first request
            return ResponseEntity.ok(existing.get());
        }
        
        // Process request for first time
        Order order = orderService.createOrder(request);
        
        // Cache result with idempotency key
        idempotencyService.storeResult(idempotencyKey, order);
        
        return ResponseEntity
            .status(HttpStatus.CREATED)
            .body(order);
    }
}

// Idempotency service implementation
@Service
public class IdempotencyService {
    
    @Autowired
    private RedisTemplate<String, Object> redisTemplate;
    
    private static final String KEY_PREFIX = "idempotency:";
    private static final Duration TTL = Duration.ofHours(24);
    
    public Optional<Order> getResult(String idempotencyKey) {
        String key = KEY_PREFIX + idempotencyKey;
        Order order = (Order) redisTemplate.opsForValue().get(key);
        return Optional.ofNullable(order);
    }
    
    public void storeResult(String idempotencyKey, Order order) {
        String key = KEY_PREFIX + idempotencyKey;
        redisTemplate.opsForValue().set(key, order, TTL);
    }
}

// Usage:
// Request 1: POST /api/orders with Idempotency-Key: abc123
//            → Creates Order(id=1), stores in cache
// Request 2: POST /api/orders with Idempotency-Key: abc123
//            → Returns cached Order(id=1) ✓ Idempotent!
// Request 3: POST /api/orders with Idempotency-Key: abc123
//            → Returns cached Order(id=1) ✓ Idempotent!
```

```java
    @Column(unique = true, nullable = false)
    private String orderNumber; // Unique constraint
    
    private BigDecimal total;
    // ... other fields
}

@RestController
@RequestMapping("/api/orders")
public class OrderController {
    
    @PostMapping
    public ResponseEntity<Order> createOrder(@RequestBody CreateOrderRequest request) {
        try {
            // Generate deterministic order number from request
            String orderNumber = generateOrderNumber(request);
            
            Order order = new Order();
            order.setOrderNumber(orderNumber);
            order.setTotal(request.getTotal());
            
            Order created = orderService.save(order);
            
            return ResponseEntity
                .status(HttpStatus.CREATED)
                .body(created);
                
        } catch (DataIntegrityViolationException e) {
            // Duplicate order number - return existing
            Order existing = orderService.findByOrderNumber(
                generateOrderNumber(request)
            );
            return ResponseEntity.ok(existing);
        }
    }
    
    private String generateOrderNumber(CreateOrderRequest request) {
        // Generate deterministic order number
        // Same input → same order number
        return DigestUtils.md5Hex(
            request.getUserId() + ":" +
            request.getItems().toString() + ":" +
            request.getTimestamp()
        );
    }
}
```

**PATCH (Conditional Idempotency)**:

```java
@RestController
@RequestMapping("/api/users")
public class UserController {
    
    // ❌ Not idempotent - increments counter
    @PatchMapping("/{id}/increment-login-count")
    public ResponseEntity<User> incrementLoginCount(@PathVariable Long id) {
        User user = userService.findById(id).orElseThrow();
        user.setLoginCount(user.getLoginCount() + 1); // NOT idempotent!
        return ResponseEntity.ok(userService.save(user));
    }
    
    // ✅ Idempotent - sets specific value
    @PatchMapping("/{id}")
    public ResponseEntity<User> updateUser(
            @PathVariable Long id,
            @RequestBody Map<String, Object> updates) {
        
        User user = userService.findById(id).orElseThrow();
        
        if (updates.containsKey("name")) {
            user.setName((String) updates.get("name"));
        }
        if (updates.containsKey("email")) {
            user.setEmail((String) updates.get("email"));
        }
        
        // Setting specific values IS idempotent
        return ResponseEntity.ok(userService.save(user));
    }
}
```

**Payment Processing (Critical Idempotency)**:

```java
@RestController
@RequestMapping("/api/payments")
public class PaymentController {
    
    @Autowired
    private PaymentService paymentService;
    
    @Autowired
    private IdempotencyService idempotencyService;
    
    @PostMapping
    public ResponseEntity<Payment> processPayment(
            @RequestHeader("Idempotency-Key") String idempotencyKey,
            @RequestBody PaymentRequest request) {
        
        // Validate idempotency key
        if (idempotencyKey == null || idempotencyKey.isEmpty()) {
            return ResponseEntity.badRequest()
                .header("Error", "Idempotency-Key required for payments")
                .build();
        }
        
        // Check if payment already processed
        Optional<Payment> existing = idempotencyService
            .getPayment(idempotencyKey);
        
        if (existing.isPresent()) {
            Payment payment = existing.get();
            
            if (payment.getStatus() == PaymentStatus.COMPLETED) {
                // Payment already succeeded
                return ResponseEntity.ok(payment);
                
            } else if (payment.getStatus() == PaymentStatus.PROCESSING) {
                // Payment still processing (long-running)
                return ResponseEntity
                    .status(HttpStatus.CONFLICT)
                    .header("Retry-After", "5")
                    .body(payment);
                
            } else {
                // Payment failed previously - allow retry
                // Fall through to process again
            }
        }
        
        // Process payment for first time
        Payment payment = paymentService.processPayment(request, idempotencyKey);
        
        // Store result
        idempotencyService.storePayment(idempotencyKey, payment);
        
        return ResponseEntity
            .status(HttpStatus.CREATED)
            .body(payment);
    }
}
```

**Idempotency with Optimistic Locking**:

```java
@Entity
public class Wallet {
    @Id
    private Long id;
    
    private BigDecimal balance;
    
    @Version
    private Long version; // For optimistic locking
}

@RestController
@RequestMapping("/api/wallets")
public class WalletController {
    
    @PutMapping("/{id}/balance")
    public ResponseEntity<Wallet> updateBalance(
            @PathVariable Long id,
            @RequestBody UpdateBalanceRequest request) {
        
        // Idempotent: Sets balance to specific value
        // Optimistic locking prevents lost updates
        
        try {
            Wallet wallet = walletService.findById(id).orElseThrow();
            wallet.setBalance(request.getNewBalance());
            Wallet updated = walletService.save(wallet);
            
            return ResponseEntity.ok(updated);
            
        } catch (OptimisticLockException e) {
            // Concurrent modification - client should retry
            return ResponseEntity
                .status(HttpStatus.CONFLICT)
                .header("Retry-After", "1")
                .build();
        }
    }
}
```
