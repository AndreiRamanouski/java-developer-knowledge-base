## Overview

This guide covers the most common mistakes developers make with Spring Boot and industry best practices to write clean, maintainable, production-ready code. Learning from these pitfalls can save hours of debugging and prevent subtle bugs.

**Key Concept**: Spring Boot is powerful but has many hidden traps. Understanding these pitfalls and following best practices leads to robust, maintainable applications.

---

## 1. Common Mistakes

### Field Injection vs Constructor Injection

```java
/**
 * ❌ PROBLEM: Field Injection
 */
@Service
public class UserService {
    
    @Autowired
    private UserRepository userRepository;  // PROBLEMS:
    
    @Autowired
    private EmailService emailService;
    
    @Autowired
    private ValidationService validationService;
    
    // Issues:
    // 1. Hard to test (can't inject mocks without Spring)
    // 2. Hidden dependencies (constructor shows nothing)
    // 3. Not immutable (fields can be reassigned)
    // 4. Circular dependencies harder to detect
    // 5. Can't create object with 'new' keyword
    // 6. NullPointerException if Spring context not loaded
}

// Testing nightmare:
class UserServiceTest {
    
    @Test
    void testCreateUser() {
        UserService service = new UserService();
        // NullPointerException! Dependencies not injected
        // Must use Spring test context (slow, complex)
    }
}

/**
 * ⚠️ PROBLEM: Setter Injection
 */
@Service
public class UserService {
    
    private UserRepository userRepository;
    private EmailService emailService;
    
    @Autowired
    public void setUserRepository(UserRepository userRepository) {
        this.userRepository = userRepository;
    }
    
    @Autowired
    public void setEmailService(EmailService emailService) {
        this.emailService = emailService;
    }
    
    // Issues:
    // 1. Optional dependencies (can be null)
    // 2. Not immutable
    // 3. Dependencies can be changed after construction
    // 4. Still hard to test
}

/**
 * ✅ SOLUTION: Constructor Injection (BEST PRACTICE)
 */
@Service
public class UserService {
    
    private final UserRepository userRepository;  // final = immutable
    private final EmailService emailService;
    private final ValidationService validationService;
    
    // Single constructor = @Autowired optional (Spring 4.3+)
    public UserService(UserRepository userRepository,
                      EmailService emailService,
                      ValidationService validationService) {
        this.userRepository = userRepository;
        this.emailService = emailService;
        this.validationService = validationService;
    }
    
    // Benefits:
    // ✅ All dependencies visible in constructor
    // ✅ Immutable (final fields)
    // ✅ Easy to test (just call constructor)
    // ✅ NPE impossible (Spring fails fast if dependency missing)
    // ✅ Circular dependencies detected at startup
    // ✅ Can use 'new' keyword in tests
}

// Easy testing:
class UserServiceTest {
    
    @Test
    void testCreateUser() {
        UserRepository mockRepo = mock(UserRepository.class);
        EmailService mockEmail = mock(EmailService.class);
        ValidationService mockValidation = mock(ValidationService.class);
        
        UserService service = new UserService(mockRepo, mockEmail, mockValidation);
        // No Spring context needed!
    }
}

/**
 * ✅ BONUS: Lombok @RequiredArgsConstructor
 */
@Service
@RequiredArgsConstructor  // Generates constructor for final fields
public class UserService {
    
    private final UserRepository userRepository;
    private final EmailService emailService;
    private final ValidationService validationService;
    
    // Constructor automatically generated
    // Clean and concise!
}

/**
 * When to use each injection type:
 */
// Constructor Injection: 99% of cases (ALWAYS prefer this)
// - Mandatory dependencies
// - Immutability desired
// - Testing important

// Setter Injection: Rare cases
// - Optional dependencies (with defaults)
// - Bean needs to be reconfigured after construction

// Field Injection: NEVER in production code
// - Only acceptable in tests or quick prototypes
// - Even then, constructor injection is better
```

### @Transactional Pitfalls

```java
/**
 * ❌ PROBLEM 1: Self-Invocation
 */
@Service
public class UserService {
    
    @Transactional
    public void createUser(User user) {
        userRepository.save(user);
    }
    
    public void registerUser(User user) {
        validateUser(user);
        createUser(user);  // ❌ Transaction NOT applied!
        
        // Why? Spring uses proxies for @Transactional
        // this.createUser() bypasses proxy → no transaction
    }
}

/**
 * ✅ SOLUTION 1a: Extract to separate service
 */
@Service
@RequiredArgsConstructor
public class UserService {
    
    private final UserTransactionService transactionService;
    
    public void registerUser(User user) {
        validateUser(user);
        transactionService.createUser(user);  // ✅ Proxy used, transaction works
    }
}

@Service
@RequiredArgsConstructor
class UserTransactionService {
    
    private final UserRepository userRepository;
    
    @Transactional
    public void createUser(User user) {
        userRepository.save(user);
    }
}

/**
 * ✅ SOLUTION 1b: Self-injection (less elegant)
 */
@Service
public class UserService {
    
    @Autowired
    private UserService self;  // Inject proxy of itself
    
    @Transactional
    public void createUser(User user) {
        userRepository.save(user);
    }
    
    public void registerUser(User user) {
        validateUser(user);
        self.createUser(user);  // ✅ Uses proxy
    }
}

/**
 * ❌ PROBLEM 2: Wrong Propagation
 */
@Service
public class OrderService {
    
    @Transactional
    public void createOrder(Order order) {
        orderRepository.save(order);
        
        try {
            paymentService.processPayment(order);  // Creates new transaction
        } catch (PaymentException e) {
            // Order still saved! Payment failed in separate transaction
        }
    }
}

@Service
class PaymentService {
    
    @Transactional(propagation = Propagation.REQUIRES_NEW)
    public void processPayment(Order order) {
        // New transaction, independent of caller
        // If this fails, order transaction continues!
    }
}

/**
 * ✅ SOLUTION 2: Use default propagation
 */
@Service
class PaymentService {
    
    @Transactional  // Default: REQUIRED (join existing transaction)
    public void processPayment(Order order) {
        // Joins order transaction
        // If this fails, entire transaction rolls back
    }
}

/**
 * ❌ PROBLEM 3: Catching Exceptions
 */
@Service
public class UserService {
    
    @Transactional
    public void createUser(User user) {
        try {
            userRepository.save(user);
            emailService.sendWelcomeEmail(user);  // Throws exception
        } catch (Exception e) {
            log.error("Error creating user", e);
            // ❌ Exception caught, transaction still commits!
        }
    }
}

/**
 * ✅ SOLUTION 3a: Rethrow exception
 */
@Service
public class UserService {
    
    @Transactional
    public void createUser(User user) {
        try {
            userRepository.save(user);
            emailService.sendWelcomeEmail(user);
        } catch (Exception e) {
            log.error("Error creating user", e);
            throw e;  // ✅ Rethrow to trigger rollback
        }
    }
}

/**
 * ✅ SOLUTION 3b: Mark rollback manually
 */
@Service
public class UserService {
    
    @Transactional
    public void createUser(User user) {
        try {
            userRepository.save(user);
            emailService.sendWelcomeEmail(user);
        } catch (Exception e) {
            log.error("Error creating user", e);
            TransactionAspectSupport.currentTransactionStatus()
                .setRollbackOnly();  // ✅ Force rollback
        }
    }
}

/**
 * ❌ PROBLEM 4: Wrong Rollback Configuration
 */
@Service
public class UserService {
    
    @Transactional  // Only rolls back for RuntimeException by default
    public void createUser(User user) throws IOException {
        userRepository.save(user);
        sendWelcomeEmail(user);  // Throws IOException
        // ❌ IOException doesn't trigger rollback!
    }
}

/**
 * ✅ SOLUTION 4: Specify rollback exceptions
 */
@Service
public class UserService {
    
    @Transactional(rollbackFor = Exception.class)  // ✅ All exceptions
    public void createUser(User user) throws IOException {
        userRepository.save(user);
        sendWelcomeEmail(user);
    }
    
    // Or specific exceptions
    @Transactional(rollbackFor = {IOException.class, SQLException.class})
    public void updateUser(User user) throws IOException, SQLException {
        // Rolls back for IOException and SQLException
    }
}

/**
 * ❌ PROBLEM 5: Read-Only Not Set
 */
@Service
public class UserService {
    
    @Transactional  // Read-write transaction (default)
    public List<User> getAllUsers() {
        return userRepository.findAll();  // Just reading, but RW transaction
        // Wastes resources, holds locks longer
    }
}

/**
 * ✅ SOLUTION 5: Use readOnly for queries
 */
@Service
public class UserService {
    
    @Transactional(readOnly = true)  // ✅ Optimized for reads
    public List<User> getAllUsers() {
        return userRepository.findAll();
        // Benefits:
        // - Database optimizations
        // - No flush to database
        // - Faster performance
    }
}
```

### N+1 Queries Problem

```java
/**
 * ❌ PROBLEM: N+1 Queries
 */
@Entity
public class User {
    @Id
    private Long id;
    private String name;
    
    @OneToMany(mappedBy = "user")
    private List<Order> orders;  // Lazy loaded by default
}

@Service
public class UserService {
    
    public List<UserDTO> getAllUsers() {
        List<User> users = userRepository.findAll();  // 1 query
        
        return users.stream()
            .map(user -> {
                UserDTO dto = new UserDTO();
                dto.setName(user.getName());
                dto.setOrderCount(user.getOrders().size());  // N queries!
                return dto;
            })
            .collect(Collectors.toList());
        
        // If 100 users: 1 + 100 = 101 queries!
        // SELECT * FROM users;
        // SELECT * FROM orders WHERE user_id = 1;
        // SELECT * FROM orders WHERE user_id = 2;
        // ... (100 times)
    }
}

/**
 * ✅ SOLUTION 1: JOIN FETCH
 */
@Repository
public interface UserRepository extends JpaRepository<User, Long> {
    
    @Query("SELECT u FROM User u LEFT JOIN FETCH u.orders")
    List<User> findAllWithOrders();  // ✅ Single query with join
    
    // Or using EntityGraph
    @EntityGraph(attributePaths = "orders")
    List<User> findAll();
}

/**
 * ✅ SOLUTION 2: @EntityGraph
 */
@Entity
@NamedEntityGraph(
    name = "User.withOrders",
    attributeNodes = @NamedAttributeNode("orders")
)
public class User {
    // ...
}

@Repository
public interface UserRepository extends JpaRepository<User, Long> {
    
    @EntityGraph("User.withOrders")
    List<User> findAll();
}

/**
 * ✅ SOLUTION 3: Batch Fetching
 */
@Entity
public class User {
    @Id
    private Long id;
    
    @OneToMany(mappedBy = "user")
    @BatchSize(size = 10)  // Fetch 10 at a time
    private List<Order> orders;
}

// If 100 users: 1 + 10 = 11 queries (instead of 101)
// SELECT * FROM users;
// SELECT * FROM orders WHERE user_id IN (1,2,3,...,10);
// SELECT * FROM orders WHERE user_id IN (11,12,13,...,20);
// ... (10 times)

/**
 * ✅ SOLUTION 4: DTO Projection
 */
@Repository
public interface UserRepository extends JpaRepository<User, Long> {
    
    @Query("SELECT new com.example.dto.UserDTO(u.name, COUNT(o)) " +
           "FROM User u LEFT JOIN u.orders o " +
           "GROUP BY u.id, u.name")
    List<UserDTO> findAllWithOrderCount();  // ✅ Single query, only needed data
}

/**
 * Detecting N+1 Queries
 */
// application.properties
spring.jpa.show-sql=true
spring.jpa.properties.hibernate.format_sql=true

// Or use hibernate-statistics
spring.jpa.properties.hibernate.generate_statistics=true

// Or datasource-proxy
// <dependency>
//   <groupId>net.ttddyy</groupId>
//   <artifactId>datasource-proxy</artifactId>
// </dependency>
```

### Circular Dependencies

```java
/**
 * ❌ PROBLEM: Circular Dependencies
 */
@Service
public class UserService {
    
    @Autowired
    private OrderService orderService;  // Depends on OrderService
    
    public void createUser(User user) {
        userRepository.save(user);
        orderService.createWelcomeOrder(user);
    }
}

@Service
public class OrderService {
    
    @Autowired
    private UserService userService;  // Depends on UserService
    
    public void createOrder(Order order) {
        User user = userService.findById(order.getUserId());
        // Process order
    }
}

// Error on startup:
// The dependencies of some of the beans in the application context form a cycle:
// ┌─────┐
// |  userService defined in file [UserService.class]
// ↑     ↓
// |  orderService defined in file [OrderService.class]
// └─────┘

/**
 * ✅ SOLUTION 1: Refactor (Best)
 */
// Extract common logic to separate service
@Service
public class UserOrderService {
    
    private final UserRepository userRepository;
    private final OrderRepository orderRepository;
    
    public void createUserWithWelcomeOrder(User user) {
        userRepository.save(user);
        
        Order welcomeOrder = new Order();
        welcomeOrder.setUser(user);
        orderRepository.save(welcomeOrder);
    }
}

/**
 * ✅ SOLUTION 2: @Lazy (Quick fix, not ideal)
 */
@Service
public class UserService {
    
    private final OrderService orderService;
    
    public UserService(@Lazy OrderService orderService) {
        this.orderService = orderService;
    }
}

/**
 * ✅ SOLUTION 3: Use events (Decoupling)
 */
@Service
@RequiredArgsConstructor
public class UserService {
    
    private final ApplicationEventPublisher eventPublisher;
    private final UserRepository userRepository;
    
    public void createUser(User user) {
        userRepository.save(user);
        
        // Publish event instead of calling OrderService
        eventPublisher.publishEvent(new UserCreatedEvent(user));
    }
}

@Service
@RequiredArgsConstructor
public class OrderService {
    
    private final OrderRepository orderRepository;
    
    @EventListener
    public void onUserCreated(UserCreatedEvent event) {
        // No dependency on UserService
        Order welcomeOrder = new Order();
        welcomeOrder.setUser(event.getUser());
        orderRepository.save(welcomeOrder);
    }
}

/**
 * ✅ SOLUTION 4: Setter injection (Last resort)
 */
@Service
public class UserService {
    
    @Autowired
    private OrderService orderService;
    
    // Constructor injection for other dependencies
}
```

### Memory Leaks

```java
/**
 * ❌ PROBLEM 1: Prototype Beans in Singleton
 */
@Component
public class SingletonService {
    
    @Autowired
    @Scope("prototype")  // ❌ Injected once, never refreshed!
    private PrototypeBean prototypeBean;
    
    public void doSomething() {
        prototypeBean.process();
        // Always uses same instance, defeats purpose of prototype
    }
}

/**
 * ✅ SOLUTION 1a: ObjectProvider
 */
@Component
@RequiredArgsConstructor
public class SingletonService {
    
    private final ObjectProvider<PrototypeBean> prototypeBeanProvider;
    
    public void doSomething() {
        PrototypeBean bean = prototypeBeanProvider.getObject();  // ✅ New instance
        bean.process();
    }
}

/**
 * ✅ SOLUTION 1b: ApplicationContext lookup
 */
@Component
@RequiredArgsConstructor
public class SingletonService {
    
    private final ApplicationContext context;
    
    public void doSomething() {
        PrototypeBean bean = context.getBean(PrototypeBean.class);  // ✅ New instance
        bean.process();
    }
}

/**
 * ❌ PROBLEM 2: Event Listeners Never Unregistered
 */
@Component
public class UserActivityTracker {
    
    @Autowired
    private ApplicationEventPublisher publisher;
    
    private List<String> activities = new ArrayList<>();  // LEAK!
    
    @EventListener
    public void onUserActivity(UserActivityEvent event) {
        activities.add(event.getActivity());
        // List grows forever, never cleared
    }
}

/**
 * ✅ SOLUTION 2: Implement DisposableBean
 */
@Component
public class UserActivityTracker implements DisposableBean {
    
    private List<String> activities = new ArrayList<>();
    
    @EventListener
    public void onUserActivity(UserActivityEvent event) {
        activities.add(event.getActivity());
    }
    
    @Override
    public void destroy() {
        activities.clear();  // ✅ Cleanup on bean destruction
    }
}

/**
 * ❌ PROBLEM 3: Static Collections
 */
public class UserCache {
    
    private static final Map<Long, User> cache = new HashMap<>();  // LEAK!
    
    public static void cacheUser(User user) {
        cache.put(user.getId(), user);  // Never removed, grows forever
    }
}

/**
 * ✅ SOLUTION 3a: Bounded cache
 */
@Component
public class UserCache {
    
    private final Cache<Long, User> cache = Caffeine.newBuilder()
        .maximumSize(1000)  // ✅ Max 1000 entries
        .expireAfterWrite(10, TimeUnit.MINUTES)  // ✅ Auto-expire
        .build();
}

/**
 * ✅ SOLUTION 3b: WeakHashMap
 */
public class UserCache {
    
    private static final Map<Long, User> cache = 
        Collections.synchronizedMap(new WeakHashMap<>());
    // ✅ Entries garbage collected when no strong references
}

/**
 * ❌ PROBLEM 4: ThreadLocal Not Cleaned
 */
@Component
public class RequestContext {
    
    private static final ThreadLocal<User> currentUser = new ThreadLocal<>();
    
    public void setCurrentUser(User user) {
        currentUser.set(user);  // LEAK in thread pool!
        // Thread reused → ThreadLocal value remains
    }
}

/**
 * ✅ SOLUTION 4: Always clean ThreadLocal
 */
@Component
public class RequestInterceptor implements HandlerInterceptor {
    
    private static final ThreadLocal<User> currentUser = new ThreadLocal<>();
    
    @Override
    public boolean preHandle(HttpServletRequest request, 
                            HttpServletResponse response, 
                            Object handler) {
        currentUser.set(extractUser(request));
        return true;
    }
    
    @Override
    public void afterCompletion(HttpServletRequest request,
                               HttpServletResponse response,
                               Object handler,
                               Exception ex) {
        currentUser.remove();  // ✅ ALWAYS clean up!
    }
}
```

### Not Closing Resources

```java
/**
 * ❌ PROBLEM: Resources Not Closed
 */
@Service
public class FileService {
    
    public void processFile(String path) throws IOException {
        FileInputStream fis = new FileInputStream(path);  // LEAK!
        BufferedReader reader = new BufferedReader(new InputStreamReader(fis));
        
        String line = reader.readLine();
        // If exception occurs, stream never closed
        // File handle remains open
    }
}

/**
 * ✅ SOLUTION: Try-with-resources
 */
@Service
public class FileService {
    
    public void processFile(String path) throws IOException {
        try (FileInputStream fis = new FileInputStream(path);
             BufferedReader reader = new BufferedReader(new InputStreamReader(fis))) {
            
            String line = reader.readLine();
            // ✅ Automatically closed, even on exception
        }
    }
}

/**
 * ❌ PROBLEM: Database Connections Not Closed
 */
@Service
public class ReportService {
    
    @Autowired
    private DataSource dataSource;
    
    public void generateReport() throws SQLException {
        Connection conn = dataSource.getConnection();  // LEAK!
        Statement stmt = conn.createStatement();
        ResultSet rs = stmt.executeQuery("SELECT * FROM users");
        
        // Process results
        // If exception, connection never closed → pool exhausted
    }
}

/**
 * ✅ SOLUTION: Try-with-resources
 */
@Service
public class ReportService {
    
    @Autowired
    private DataSource dataSource;
    
    public void generateReport() throws SQLException {
        try (Connection conn = dataSource.getConnection();
             Statement stmt = conn.createStatement();
             ResultSet rs = stmt.executeQuery("SELECT * FROM users")) {
            
            // Process results
            // ✅ All resources automatically closed
        }
    }
}

/**
 * ✅ BETTER: Use Spring's JdbcTemplate (handles resources)
 */
@Service
@RequiredArgsConstructor
public class ReportService {
    
    private final JdbcTemplate jdbcTemplate;
    
    public void generateReport() {
        jdbcTemplate.query("SELECT * FROM users", rs -> {
            // Process results
            // Spring handles connection management
        });
    }
}

/**
 * ❌ PROBLEM: HTTP Connections Not Closed
 */
@Service
public class ExternalApiService {
    
    public String callApi() throws IOException {
        URL url = new URL("https://api.example.com/data");
        HttpURLConnection conn = (HttpURLConnection) url.openConnection();  // LEAK!
        
        InputStream is = conn.getInputStream();
        // If exception, connection never closed
        
        return new String(is.readAllBytes());
    }
}

/**
 * ✅ SOLUTION 1: Try-with-resources
 */
@Service
public class ExternalApiService {
    
    public String callApi() throws IOException {
        URL url = new URL("https://api.example.com/data");
        HttpURLConnection conn = (HttpURLConnection) url.openConnection();
        
        try (InputStream is = conn.getInputStream()) {
            return new String(is.readAllBytes());
        } finally {
            conn.disconnect();  // ✅ Explicitly disconnect
        }
    }
}

/**
 * ✅ SOLUTION 2: Use RestTemplate (handles resources)
 */
@Service
@RequiredArgsConstructor
public class ExternalApiService {
    
    private final RestTemplate restTemplate;
    
    public String callApi() {
        return restTemplate.getForObject(
            "https://api.example.com/data", 
            String.class
        );
        // Spring handles connection management
    }
}
```

## 2. Best Practices

### Constructor Injection (Immutability)

```java
/**
 * ✅ BEST PRACTICE: Constructor Injection with final fields
 */
@Service
public class UserService {
    
    private final UserRepository userRepository;
    private final EmailService emailService;
    private final ValidationService validationService;
    
    // Benefits:
    // ✅ Immutable (final fields can't be reassigned)
    // ✅ Thread-safe by design
    // ✅ All dependencies required (NPE impossible)
    // ✅ Easy to test (no Spring context needed)
    // ✅ Dependencies explicit and visible
    
    public UserService(UserRepository userRepository,
                      EmailService emailService,
                      ValidationService validationService) {
        this.userRepository = userRepository;
        this.emailService = emailService;
        this.validationService = validationService;
    }
    
    public User createUser(UserRequest request) {
        // userRepository can't be null
        // emailService can't be reassigned
        // validationService always available
        
        validationService.validate(request);
        User user = userRepository.save(toEntity(request));
        emailService.sendWelcomeEmail(user);
        
        return user;
    }
}

/**
 * ✅ With Lombok (even cleaner)
 */
@Service
@RequiredArgsConstructor  // Generates constructor for final fields
public class UserService {
    
    private final UserRepository userRepository;
    private final EmailService emailService;
    private final ValidationService validationService;
    
    // Constructor automatically generated
    
    public User createUser(UserRequest request) {
        validationService.validate(request);
        User user = userRepository.save(toEntity(request));
        emailService.sendWelcomeEmail(user);
        return user;
    }
}

/**
 * ✅ Value Objects (Immutable DTOs)
 */
@Value  // Lombok: generates constructor, getters, equals, hashCode, toString
public class UserDTO {
    Long id;
    String email;
    String name;
    LocalDateTime createdAt;
    
    // All fields final and private
    // No setters
    // Immutable by design
}

/**
 * ✅ Builder Pattern for Complex Objects
 */
@Builder
@Value
public class OrderRequest {
    Long userId;
    Long productId;
    Integer quantity;
    BigDecimal price;
    String currency;
    PaymentMethod paymentMethod;
    ShippingAddress shippingAddress;

    .shippingAddress(address)
    .build();
```

### Single Responsibility Principle

```java
/**
 * ❌ BAD: God Service (multiple responsibilities)
 */
@Service
public class UserService {
    
    // User management
    public User createUser(UserRequest request) { }
    public User updateUser(Long id, UserRequest request) { }
    public void deleteUser(Long id) { }
    
    // Email operations
    public void sendWelcomeEmail(User user) { }
    public void sendPasswordResetEmail(User user) { }
    
    // Validation
    public void validateEmail(String email) { }
    public void validatePassword(String password) { }
    
    // Payment processing
    public void processPayment(User user, Payment payment) { }
    public void refundPayment(Payment payment) { }
    
    // Reporting
    public Report generateUserReport(Long userId) { }
    public Report generateActivityReport() { }
    
    // Too many responsibilities!
    // Hard to test, maintain, understand
}

/**
 * ✅ GOOD: Single Responsibility (one reason to change)
 */
@Service
@RequiredArgsConstructor
public class UserService {
    
    private final UserRepository userRepository;
    
    // ONLY user CRUD operations
    public User createUser(User user) {
        return userRepository.save(user);
    }
    
    public User updateUser(Long id, User user) {
        User existing = userRepository.findById(id)
            .orElseThrow(() -> new UserNotFoundException(id));
        
        existing.setEmail(user.getEmail());
        existing.setName(user.getName());
        
        return userRepository.save(existing);
    }
    
    public void deleteUser(Long id) {
        userRepository.deleteById(id);
    }
}

@Service
@RequiredArgsConstructor
public class EmailService {
    
    private final JavaMailSender mailSender;
    
    // ONLY email operations
    public void sendWelcomeEmail(User user) {
        sendEmail(user.getEmail(), "Welcome!", "Welcome to our platform!");
    }
    
    public void sendPasswordResetEmail(User user, String resetToken) {
        String link = "https://example.com/reset?token=" + resetToken;
        sendEmail(user.getEmail(), "Password Reset", "Reset link: " + link);
    }
```

```java
public class UserNotFoundException extends RuntimeException {

    }

public class ValidationException extends RuntimeException {
    private final Map<String, String> errors;
    
    public ValidationException(String message, Map<String, String> errors) {
        super(message);
        this.errors = errors;
    }
    
    public Map<String, String> getErrors() {
        return errors;
    }
}

/**
 * ✅ Global Exception Handler
 */
@RestControllerAdvice
@Slf4j
public class GlobalExceptionHandler {
    
    @ExceptionHandler(UserNotFoundException.class)
    @ResponseStatus(HttpStatus.NOT_FOUND)
    public ErrorResponse handleUserNotFound(UserNotFoundException ex) {
        log.warn("User not found: {}", ex.getMessage());
        
        return ErrorResponse.builder()
            .timestamp(LocalDateTime.now())
            .status(HttpStatus.NOT_FOUND.value())
            .error("Not Found")
            .message(ex.getMessage())
            .build();
    }
    
    @ExceptionHandler(ValidationException.class)
    @ResponseStatus(HttpStatus.BAD_REQUEST)
    public ErrorResponse handleValidation(ValidationException ex) {
        log.warn("Validation error: {}", ex.getMessage());
        
        return ErrorResponse.builder()
            .timestamp(LocalDateTime.now())
            .status(HttpStatus.BAD_REQUEST.value())
            .error("Validation Error")
            .message(ex.getMessage())
            .errors(ex.getErrors())
            .build();
    }
    
    @ExceptionHandler(Exception.class)
    @ResponseStatus(HttpStatus.INTERNAL_SERVER_ERROR)
    public ErrorResponse handleGenericException(Exception ex) {
        log.error("Unexpected error", ex);
        
        return ErrorResponse.builder()
            .timestamp(LocalDateTime.now())
            .status(HttpStatus.INTERNAL_SERVER_ERROR.value())
            .error("Internal Server Error")
            .message("An unexpected error occurred")
            .build();
    }
}

@Builder
public class ErrorResponse {
    private LocalDateTime timestamp;
    private int status;
    private String error;
    private String message;
    private Map<String, String> errors;
```
