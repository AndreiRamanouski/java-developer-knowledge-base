# Java Cryptography Architecture (JCA)

## Overview

Java Cryptography Architecture provides a framework for secure cryptographic operations—essential for protecting data, authenticating users, and ensuring system security. This guide covers seven critical JCA concepts: MessageDigest for Hashing (creating fingerprints of data—SHA-256/SHA-3 for integrity checks, NEVER use MD5/SHA-1 for security, use BCrypt/Argon2 for passwords not plain SHA, collision-resistant and one-way), Cipher for Encryption/Decryption (protecting confidentiality—AES-256-GCM for symmetric encryption, RSA-2048+ for asymmetric, ALWAYS use authenticated encryption like GCM, never use ECB mode, IV must be random), KeyGenerator and SecureRandom (generating cryptographic keys—SecureRandom for unpredictable randomness, KeyGenerator for symmetric keys, KeyPairGenerator for RSA/EC keys, key size matters: AES-256, RSA-2048+, EC-256+), Digital Signatures (non-repudiation and authentication—sign with private key, verify with public key, SHA-256withRSA or SHA-256withECDSA, proves message origin and integrity), Certificate Management (PKI infrastructure—X.509 certificates, KeyStore for private keys, TrustStore for CA certificates, certificate validation chains), Provider Architecture (pluggable crypto implementations—default SunJCE provider, BouncyCastle for advanced features, providers implement algorithms, allows FIPS compliance), and Best Practices (security guidelines—use modern algorithms, proper key management, secure random, authenticated encryption, regular updates, never roll your own crypto). The key insight: cryptography is easy to use wrong—small mistakes like reusing IVs, weak keys, or unauthenticated encryption can completely break security. Common mistakes: using MD5/SHA-1 (broken), ECB mode (insecure pattern leakage), weak keys (<2048 RSA, <128 AES), hardcoded keys (never!), reusing IVs (breaks security), not using authenticated encryption (vulnerable to tampering), storing keys in code (security breach), using Random instead of SecureRandom. The relationship between concepts: SecureRandom generates randomness → KeyGenerator creates keys → Cipher encrypts with keys → MessageDigest creates hashes → Signatures prove authenticity → Certificates validate identities → Providers implement all operations. Real-world security strategy: use AES-256-GCM for data encryption, RSA-2048+ or ECDSA for signatures, BCrypt for password hashing, SecureRandom for IVs and salts, store keys in KeyStore/HSM, use TLS for transport, validate certificates, rotate keys regularly, follow OWASP guidelines.

## 1. MessageDigest for Hashing

### Creating Data Fingerprints

```java
/**
 * MESSAGE DIGEST: One-way hash functions
 * 
 * Properties:
 * - Deterministic (same input → same hash)
 * - Fast to compute
 * - Collision-resistant (hard to find two inputs with same hash)
 * - One-way (can't reverse hash to get input)
 * - Avalanche effect (small input change → completely different hash)
 */

import java.security.MessageDigest;
import java.security.NoSuchAlgorithmException;
import javax.xml.bind.DatatypeConverter;

// ========================================
// SHA-256 HASHING (RECOMMENDED)
// ========================================

public class HashingExamples {
    
    public String sha256Hash(String input) throws NoSuchAlgorithmException {
        MessageDigest digest = MessageDigest.getInstance("SHA-256");
        
        byte[] hashBytes = digest.digest(input.getBytes(StandardCharsets.UTF_8));
        
        // Convert to hex string
        return bytesToHex(hashBytes);
    }
    
    private String bytesToHex(byte[] bytes) {
        StringBuilder result = new StringBuilder();
        for (byte b : bytes) {
            result.append(String.format("%02x", b));
        }
        return result;
    }
    
    // Example:
    // Input: "Hello, World!"
    // SHA-256: dffd6021bb2bd5b0af676290809ec3a53191dd81c7f70a4b28688a362182986f
    
    // ========================================
    // SHA-3 HASHING (MODERN)
    // ========================================
    
    public String sha3Hash(String input) throws NoSuchAlgorithmException {
        MessageDigest digest = MessageDigest.getInstance("SHA3-256");
        byte[] hashBytes = digest.digest(input.getBytes(StandardCharsets.UTF_8));
        return bytesToHex(hashBytes);
    }
    
    // ========================================
    // FILE HASHING (INTEGRITY CHECK)
    // ========================================
    
    public String hashFile(File file) throws Exception {
        MessageDigest digest = MessageDigest.getInstance("SHA-256");
        
        try (FileInputStream fis = new FileInputStream(file);
             DigestInputStream dis = new DigestInputStream(fis, digest)) {
            
            byte[] buffer = new byte[8192];
            while (dis.read(buffer) != -1) {
                // Reading updates the digest
            }
        }
        
        byte[] hashBytes = digest.digest();
        return bytesToHex(hashBytes);
    }
    
    // ========================================
    // INCREMENTAL HASHING
    // ========================================
    
    public String hashLargeData(InputStream input) throws Exception {
        MessageDigest digest = MessageDigest.getInstance("SHA-256");
        
        byte[] buffer = new byte[8192];
        int bytesRead;
        
        while ((bytesRead = input.read(buffer)) != -1) {
            digest.update(buffer, 0, bytesRead);
        }
        
        byte[] hashBytes = digest.digest();
        return bytesToHex(hashBytes);
    }
}

// ⚠️ WARNING: PASSWORD HASHING
// ========================================

/**
 * ❌ WRONG: Plain SHA-256 for passwords
 * 
 * Problems:
 * - Too fast (enables brute force)
 * - No salt (vulnerable to rainbow tables)
 * - Same password → same hash
 */
public class WrongPasswordHashing {
    
    // ❌ NEVER DO THIS!
    public String hashPasswordWrong(String password) throws Exception {
        MessageDigest digest = MessageDigest.getInstance("SHA-256");
        return bytesToHex(digest.digest(password.getBytes()));
    }
    
    // Problems:
    // 1. "password123" always hashes to same value
    // 2. Attacker can precompute hashes (rainbow table)
    // 3. GPU can try billions of hashes per second
    // 4. No salt makes dictionary attacks easy
}

/**
 * ✅ CORRECT: BCrypt or Argon2 for passwords
 */
import org.springframework.security.crypto.bcrypt.BCryptPasswordEncoder;
import org.springframework.security.crypto.argon2.Argon2PasswordEncoder;

public class CorrectPasswordHashing {
    
    // ✅ USE BCRYPT (recommended)
    public String hashPasswordBCrypt(String password) {
        BCryptPasswordEncoder encoder = new BCryptPasswordEncoder(12);  // Work factor
        return encoder.encode(password);
    }
    
    // Output: $2a$12$R9h/cIPz0gi.URNNX3kh2OPST9/PgBkqquzi.Ss7KIUgO2t0jWMUW
    //         |  |  |                                          |
    //         |  |  salt (random)                              hash
    //         |  work factor (2^12 iterations)
    //         BCrypt identifier
    
    public boolean verifyPasswordBCrypt(String password, String hash) {
        BCryptPasswordEncoder encoder = new BCryptPasswordEncoder();
        return encoder.matches(password, hash);
    }
    
    // ✅ USE ARGON2 (modern, preferred)
    public String hashPasswordArgon2(String password) {
        Argon2PasswordEncoder encoder = Argon2PasswordEncoder.defaultsForSpringSecurity_v5_8();
        return encoder.encode(password);
    }
    
    // Why BCrypt/Argon2?
    // - Slow by design (prevents brute force)
    // - Built-in salt (prevents rainbow tables)
    // - Adaptive (can increase work factor over time)
    // - Same password → different hash each time
}

// ========================================
// HASH COMPARISON (TIMING-SAFE)
// ========================================

import java.security.MessageDigest;

public class SecureHashComparison {
    
    // ❌ WRONG: Vulnerable to timing attacks
    public boolean compareHashesWrong(String hash1, String hash2) {
        return hash1.equals(hash2);  // Stops at first difference!
    }
    
    // ✅ CORRECT: Constant-time comparison
    public boolean compareHashesSafe(byte[] hash1, byte[] hash2) {
        return MessageDigest.isEqual(hash1, hash2);
    }
    
    // Why constant-time?
    // - equals() returns early on mismatch
    // - Attacker can measure timing differences
    // - Can recover hash byte-by-byte
    // - isEqual() always compares all bytes
}

/**
 * ✅ SECURE (Use these):
 * - SHA-256, SHA-384, SHA-512 (SHA-2 family)
 * - SHA3-256, SHA3-384, SHA3-512
 * - BLAKE2b, BLAKE3 (via BouncyCastle)
 * 
 * ⚠️ LEGACY (Avoid):
 * - SHA-1 (collision attacks exist, deprecated)
 * 
 * ❌ BROKEN (Never use):
 * - MD5 (completely broken, trivial collisions)
 * 
 * FOR PASSWORDS (Special purpose):
 * - BCrypt (recommended)
 * - Argon2 (modern, preferred)
 * - PBKDF2 (acceptable)
 * - scrypt (acceptable)
 * 
 * NEVER use plain SHA-256 for passwords!
 */

// ========================================
// HMAC (KEYED HASHING)
// ========================================

import javax.crypto.Mac;
import javax.crypto.spec.SecretKeySpec;

public class HmacExamples {
    
    public String hmacSha256(String message, String secret) throws Exception {
        Mac mac = Mac.getInstance("HmacSHA256");
        SecretKeySpec secretKey = new SecretKeySpec(
            secret.getBytes(StandardCharsets.UTF_8), 
            "HmacSHA256"
        );
        mac.init(secretKey);
        
        byte[] hmacBytes = mac.doFinal(message.getBytes(StandardCharsets.UTF_8));
        return bytesToHex(hmacBytes);
    }
    
    // Use cases:
    // - API authentication (sign requests)
    // - Message authentication codes
    // - JWT token signing
    // - Webhook signatures
    
    // Example: GitHub webhook
    // X-Hub-Signature-256: sha256=<HMAC>
    public boolean verifyWebhook(String payload, String signature, String secret) 
            throws Exception {
        String computed = "sha256=" + hmacSha256(payload, secret);
        return MessageDigest.isEqual(
            computed.getBytes(), 
            signature.getBytes()
        );
    }
}
```

## 2. Cipher for Encryption/Decryption

### Protecting Data Confidentiality

```java
/**
 * CIPHER: Encryption and decryption
 * 
 * Symmetric (same key):
 * - AES (Advanced Encryption Standard) - RECOMMENDED
 * - ChaCha20 (stream cipher)
 * 
 * Asymmetric (public/private key):
 * - RSA (for key exchange, small data)
 * - ECDSA (elliptic curve, smaller keys)
 */

import javax.crypto.Cipher;
import javax.crypto.KeyGenerator;
import javax.crypto.SecretKey;
import javax.crypto.spec.GCMParameterSpec;
import javax.crypto.spec.SecretKeySpec;
import java.security.SecureRandom;

// ========================================
// AES-256-GCM ENCRYPTION (RECOMMENDED)
// ========================================

public class AesGcmEncryption {
    
    private static final int GCM_TAG_LENGTH = 128;  // bits
    private static final int GCM_IV_LENGTH = 12;     // bytes
    
    /**
     * ✅ CORRECT: AES-256-GCM
     * 
     * GCM = Galois/Counter Mode
     * - Authenticated encryption (detects tampering)
     * - Fast and secure
     * - Industry standard
     */
    public byte[] encrypt(String plaintext, SecretKey key) throws Exception {
        // Generate random IV (MUST be unique per encryption!)
        byte[] iv = new byte[GCM_IV_LENGTH];
        SecureRandom random = new SecureRandom();
        random.nextBytes(iv);
        
        // Initialize cipher
        Cipher cipher = Cipher.getInstance("AES/GCM/NoPadding");
        GCMParameterSpec spec = new GCMParameterSpec(GCM_TAG_LENGTH, iv);
        cipher.init(Cipher.ENCRYPT_MODE, key, spec);
        
        // Encrypt
        byte[] ciphertext = cipher.doFinal(plaintext.getBytes(StandardCharsets.UTF_8));
        
        // Prepend IV to ciphertext
        byte[] encrypted = new byte[iv.length + ciphertext.length];
        System.arraycopy(iv, 0, encrypted, 0, iv.length);
        System.arraycopy(ciphertext, 0, encrypted, iv.length, ciphertext.length);
        
        return encrypted;
    }
    
    public String decrypt(byte[] encrypted, SecretKey key) throws Exception {
        // Extract IV
        byte[] iv = new byte[GCM_IV_LENGTH];
        System.arraycopy(encrypted, 0, iv, 0, iv.length);
        
        // Extract ciphertext
        byte[] ciphertext = new byte[encrypted.length - GCM_IV_LENGTH];
        System.arraycopy(encrypted, GCM_IV_LENGTH, ciphertext, 0, ciphertext.length);
        
        // Initialize cipher
        Cipher cipher = Cipher.getInstance("AES/GCM/NoPadding");
        GCMParameterSpec spec = new GCMParameterSpec(GCM_TAG_LENGTH, iv);
        cipher.init(Cipher.DECRYPT_MODE, key, spec);
        
        // Decrypt
        byte[] plaintext = cipher.doFinal(ciphertext);
        return new String(plaintext, StandardCharsets.UTF_8);
    }
    
    // Example usage:
    public void example() throws Exception {
        // Generate key
        KeyGenerator keyGen = KeyGenerator.getInstance("AES");
        keyGen.init(256);  // AES-256
        SecretKey key = keyGen.generateKey();
        
        // Encrypt
        String plaintext = "Secret message";
        byte[] encrypted = encrypt(plaintext, key);
        
        // Decrypt
        String decrypted = decrypt(encrypted, key);
        
        System.out.println("Original: " + plaintext);
        System.out.println("Decrypted: " + decrypted);
    }
}

// ========================================

public class EncryptionMistakes {
    
    // ❌ MISTAKE 1: ECB Mode
    public void ecbModeWrong() throws Exception {
        Cipher cipher = Cipher.getInstance("AES/ECB/PKCS5Padding");  // ❌ NEVER USE!
        
        // Problem: Same plaintext block → same ciphertext block
        // Leaks patterns in data
        // Example: Encrypted image shows original shapes
    }
    
    // ❌ MISTAKE 2: Reusing IV
    private byte[] staticIV = new byte[16];  // ❌ NEVER DO THIS!
    
    public void reusingIV() throws Exception {
        Cipher cipher = Cipher.getInstance("AES/CBC/PKCS5Padding");
        cipher.init(Cipher.ENCRYPT_MODE, key, new IvParameterSpec(staticIV));
        
        // Problem: Reusing IV with same key breaks security
        // Attacker can XOR two ciphertexts to get XOR of plaintexts
        // ALWAYS generate random IV per encryption!
    }
    
    // ❌ MISTAKE 3: No Authentication
    public void noAuthenticationWrong() throws Exception {
        Cipher cipher = Cipher.getInstance("AES/CBC/PKCS5Padding");
        
        // Problem: Attacker can modify ciphertext
        // Padding oracle attacks
        // Bit-flipping attacks
        // 
        // ALWAYS use authenticated encryption (GCM)!
    }
    
    // ❌ MISTAKE 4: Weak Key
    public void weakKeyWrong() throws Exception {
        KeyGenerator keyGen = KeyGenerator.getInstance("AES");
        keyGen.init(128);  // ⚠️ Minimum acceptable
        
        // Better: Use 256-bit keys
        keyGen.init(256);  // ✅ Recommended
    }
}

// ========================================
// RSA ENCRYPTION (ASYMMETRIC)
// ========================================

import java.security.KeyPair;
import java.security.KeyPairGenerator;
import java.security.PrivateKey;
import java.security.PublicKey;

public class RsaEncryption {
    
    /**
     * RSA for small data or key exchange
     * 
     * ⚠️ Don't use RSA for bulk encryption (slow)
     * ✅ Use for encrypting AES keys (hybrid encryption)
     */
    public KeyPair generateKeyPair() throws Exception {
        KeyPairGenerator keyGen = KeyPairGenerator.getInstance("RSA");
        keyGen.initialize(2048);  // Minimum 2048-bit
        return keyGen.generateKeyPair();
    }
    
    public byte[] encrypt(String plaintext, PublicKey publicKey) throws Exception {
        Cipher cipher = Cipher.getInstance("RSA/ECB/OAEPWithSHA-256AndMGF1Padding");
        cipher.init(Cipher.ENCRYPT_MODE, publicKey);
        return cipher.doFinal(plaintext.getBytes(StandardCharsets.UTF_8));
    }
    
    public String decrypt(byte[] ciphertext, PrivateKey privateKey) throws Exception {
        Cipher cipher = Cipher.getInstance("RSA/ECB/OAEPWithSHA-256AndMGF1Padding");
        cipher.init(Cipher.DECRYPT_MODE, privateKey);
        byte[] plaintext = cipher.doFinal(ciphertext);
        return new String(plaintext, StandardCharsets.UTF_8);
    }
    
    // ⚠️ RSA Limitations:
    // - Can only encrypt small data (< 190 bytes for 2048-bit key)
    // - Much slower than AES
    // - Use for key exchange, not bulk encryption
}

// ========================================
// HYBRID ENCRYPTION (BEST PRACTICE)
// ========================================

public class HybridEncryption {
    
    /**
     * ✅ RECOMMENDED: Hybrid encryption
     * 
     * 1. Generate random AES key
     * 2. Encrypt data with AES (fast)
     * 3. Encrypt AES key with RSA (secure key exchange)
     * 4. Send both encrypted AES key and encrypted data
     */
    public EncryptedData encrypt(String plaintext, PublicKey recipientPublicKey) 
            throws Exception {
        // Generate random AES key
        KeyGenerator keyGen = KeyGenerator.getInstance("AES");
        keyGen.init(256);
        SecretKey aesKey = keyGen.generateKey();
        
        // Encrypt data with AES
        AesGcmEncryption aes = new AesGcmEncryption();
        byte[] encryptedData = aes.encrypt(plaintext, aesKey);
        
        // Encrypt AES key with RSA
        Cipher rsaCipher = Cipher.getInstance("RSA/ECB/OAEPWithSHA-256AndMGF1Padding");
        rsaCipher.init(Cipher.ENCRYPT_MODE, recipientPublicKey);
        byte[] encryptedKey = rsaCipher.doFinal(aesKey.getEncoded());
        
        return new EncryptedData(encryptedKey, encryptedData);
    }
    
    public String decrypt(EncryptedData encrypted, PrivateKey privateKey) 
            throws Exception {
        // Decrypt AES key with RSA
        Cipher rsaCipher = Cipher.getInstance("RSA/ECB/OAEPWithSHA-256AndMGF1Padding");
        rsaCipher.init(Cipher.DECRYPT_MODE, privateKey);
        byte[] aesKeyBytes = rsaCipher.doFinal(encrypted.getEncryptedKey());
        SecretKey aesKey = new SecretKeySpec(aesKeyBytes, "AES");
        
        // Decrypt data with AES
        AesGcmEncryption aes = new AesGcmEncryption();
        return aes.decrypt(encrypted.getEncryptedData(), aesKey);
    }
    
    static class EncryptedData {
        private final byte[] encryptedKey;
        private final byte[] encryptedData;
        
        public EncryptedData(byte[] encryptedKey, byte[] encryptedData) {
            this.encryptedKey = encryptedKey;
            this.encryptedData = encryptedData;
        }
        
        public byte[] getEncryptedKey() { return encryptedKey; }
        public byte[] getEncryptedData() { return encryptedData; }
    }
}
```

## 4. Digital Signatures

### Proving Authenticity and Integrity

```java
/**
 * DIGITAL SIGNATURES
 * 
 * Purpose:
 * - Authentication (proves who sent message)
 * - Integrity (detects tampering)
 * - Non-repudiation (can't deny signing)
 * 
 * Process:
 * 1. Hash the message
 * 2. Encrypt hash with private key → signature
 * 3. Verify by decrypting with public key
 */

import java.security.*;

// ========================================
// RSA SIGNATURES
// ========================================

public class RsaSignature {
    
    public byte[] signData(String message, PrivateKey privateKey) throws Exception {
        Signature signature = Signature.getInstance("SHA256withRSA");
        signature.initSign(privateKey);
        signature.update(message.getBytes(StandardCharsets.UTF_8));
        return signature.sign();
    }
    
    public boolean verifySignature(
            String message, 
            byte[] signatureBytes, 
            PublicKey publicKey) throws Exception {
        
        Signature signature = Signature.getInstance("SHA256withRSA");
        signature.initVerify(publicKey);

public class JwtSigning {
    
    /**
     * JWT (JSON Web Token) uses signatures
     * 
     * Structure: header.payload.signature
     * 
     * Algorithms:
     * - HS256: HMAC with SHA-256 (symmetric)
     * - RS256: RSA with SHA-256 (asymmetric)
     * - ES256: ECDSA with P-256 and SHA-256 (asymmetric)
     */
    public String createJwt(String payload, PrivateKey privateKey) throws Exception {
        // Header
        String header = Base64.getUrlEncoder().withoutPadding()
            .encodeToString("{\"alg\":\"RS256\",\"typ\":\"JWT\"}".getBytes());
        
        // Payload
        String payloadEncoded = Base64.getUrlEncoder().withoutPadding()
            .encodeToString(payload.getBytes());
        
        // Sign header.payload
        String data = header + "." + payloadEncoded;
        Signature signature = Signature.getInstance("SHA256withRSA");
        signature.initSign(privateKey);
        signature.update(data.getBytes());
        byte[] signatureBytes = signature.sign();
        
        String signatureEncoded = Base64.getUrlEncoder().withoutPadding()
            .encodeToString(signatureBytes);
        
        return data + "." + signatureEncoded;
    }
}
```

```java
public class CertificateValidation {
    
    public boolean validateCertificate(
            X509Certificate certificate,
            KeyStore trustStore) throws Exception {
        
        // Create certificate path
        CertificateFactory certFactory = CertificateFactory.getInstance("X.509");
        CertPath certPath = certFactory.generateCertPath(Arrays.asList(certificate));
        
        // Create validator parameters
        PKIXParameters params = new PKIXParameters(trustStore);
        params.setRevocationEnabled(false);  // For simplicity (enable in production!)
        
        // Validate
        CertPathValidator validator = CertPathValidator.getInstance("PKIX");
        try {
            validator.validate(certPath, params);
            return true;
        } catch (Exception e) {
            return false;
        }
    }
    
    public void checkCertificateValidity(X509Certificate cert) throws Exception {
        // Check if certificate is currently valid
        cert.checkValidity();  // Throws if expired or not yet valid
        
        // Get validity period
        Date notBefore = cert.getNotBefore();
        Date notAfter = cert.getNotAfter();
        
        System.out.println("Valid from: " + notBefore);
        System.out.println("Valid until: " + notAfter);
    }
    
    public void inspectCertificate(X509Certificate cert) {
        // Subject (who owns this certificate)
        String subject = cert.getSubjectDN().getName();
        
        // Issuer (who signed this certificate)
        String issuer = cert.getIssuerDN().getName();
        
        // Serial number
        BigInteger serial = cert.getSerialNumber();
        
        System.out.println("Subject: " + subject);
        System.out.println("Issuer: " + issuer);
        System.out.println("Serial: " + serial);
    }
}
```
