# Class-File API (Preview Java 22-23)

## Overview

The Class-File API, introduced as a preview in Java 22 (JEP 457) and refined in Java 23 (JEP 466), provides a standard, built-in API for parsing, generating, and transforming Java class files. Before this API, developers had to use third-party libraries like ASM, BCEL, or Javassist for bytecode manipulation, each with their own learning curve, maintenance burden, and version compatibility issues. The Class-File API solves this by providing a JDK-integrated solution that tracks class file format changes automatically, offers a modern, composable API design, and achieves performance comparable to or better than ASM. The API is structured around three core abstractions: ClassFile (entry point for reading/writing), ClassModel/MethodModel/CodeModel (immutable representations of parsed bytecode), and builders/transforms (for generating and modifying bytecode). This guide covers the motivation for the API, core concepts like models and transforms, practical examples of parsing and generating class files, constant pool handling, real-world use cases including instrumentation and code generation, performance characteristics, and comparison with existing libraries. While still in preview, this API represents the future of bytecode manipulation in Java.


**Preview Status**: Requires `--enable-preview` flag

## 2. Core Concepts: Models and Builders

### Understanding the API Structure

```java
/**
 * CORE CONCEPTS
 * 
 * Models, Builders, and Transforms
 */

import java.lang.classfile.*;
import java.lang.classfile.attribute.*;
import java.lang.constant.*;
import java.io.IOException;
import java.nio.file.*;

public class CoreConcepts {
    
    /**
     * ClassFile - Entry point
     */
    public static void classFileEntryPoint() {
        /**
         * CLASSFILE:
         * 
         * Entry point for all operations:
         * 
         * ClassFile cf = ClassFile.of();
         * 
         * Three main operations:
         * 1. Parse: Read existing class file
         * 2. Build: Generate new class file
         * 3. Transform: Modify existing class file
         * 
         * 
         * PARSING:
         * 
         * byte[] classBytes = Files.readAllBytes(path);
         * ClassModel model = cf.parse(classBytes);
         * 
         * 
         * BUILDING:
         * 
         * byte[] classBytes = cf.build(
         *   ClassDesc.of("MyClass"),
         *   classBuilder -> {
         *     // Build class
         *   }
         * );
         * 
         * 
         * TRANSFORMING:
         * 
         * byte[] newBytes = cf.transform(
         *   model,
         *   classTransform
         * );
         */
    }
    
    /**
     * Models - Immutable representations
     */
    public static void models() {
        /**
         * MODELS:
         * 
         * Immutable representations of bytecode:
         * 
         * ClassModel:
         * - Represents entire class file
         * - Contains methods, fields, attributes
         * - Immutable
         * 
         * MethodModel:
         * - Represents a method
         * - Contains code, attributes
         * - Immutable
         * 
         * CodeModel:
         * - Represents method body
         * - Contains instructions
         * - Immutable
         * 
         * FieldModel:
         * - Represents a field
         * - Contains attributes
         * - Immutable
         * 
         * 
         * HIERARCHY:
         * 
         * ClassModel
         *   ├── MethodModel
         *   │   └── CodeModel
         *   │       └── Instructions
         *   └── FieldModel
         * 
         * 
         * All models are immutable!
         * Modifications create new models!
         */
    }
    
    /**
     * Builders - Creating bytecode
     */
    public static void builders() {
        /**
         * BUILDERS:
         * 
         * Fluent API for creating bytecode:
         * 
         * ClassBuilder:
         * - Build class elements
         * - Add methods, fields
         * - Set attributes
         * 
         * MethodBuilder:
         * - Build method
         * - Set access flags
         * - Build code
         * 
         * CodeBuilder:
         * - Build method body
         * - Add instructions
         * - Exception handlers
         * 
         * 
         * EXAMPLE:
         * 
         * cf.build(
         *   ClassDesc.of("MyClass"),
         *   classBuilder -> {
         *     classBuilder.withMethod(
         *       "myMethod",
         *       MethodTypeDesc.of(CD_void),
         *       ACC_PUBLIC,
         *       methodBuilder -> {
         *         methodBuilder.withCode(
         *           codeBuilder -> {
         *             codeBuilder.return_();
         *           }
         *         );
         *       }
         *     );
         *   }
         * );
         * 
         * Fluent, composable, type-safe!
         */
    }
    
    /**
     * Transforms - Modifying bytecode
     */
    public static void transforms() {
        /**
         * TRANSFORMS:
         * 
         * Composable transformations:
         * 
         * ClassTransform:
         * - Transform class elements
         * - Add/remove/modify methods
         * - Composable
         * 
         * MethodTransform:
         * - Transform method
         * - Modify code
         * - Composable
         * 
         * CodeTransform:
         * - Transform instructions
         * - Insert/modify/remove
         * - Composable
         * 
         * 
         * COMPOSITION:
         * 
         * ClassTransform t1 = ...;
         * ClassTransform t2 = ...;
         * ClassTransform combined = t1.andThen(t2);
         * 
         * 
         * EXAMPLE:
         * 
         * cf.transform(model, (classBuilder, classElement) -> {
         *   if (classElement instanceof MethodModel method) {
         *     // Transform method
         *     classBuilder.transformMethod(
         *       method,
         *       methodTransform
         *     );
         *   } else {
         *     // Keep element as-is
         *     classBuilder.with(classElement);
         *   }
         * });
         * 
         * Functional, composable, powerful!
         */
    }
    
    /**
     * Elements - Building blocks
     */
    public static void elements() {
        /**
         * ELEMENTS:
         * 
         * Everything is an element:
         * 
         * ClassElement:
         * - Method
         * - Field
         * - Attribute
         * - Superclass
         * - Interface
         * 
         * MethodElement:
         * - Code
         * - Attribute
         * - Access flags
         * 
         * CodeElement:
         * - Instruction
         * - Label
         * - Exception handler
         * - Local variable
         * 
         * 
         * UNIFORM INTERFACE:
         * 
         * All elements can be:
         * - Written to builder
         * - Transformed
         * - Inspected
         * 
         * builder.with(element);  // Write element
         * 
         * Simple and consistent!
         */
    }
}
```

## 3. Parsing Class Files

### Reading and Analyzing Bytecode

```java
/**
 * PARSING CLASS FILES
 * 
 * Reading and inspecting bytecode
 */

import java.lang.classfile.*;
import java.lang.classfile.instruction.*;
import java.lang.constant.*;
import java.nio.file.*;
import java.io.IOException;

public class ParsingClassFiles {
    
    /**
     * Basic parsing
     */
    public static void basicParsing() throws IOException {
        System.out.println("=== Basic Parsing ===");
        
        // Get ClassFile API
        ClassFile cf = ClassFile.of();
        
        // Read class file bytes
        byte[] bytes = Files.readAllBytes(
            Path.of("build/classes/java/main/MyClass.class")
        );
        
        // Parse into model
        ClassModel classModel = cf.parse(bytes);
        
        // Inspect class
        System.out.println("Class: " + classModel.thisClass().asInternalName());
        System.out.println("Super: " + classModel.superclass()
            .map(c -> c.asInternalName())
            .orElse("none"));
        System.out.println("Version: " + classModel.majorVersion());
        
        /**
         * PARSING:
         * 
         * ClassFile.of().parse(bytes)
         * 
         * Returns ClassModel:
         * - Immutable
         * - Lazy parsing
         * - Efficient
         */
    }
    
    /**
     * Inspecting methods
     */
    public static void inspectingMethods() throws IOException {
        System.out.println("\n=== Inspecting Methods ===");
        
        ClassFile cf = ClassFile.of();
        byte[] bytes = getClassBytes();
        ClassModel model = cf.parse(bytes);
        
        // Iterate methods
        for (MethodModel method : model.methods()) {
            System.out.println("\nMethod: " + method.methodName().stringValue());
            System.out.println("  Descriptor: " + method.methodType().stringValue());
            System.out.println("  Flags: " + method.flags().flagsMask());
            
            // Get code
            method.code().ifPresent(code -> {
                System.out.println("  Code size: " + code.codeLength());
                System.out.println("  Max stack: " + code.maxStack());
                System.out.println("  Max locals: " + code.maxLocals());
            });
        }
        
        /**
         * METHOD INSPECTION:
         * 
         * model.methods():
         * - Returns list of MethodModel
         * - Each method immutable
         * - Contains code, attributes
         */
    }
    
    /**
     * Analyzing instructions
     */
    public static void analyzingInstructions() throws IOException {
        System.out.println("\n=== Analyzing Instructions ===");
        
        ClassFile cf = ClassFile.of();
        byte[] bytes = getClassBytes();
        ClassModel model = cf.parse(bytes);
        
        // Find specific method
        MethodModel method = model.methods().stream()
            .filter(m -> m.methodName().stringValue().equals("example"))
            .findFirst()
            .orElseThrow();
        
        // Analyze code
        method.code().ifPresent(code -> {
            System.out.println("Instructions in 'example' method:");
            
            for (CodeElement element : code) {
                if (element instanceof Instruction inst) {
                    System.out.println("  " + inst);
                    
                    // Pattern match on instruction type
                    switch (inst) {
                        case InvokeInstruction invoke ->
                            System.out.println("    Calling: " + 
                                invoke.method().name().stringValue());
                        
                        case FieldInstruction field ->
                            System.out.println("    Field: " + 
                                field.field().name().stringValue());
                        
                        case LoadInstruction load ->
                            System.out.println("    Loading local: " + load.slot());
                        
                        default -> {}
                    }
                }
            }
        });
        
        /**
         * INSTRUCTION ANALYSIS:
         * 
         * code.elements():
         * - Iterate all code elements
         * - Instructions, labels, handlers
         * - Pattern matching on types
         */
    }
    
    /**
     * Finding method calls
     */
    public static void findingMethodCalls() throws IOException {
        System.out.println("\n=== Finding Method Calls ===");
        
        ClassFile cf = ClassFile.of();
        byte[] bytes = getClassBytes();
        ClassModel model = cf.parse(bytes);
        
        System.out.println("Method calls in class:");
        
        for (MethodModel method : model.methods()) {
            method.code().ifPresent(code -> {
                for (CodeElement element : code) {
                    if (element instanceof InvokeInstruction invoke) {
                        ClassDesc owner = invoke.owner().asSymbol();
                        String methodName = invoke.name().stringValue();
                        
                        System.out.printf("  %s.%s() called from %s%n",
                            owner.displayName(),
                            methodName,
                            method.methodName().stringValue()
                        );
                    }
                }
            });
        }
        
        /**
         * USE CASE:
         * 
         * Find all method calls
         * - API usage analysis
         * - Dependency tracking
         * - Dead code detection
         */
    }
    
    /**
     * Analyzing constant pool
     */
    public static void analyzingConstantPool() throws IOException {
        System.out.println("\n=== Constant Pool Analysis ===");
        
        ClassFile cf = ClassFile.of();
        byte[] bytes = getClassBytes();
        ClassModel model = cf.parse(bytes);
        
        // Access constant pool
        ConstantPool pool = model.constantPool();
        
        System.out.println("Constant pool size: " + pool.entryCount());
        
        // Iterate entries (note: 1-indexed!)
        for (int i = 1; i < pool.entryCount(); i++) {
            try {
                PoolEntry entry = pool.entryByIndex(i);
                
                switch (entry) {
                    case ClassEntry ce ->
                        System.out.println("  Class: " + ce.asInternalName());
                    
                    case Utf8Entry utf8 ->
                        System.out.println("  UTF8: " + utf8.stringValue());
                    
                    case MethodRefEntry mref ->
                        System.out.println("  Method: " + 
                            mref.owner().asInternalName() + "." +
                            mref.name().stringValue());
                    
                    default ->
                        System.out.println("  Other: " + entry.getClass().getSimpleName());
                }
            } catch (Exception e) {
                // Some indices may be invalid (long/double use 2 slots)
            }
        }
        
        /**
         * CONSTANT POOL:
         * 
         * model.constantPool():
         * - Access pool entries
         * - Type-safe API
         * - Pattern matching
         */
    }
    
    private static byte[] getClassBytes() throws IOException {
        // Return sample class bytes
        return ParsingClassFiles.class.getName()
            .getResourceAsStream(ParsingClassFiles.class.getSimpleName() + ".class")
            .readAllBytes();
    }
    
    public static void main(String[] args) throws IOException {
        basicParsing();
        inspectingMethods();
        analyzingInstructions();
        findingMethodCalls();
        analyzingConstantPool();
    }
}
```

## 4. Generating Class Files

### Creating Bytecode from Scratch

```java
/**
 * GENERATING CLASS FILES
 * 
 * Building bytecode from scratch
 */

import java.lang.classfile.*;
import java.lang.constant.*;
import java.nio.file.*;

import static java.lang.constant.ConstantDescs.*;
import static java.lang.classfile.ClassFile.*;

public class GeneratingClassFiles {
    
    /**
     * Generate simple class
     */
    public static void generateSimpleClass() throws Exception {
        System.out.println("=== Generate Simple Class ===");
        
        ClassFile cf = ClassFile.of();
        
        // Generate: public class HelloWorld {}
        byte[] bytes = cf.build(
            ClassDesc.of("HelloWorld"),
            classBuilder -> {
                classBuilder
                    .withFlags(ACC_PUBLIC)
                    .withVersion(61, 0);  // Java 17
            }
        );
        
        // Write to file
        Files.write(Path.of("HelloWorld.class"), bytes);
        System.out.println("Generated HelloWorld.class");
        
        /**
         * BUILDING:
         * 
         * ClassFile.of().build(
         *   className,
         *   builder -> { ... }
         * )
         * 
         * Returns byte[] of class file
         */
    }
    
    /**
     * Generate class with method
     */
    public static void generateClassWithMethod() throws Exception {
        System.out.println("\n=== Generate Class with Method ===");
        
        ClassFile cf = ClassFile.of();
        
        // Generate:
        // public class Greeter {
        //   public static void greet(String name) {
        //     System.out.println("Hello, " + name);
        //   }
        // }
        
        byte[] bytes = cf.build(
            ClassDesc.of("Greeter"),
            classBuilder -> {
                classBuilder.withFlags(ACC_PUBLIC);
                
                // Add method
                classBuilder.withMethod(
                    "greet",
                    MethodTypeDesc.of(CD_void, CD_String),
                    ACC_PUBLIC | ACC_STATIC,
                    methodBuilder -> {
                        methodBuilder.withCode(codeBuilder -> {
                            // System.out
                            codeBuilder
                                .getstatic(
                                    ClassDesc.of("java.lang.System"),
                                    "out",
                                    ClassDesc.of("java.io.PrintStream")
                                )
                                
                                // "Hello, "
                                .ldc("Hello, ")
                                
                                // Load parameter (name)
                                .aload(0)
                                
                                // Concatenate
                                .invokevirtual(
                                    CD_String,
                                    "concat",
                                    MethodTypeDesc.of(CD_String, CD_String)
                                )
                                
                                // Print
                                .invokevirtual(
                                    ClassDesc.of("java.io.PrintStream"),
                                    "println",
                                    MethodTypeDesc.of(CD_void, CD_String)
                                )
                                
                                // Return
                                .return_();
                        });
                    }
                );
            }
        );
        
        Files.write(Path.of("Greeter.class"), bytes);
        System.out.println("Generated Greeter.class");
        
        /**
         * METHOD BUILDING:
         * 
         * classBuilder.withMethod(
         *   name,
         *   type,
         *   flags,
         *   methodBuilder -> { ... }
         * )
         * 
         * methodBuilder.withCode(
         *   codeBuilder -> { ... }
         * )
         * 
         * Fluent API for instructions!
         */
    }
    
    /**
     * Generate class with field
     */
    public static void generateClassWithField() throws Exception {
        System.out.println("\n=== Generate Class with Field ===");
        
        ClassFile cf = ClassFile.of();
        
        // Generate:
        // public class Counter {
        //   private int count;
        //   
        //   public void increment() {
        //     count++;
        //   }
        // }
        
        byte[] bytes = cf.build(
            ClassDesc.of("Counter"),
            classBuilder -> {
                classBuilder.withFlags(ACC_PUBLIC);
                
                // Add field
                classBuilder.withField(
                    "count",
                    CD_int,
                    ACC_PRIVATE
                );
                
                // Add method
                classBuilder.withMethod(
                    "increment",
                    MethodTypeDesc.of(CD_void),
                    ACC_PUBLIC,
                    methodBuilder -> {
                        methodBuilder.withCode(codeBuilder -> {
                            codeBuilder
                                // Load this
                                .aload(0)
                                
                                // Duplicate for putfield
                                .dup()
                                
                                // Get field value
                                .getfield(
                                    ClassDesc.of("Counter"),
                                    "count",
                                    CD_int
                                )
                                
                                // Increment
                                .iconst_1()
                                .iadd()
                                
                                // Store back
                                .putfield(
                                    ClassDesc.of("Counter"),
                                    "count",
                                    CD_int
                                )
                                
                                .return_();
                        });
                    }
                );
            }
        );
        
        Files.write(Path.of("Counter.class"), bytes);
        System.out.println("Generated Counter.class");
    }
    
    /**
     * Generate class with constructor
     */
    public static void generateClassWithConstructor() throws Exception {
        System.out.println("\n=== Generate Class with Constructor ===");
        
        ClassFile cf = ClassFile.of();
        
        // Generate:
        // public class Person {
        //   private String name;
        //   
        //   public Person(String name) {
        //     super();
        //     this.name = name;
        //   }
        // }
        
        byte[] bytes = cf.build(
            ClassDesc.of("Person"),
            classBuilder -> {
                classBuilder.withFlags(ACC_PUBLIC);
                
                // Add field
                classBuilder.withField("name", CD_String, ACC_PRIVATE);
                
                // Add constructor
                classBuilder.withMethod(
                    "<init>",
                    MethodTypeDesc.of(CD_void, CD_String),
                    ACC_PUBLIC,
                    methodBuilder -> {
                        methodBuilder.withCode(codeBuilder -> {
                            codeBuilder
                                // Load this
                                .aload(0)
                                
                                // Call super()
                                .invokespecial(
                                    CD_Object,
                                    "<init>",
                                    MethodTypeDesc.of(CD_void)
                                )
                                
                                // Load this again
                                .aload(0)
                                
                                // Load parameter (name)
                                .aload(1)
                                
                                // Store field
                                .putfield(
                                    ClassDesc.of("Person"),
                                    "name",
                                    CD_String
                                )
                                
                                .return_();
                        });
                    }
                );
            }
        );
        
        Files.write(Path.of("Person.class"), bytes);
        System.out.println("Generated Person.class");
        
        /**
         * CONSTRUCTOR:
         * 
         * Method name: "<init>"
         * Must call super constructor
         * Return type: void
         */
    }
    
    public static void main(String[] args) throws Exception {
        generateSimpleClass();
        generateClassWithMethod();
        generateClassWithField();
        generateClassWithConstructor();
    }
}
```

## 5. Transforming Class Files

### Modifying Existing Bytecode

```java
/**
 * TRANSFORMING CLASS FILES
 * 
 * Modifying existing bytecode
 */

import java.lang.classfile.*;
import java.lang.classfile.instruction.*;
import java.lang.constant.*;
import java.nio.file.*;

import static java.lang.classfile.ClassFile.*;

public class TransformingClassFiles {
    
    /**
     * Add method to class
     */
    public static void addMethodToClass() throws Exception {
        System.out.println("=== Add Method to Class ===");
        
        ClassFile cf = ClassFile.of();
        
        // Read existing class
        byte[] original = Files.readAllBytes(Path.of("MyClass.class"));
        ClassModel model = cf.parse(original);
        
        // Transform: add new method
        byte[] transformed = cf.transform(model, (classBuilder, classElement) -> {
            // Keep all existing elements
            classBuilder.with(classElement);
            
            // Add new method after processing all elements
            if (classElement == model.elements().getLast()) {
                classBuilder.withMethod(
                    "addedMethod",
                    MethodTypeDesc.of(ConstantDescs.CD_void),
                    ACC_PUBLIC,
                    methodBuilder -> {
                        methodBuilder.withCode(codeBuilder -> {
                            codeBuilder.return_();
                        });
                    }
                );
            }
        });
        
        Files.write(Path.of("MyClass_transformed.class"), transformed);
        System.out.println("Added method to class");
        
        /**
         * TRANSFORMATION:
         * 
         * cf.transform(model, (builder, element) -> {
         *   // Process each element
         *   // Keep, modify, or skip
         * })
         * 
         * Returns new byte[] with changes
         */
    }
    
    /**
     * Modify method bytecode
     */
    public static void modifyMethodBytecode() throws Exception {
        System.out.println("\n=== Modify Method Bytecode ===");
        
        ClassFile cf = ClassFile.of();
        byte[] original = getClassBytes();
        ClassModel model = cf.parse(original);
        
        // Transform: add logging to all methods
        byte[] transformed = cf.transform(model, (classBuilder, classElement) -> {
            if (classElement instanceof MethodModel method) {
                // Transform method
                classBuilder.transformMethod(method, (methodBuilder, methodElement) -> {
                    if (methodElement instanceof CodeModel code) {
                        // Transform code
                        methodBuilder.withCode(codeBuilder -> {
                            // Add logging at start
                            codeBuilder
                                .getstatic(
                                    ClassDesc.of("java.lang.System"),
                                    "out",
                                    ClassDesc.of("java.io.PrintStream")
                                )
                                .ldc("Entering method: " + method.methodName().stringValue())
                                .invokevirtual(
                                    ClassDesc.of("java.io.PrintStream"),
                                    "println",
                                    MethodTypeDesc.of(ConstantDescs.CD_void, ConstantDescs.CD_String)
                                );
                            
                            // Keep original code
                            for (CodeElement ce : code) {
                                codeBuilder.with(ce);
                            }
                        });
                    } else {
                        methodBuilder.with(methodElement);
                    }
                });
            } else {
                classBuilder.with(classElement);
            }
        });
        
        System.out.println("Added logging to all methods");
        
        /**
         * METHOD TRANSFORMATION:
         * 
         * Transform method code:
         * - Insert instructions at start/end
         * - Modify existing instructions
         * - Remove instructions

        System.out.println("\n=== Replace Method Calls ===");
        
        ClassFile cf = ClassFile.of();
        byte[] original = getClassBytes();
        ClassModel model = cf.parse(original);
        
        // Transform: replace System.out.println with custom logger
        byte[] transformed = cf.transform(model, ClassTransform.transformingMethods(
            (methodBuilder, methodElement) -> {
                if (methodElement instanceof CodeModel code) {
                    methodBuilder.transformCode(code, (codeBuilder, codeElement) -> {
                        if (codeElement instanceof InvokeInstruction invoke) {
                            // Check if it's System.out.println
                            if (invoke.owner().asSymbol().equals(ClassDesc.of("java.io.PrintStream")) &&
                                invoke.name().stringValue().equals("println")) {
                                
                                // Replace with custom logger call
                                codeBuilder.invokestatic(
                                    ClassDesc.of("MyLogger"),
                                    "log",
                                    invoke.typeSymbol()
                                );
                                return;
                            }
                        }
                        
                        // Keep original element
                        codeBuilder.with(codeElement);
                    });
                } else {
                    methodBuilder.with(methodElement);
                }
            }
        ));
        
        System.out.println("Replaced System.out.println calls");
        
        /**
         * INSTRUCTION REPLACEMENT:
         * 
         * Iterate instructions
         * Replace specific calls

    /**
     * Add exception handling
     */
    public static void addExceptionHandling() throws Exception {
        System.out.println("\n=== Add Exception Handling ===");
        
        ClassFile cf = ClassFile.of();
        byte[] original = getClassBytes();
        ClassModel model = cf.parse(original);
        
        // Transform: wrap method in try-catch
        byte[] transformed = cf.transform(model, ClassTransform.transformingMethods(
            (methodBuilder, methodElement) -> {
                if (methodElement instanceof CodeModel code) {
                    methodBuilder.withCode(codeBuilder -> {
                        Label startLabel = codeBuilder.newLabel();
                        Label endLabel = codeBuilder.newLabel();
                        Label handlerLabel = codeBuilder.newLabel();
                        
                        // Try block
                        codeBuilder.labelBinding(startLabel);
                        
                        // Original code
                        for (CodeElement ce : code) {
                            codeBuilder.with(ce);
                        }
                        
                        codeBuilder.labelBinding(endLabel);
                        
                        // Jump over handler
                        Label afterHandler = codeBuilder.newLabel();
                        codeBuilder.goto_(afterHandler);
                        
                        // Exception handler
                        codeBuilder.labelBinding(handlerLabel);
                        codeBuilder
                            .astore(0)  // Store exception
                            // Log exception
                            .getstatic(
                                ClassDesc.of("java.lang.System"),
                                "out",
                                ClassDesc.of("java.io.PrintStream")
                            )
                            .ldc("Exception caught!")
                            .invokevirtual(
                                ClassDesc.of("java.io.PrintStream"),
                                "println",
                                MethodTypeDesc.of(ConstantDescs.CD_void, ConstantDescs.CD_String)
                            )
                            .return_();
                        
                        codeBuilder.labelBinding(afterHandler);
                        
                        // Add exception table entry
                        codeBuilder.exceptionCatch(
                            startLabel,
                            endLabel,
                            handlerLabel,
                            ClassDesc.of("java.lang.Exception")
                        );
                    });
                } else {
                    methodBuilder.with(methodElement);
                }
            }
        ));
        
        System.out.println("Added exception handling");
    }

        // Return sample bytes
        return new byte[0];
```
