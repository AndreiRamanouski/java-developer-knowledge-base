# Structured Concurrency (Java 23+)

## Overview

Structured Concurrency, finalized in Java 23 (JEP 462), fundamentally changes how we write concurrent code in Java by treating multiple tasks as a single unit of work with a clear lifecycle. Before structured concurrency, using ExecutorService and CompletableFuture led to common problems: thread leaks (submitted tasks outliving their parent), unclear cancellation semantics (what happens when one task fails?), complex exception handling (aggregating errors from multiple tasks), and difficulty propagating context (thread locals, security). Structured concurrency solves these with StructuredTaskScope, which ensures that all subtasks complete before the scope exits, propagates cancellation automatically, handles exceptions cleanly, and maintains parent-child relationships between threads. The API provides two built-in policies—ShutdownOnSuccess (first-to-complete wins, others cancelled) and ShutdownOnFailure (fail-fast on any error)—and allows custom policies. This guide covers the old ExecutorService problems, StructuredTaskScope basics, exception handling patterns, scope inheritance, cancellation propagation, and real-world use cases like parallel API calls, fork-join workflows, and timeout handling. Structured concurrency makes concurrent code safer, more maintainable, and easier to reason about.

## 1. The Problem with Traditional Concurrency

### Why We Needed Structured Concurrency

```java
/**
 * THE PROBLEM WITH EXECUTORSERVICE
 * 
 * Traditional concurrent programming issues
 */

import java.util.concurrent.*;
import java.time.Duration;

public class TraditionalConcurrencyProblems {
    
    /**
     * Problem 1: Thread leaks
     */
    public static void threadLeaks() {
        ExecutorService executor = Executors.newCachedThreadPool();
        
        try {
            // Submit tasks
            Future<String> future1 = executor.submit(() -> {
                Thread.sleep(5000);  // Long-running
                return "Result 1";
            });
            
            Future<String> future2 = executor.submit(() -> {
                Thread.sleep(10000);  // Even longer
                return "Result 2";
            });
            
            // Get first result with timeout
            String result = future1.get(1, TimeUnit.SECONDS);
            System.out.println(result);
            
            // BUT: future2 still running!
            // Thread leak! Task outlives scope!
            
        } catch (Exception e) {
            // Exception occurred, but tasks still running!
            e.printStackTrace();
        } finally {
            executor.shutdown();
            // Tasks may still be running...
        }
        
        /**
         * PROBLEMS:
         * 
         * ✗ Tasks outlive their "parent"
         * ✗ Thread leaks
         * ✗ Resource waste
         * ✗ Hard to track lifetime
         * ✗ No automatic cleanup
         */
    }
    
    /**
     * Problem 2: Complex cancellation
     */
    public static void complexCancellation() {
        ExecutorService executor = Executors.newCachedThreadPool();
        
        try {
            Future<String> task1 = executor.submit(() -> slowOperation("Task 1"));
            Future<String> task2 = executor.submit(() -> slowOperation("Task 2"));
            Future<String> task3 = executor.submit(() -> slowOperation("Task 3"));
            
            // Wait for first to complete
            String result = task1.get();
            
            // Manually cancel others
            task2.cancel(true);  // Must remember to cancel!
            task3.cancel(true);  // What if we forget?
            
            return result;
            
        } catch (Exception e) {
            // Need to cancel all on error
            // But which ones are still running?
            // Complex error handling!
        } finally {
            executor.shutdown();
        }
        
        /**
         * PROBLEMS:
         * 
         * ✗ Manual cancellation required
         * ✗ Easy to forget
         * ✗ No automatic cleanup
         * ✗ Error-prone
         * ✗ Boilerplate code
         */
    }
    
    private static String slowOperation(String name) throws InterruptedException {
        Thread.sleep(5000);
        return name + " completed";
    }
    
    /**
     * Problem 3: Exception aggregation
     */
    public static void exceptionAggregation() {
        ExecutorService executor = Executors.newCachedThreadPool();
        
        try {
            Future<String> task1 = executor.submit(() -> {
                throw new RuntimeException("Task 1 failed");
            });
            
            Future<String> task2 = executor.submit(() -> {
                throw new RuntimeException("Task 2 failed");
            });
            
            Future<String> task3 = executor.submit(() -> {
                return "Task 3 ok";
            });
            
            // Collect results and exceptions
            try {
                task1.get();
            } catch (Exception e1) {
                // Got first exception
                try {
                    task2.get();
                } catch (Exception e2) {
                    // Got second exception
                    // How to aggregate them?
                    // Create custom exception with both?
                    // Just throw first? Lose information!
                }
            }
            
        } finally {
            executor.shutdown();
        }
        
        /**
         * PROBLEMS:
         * 
         * ✗ Manual exception collection
         * ✗ Easy to lose exceptions
         * ✗ No standard aggregation
         * ✗ Complex error handling
         * ✗ Verbose code
         */
    }
    
    /**
     * Problem 4: Context propagation
     */
    public static void contextPropagation() {
        ThreadLocal<String> context = new ThreadLocal<>();
        context.set("Important context");
        
        ExecutorService executor = Executors.newCachedThreadPool();
        
        try {
            Future<String> task = executor.submit(() -> {
                // Context NOT propagated!
                String ctx = context.get();  // null!
                return "Context: " + ctx;
            });
            
            System.out.println(task.get());  // "Context: null"
            
            /**
             * PROBLEM:
             * 
             * ✗ ThreadLocals not inherited
             * ✗ Security context lost
             * ✗ Tracing context lost
             * ✗ Manual propagation needed
             */
            
        } catch (Exception e) {
            e.printStackTrace();
        } finally {
            executor.shutdown();
        }
    }
    
    /**
     * Problem 5: Unclear lifecycle
     */
    public static void unclearLifecycle() {
        /**
         * LIFECYCLE CONFUSION:
         * 
         * When does a task finish?
         * - When Future.get() returns?
         * - When executor shuts down?
         * - When interrupt flag is set?
         * - When task actually stops?
         * 
         * What if task ignores interrupt?
         * What if task is stuck in native code?
         * What if exception is thrown?
         * 
         * No clear parent-child relationship!
         * Tasks are "fire and forget"
         * Hard to reason about lifetime
         */
    }
    
    /**
     * What we want
     */
    public static void whatWeWant() {
        /**
         * DESIRED PROPERTIES:
         * 
         * 1. Clear lifecycle:
         *    - Tasks have parent-child relationship
         *    - Child tasks cannot outlive parent
         *    - Automatic cleanup
         * 
         * 2. Automatic cancellation:
         *    - Cancel on first success (race)
         *    - Cancel on first failure (fail-fast)
         *    - Cancel on timeout
         * 
         * 3. Exception handling:
         *    - Automatic aggregation
         *    - No lost exceptions
         *    - Clear error reporting
         * 
         * 4. Context propagation:
         *    - ThreadLocals inherited
         *    - Security context maintained
         *    - Tracing context preserved
         * 
         * 5. Simplicity:
         *    - Less boilerplate
         *    - Obvious semantics
         *    - Hard to misuse
         * 
         * 
         * Structured Concurrency provides all of this!
         */
    }
    
    public static void main(String[] args) {
```

## 2. StructuredTaskScope Basics

### The Foundation of Structured Concurrency

```java
/**
 * STRUCTUREDTASKSCOPE BASICS
 * 
 * Core concepts and usage
 */

import java.util.concurrent.*;
import java.util.concurrent.StructuredTaskScope.*;

public class StructuredTaskScopeBasics {
    
    /**
     * Basic usage
     */
    public static void basicUsage() throws InterruptedException, ExecutionException {
        System.out.println("=== Basic Usage ===");
        
        try (var scope = new StructuredTaskScope<String>()) {
            // Fork subtasks
            Subtask<String> task1 = scope.fork(() -> {
                Thread.sleep(1000);
                return "Result 1";
            });
            
            Subtask<String> task2 = scope.fork(() -> {
                Thread.sleep(2000);
                return "Result 2";
            });
            
            // Wait for all to complete
            scope.join();
            
            // Get results
            String result1 = task1.get();
            String result2 = task2.get();
            
            System.out.println(result1);
            System.out.println(result2);
            
        } // Scope closes - all tasks must be done!
        
        /**
         * KEY CONCEPTS:
         * 
         * 1. try-with-resources:
         *    - Scope has lifetime
         *    - Automatic cleanup
         * 
         * 2. fork():
         *    - Start subtask
         *    - Returns Subtask handle
         * 
         * 3. join():
         *    - Wait for all tasks
         *    - Blocks until done
         * 
         * 4. Scope closure:
         *    - All tasks MUST complete
         *    - Cannot leave scope early
         *    - Enforced by compiler!
         */
    }
    
    /**
     * Structured guarantee
     */
    public static void structuredGuarantee() {
        /**
         * STRUCTURED GUARANTEE:
         * 
         * Scope exit → All subtasks complete
         * 
         * This is enforced:
         * 
         * try (var scope = new StructuredTaskScope<>()) {
         *   scope.fork(...);
         *   scope.fork(...);
         *   
         *   // Must call join() or joinUntil()
         *   scope.join();
         *   
         * } // Cannot exit without joining!
         * 
         * 
         * Compiler error if:
         * - No join() call
         * - Early return without join()
         * - Exception without join()
         * 
         * 
         * GUARANTEE:
         * No thread leaks!
         * All tasks cleaned up!
         * Clear lifecycle!
         */
    }
    
    /**
     * Subtask handle
     */
    public static void subtaskHandle() throws InterruptedException, ExecutionException {
        System.out.println("\n=== Subtask Handle ===");
        
        try (var scope = new StructuredTaskScope<Integer>()) {
            Subtask<Integer> task = scope.fork(() -> {
                Thread.sleep(1000);
                return 42;
            });
            
            scope.join();
            
            // Subtask methods:
            System.out.println("State: " + task.state());
            
            switch (task.state()) {
                case UNAVAILABLE -> System.out.println("Not done yet");
                case SUCCESS -> System.out.println("Result: " + task.get());
                case FAILED -> System.out.println("Exception: " + task.exception());
            }
            
            /**
             * SUBTASK METHODS:
             * 
             * state():
             * - UNAVAILABLE: Not completed
             * - SUCCESS: Completed successfully
             * - FAILED: Failed with exception
             * 
             * get():
             * - Returns result if SUCCESS
             * - Throws if not SUCCESS
             * 
             * exception():
             * - Returns exception if FAILED
             * - Throws if not FAILED
             */
        }
    }
    
    /**
     * Multiple tasks
     */
    public static void multipleTasks() throws InterruptedException, ExecutionException {
        System.out.println("\n=== Multiple Tasks ===");
        
        record User(String name, int age) {}
        record Order(String id, double amount) {}
        record Recommendation(String product) {}
        
        // Parallel API calls
        try (var scope = new StructuredTaskScope<Object>()) {
            // Fork all tasks
            Subtask<User> userTask = scope.fork(() -> 
                fetchUser("user123")
            );
            
            Subtask<Order> orderTask = scope.fork(() -> 
                fetchOrders("user123")
            );
            
            Subtask<Recommendation> recTask = scope.fork(() -> 
                fetchRecommendations("user123")
            );
            
            // Wait for all
            scope.join();
            
            // Collect results
            User user = userTask.get();
            Order order = orderTask.get();
            Recommendation rec = recTask.get();
            
            System.out.println("User: " + user);
            System.out.println("Order: " + order);
            System.out.println("Recommendation: " + rec);
        }
    }
    
    private static User fetchUser(String id) throws InterruptedException {
        Thread.sleep(100);
        return new User("Alice", 30);
    }
    
    private static Order fetchOrders(String userId) throws InterruptedException {
        Thread.sleep(200);
        return new Order("ORD-001", 99.99);
    }
    
    private static Recommendation fetchRecommendations(String userId) throws InterruptedException {
        Thread.sleep(150);
        return new Recommendation("Product X");
    }
    
    /**
     * Timeout handling
     */
    public static void timeoutHandling() throws InterruptedException, TimeoutException {
        System.out.println("\n=== Timeout Handling ===");
        
        try (var scope = new StructuredTaskScope<String>()) {
            scope.fork(() -> {
                Thread.sleep(5000);  // Too slow!
                return "Late result";
            });
            
            // Wait with timeout
            scope.joinUntil(
                java.time.Instant.now().plusSeconds(1)
            );
            
            // If we get here, timeout occurred
            System.out.println("Timeout - tasks cancelled");
            
        } catch (TimeoutException e) {
            System.out.println("Tasks took too long");
        }
        
        /**
         * TIMEOUT:
         * 
         * joinUntil(deadline):
         * - Wait until deadline
         * - Throws TimeoutException if not done
         * - Cancels all subtasks
         * 
         * Automatic cleanup!
         */
    }
    
    /**
     * Context inheritance
     */
    public static void contextInheritance() throws InterruptedException, ExecutionException {
        System.out.println("\n=== Context Inheritance ===");
        
        // ScopedValue (Java 21+)
        final ScopedValue<String> REQUEST_ID = ScopedValue.newInstance();
        
        ScopedValue.where(REQUEST_ID, "REQ-12345").run(() -> {
            try (var scope = new StructuredTaskScope<String>()) {
                Subtask<String> task = scope.fork(() -> {
                    // ScopedValue inherited!
                    String requestId = REQUEST_ID.get();
                    return "Processing: " + requestId;
                });
                
                scope.join();
                System.out.println(task.get());
                
            } catch (InterruptedException | ExecutionException e) {
                e.printStackTrace();
            }
        });
        
        /**
         * CONTEXT INHERITANCE:
         * 
         * Subtasks inherit:
         * ✓ ScopedValues
         * ✓ Security context
         * ✓ Tracing context
         * ✓ Any scoped context
         * 
         * Automatic propagation!
         * No manual setup needed!
         */
    }
    
    public static void main(String[] args) throws Exception {
        basicUsage();
        subtaskHandle();
        multipleTasks();
        timeoutHandling();
        contextInheritance();
    }
}
```

## 3. ShutdownOnSuccess and ShutdownOnFailure

### Built-in Shutdown Policies

```java
/**
 * SHUTDOWN POLICIES
 * 
 * ShutdownOnSuccess and ShutdownOnFailure
 */

import java.util.concurrent.*;
import java.util.concurrent.StructuredTaskScope.*;
import java.time.Duration;

public class ShutdownPolicies {
    
    /**
     * ShutdownOnSuccess - race to completion
     */
    public static void shutdownOnSuccess() throws InterruptedException, ExecutionException {
        System.out.println("=== ShutdownOnSuccess ===");
        
        // First to succeed wins, others cancelled
        try (var scope = new StructuredTaskScope.ShutdownOnSuccess<String>()) {
            
            // Fork multiple tasks
            scope.fork(() -> {
                Thread.sleep(2000);
                return "Slow result";
            });
            
            scope.fork(() -> {
                Thread.sleep(500);
                return "Fast result";  // This wins!
            });
            
            scope.fork(() -> {
                Thread.sleep(1000);
                return "Medium result";
            });
            
            // Wait for first success
            scope.join();
            
            // Get winning result
            String result = scope.result();
            System.out.println("Winner: " + result);
            
            /**
             * SHUTDOWNONSUCCESS:
             * 
             * - Waits for first successful completion
             * - Cancels all other tasks immediately
             * - result() returns the winning value
             * - Perfect for "fastest wins" scenarios
             * 
             * Use cases:
             * - Query multiple servers (fastest response)
             * - Try multiple algorithms (first to finish)
             * - Redundant computations (any result ok)
             */
        }
    }
    
    /**
     * ShutdownOnFailure - fail fast
     */
    public static void shutdownOnFailure() {
        System.out.println("\n=== ShutdownOnFailure ===");
        
        // Any failure cancels all
        try (var scope = new StructuredTaskScope.ShutdownOnFailure()) {
            
            scope.fork(() -> {
                Thread.sleep(1000);
                return "Task 1 ok";
            });
            
            scope.fork(() -> {
                Thread.sleep(500);
                throw new RuntimeException("Task 2 failed!");
                // This cancels all other tasks!
            });
            
            scope.fork(() -> {
                Thread.sleep(2000);
                return "Task 3 ok";  // Never completes
            });
            
            // Wait for completion or failure
            scope.join();
            
            // Check if any failed
            scope.throwIfFailed();  // Throws!
            
        } catch (InterruptedException e) {
            e.printStackTrace();
        } catch (ExecutionException e) {
            System.out.println("Caught: " + e.getCause().getMessage());
            
            /**
             * SHUTDOWNONFAILURE:
             * 
             * - Waits for all to succeed OR first failure
             * - On failure, cancels all other tasks
             * - throwIfFailed() rethrows first exception
             * - Perfect for "all or nothing" scenarios
             * 
             * Use cases:
             * - Multiple API calls (need all)
             * - Parallel validations (any failure fails all)
             * - Multi-step workflows (any error aborts)
             */
        }
    }
    
    /**
     * ShutdownOnSuccess - real example
     */
    public static void realExampleSuccess() throws InterruptedException, ExecutionException {
        System.out.println("\n=== Real Example: Fastest Server ===");
        
        // Query multiple servers, use fastest
        String result = queryMultipleServers("search query");
        System.out.println("Result: " + result);
    }
    
    private static String queryMultipleServers(String query) 
            throws InterruptedException, ExecutionException {
        
        try (var scope = new StructuredTaskScope.ShutdownOnSuccess<String>()) {
            
            // Query server 1 (US)
            scope.fork(() -> queryServer("us-server", query, 800));
            
            // Query server 2 (EU)
            scope.fork(() -> queryServer("eu-server", query, 500));
            
            // Query server 3 (ASIA)
            scope.fork(() -> queryServer("asia-server", query, 1200));
            
            scope.join();
            return scope.result();  // Fastest response
        }
    }
    
    private static String queryServer(String server, String query, long delay) 
            throws InterruptedException {
        Thread.sleep(delay);
        return "Result from " + server + " for: " + query;
    }
    
    /**
     * ShutdownOnFailure - real example
     */
    public static void realExampleFailure() {
        System.out.println("\n=== Real Example: All or Nothing ===");
        
        try {
            processOrder("ORDER-123");
            System.out.println("Order processed successfully");
        } catch (Exception e) {
            System.out.println("Order failed: " + e.getMessage());
        }
    }
    
    private static void processOrder(String orderId) 
            throws InterruptedException, ExecutionException {
        
        try (var scope = new StructuredTaskScope.ShutdownOnFailure()) {
            
            // Validate payment
            scope.fork(() -> validatePayment(orderId));
            
            // Check inventory
            scope.fork(() -> checkInventory(orderId));
            
            // Verify address
            scope.fork(() -> verifyAddress(orderId));
            
            // All must succeed
            scope.join();
            scope.throwIfFailed();  // Throws if any failed
            
            // All succeeded - commit order
            System.out.println("All validations passed");
        }
    }
    
    private static String validatePayment(String orderId) throws InterruptedException {
        Thread.sleep(100);
        // throw new RuntimeException("Payment declined");
        return "Payment ok";
    }
    
    private static String checkInventory(String orderId) throws InterruptedException {
        Thread.sleep(150);
        return "Inventory ok";
    }
    
    private static String verifyAddress(String orderId) throws InterruptedException {
        Thread.sleep(120);
        return "Address ok";
    }
    
    /**
     * Combining with timeout
     */
    public static void combiningWithTimeout() throws InterruptedException, TimeoutException {
        System.out.println("\n=== Combining with Timeout ===");
        
        try (var scope = new StructuredTaskScope.ShutdownOnSuccess<String>()) {
            
            scope.fork(() -> {
                Thread.sleep(5000);
                return "Slow result";
            });
            
            scope.fork(() -> {
                Thread.sleep(3000);
                return "Medium result";
            });
            
            // Wait for success OR timeout
            scope.joinUntil(
                java.time.Instant.now().plusSeconds(1)
            );
            
            // Will throw TimeoutException
            
        } catch (ExecutionException e) {
            System.out.println("Failed: " + e.getMessage());
        } catch (TimeoutException e) {
            System.out.println("Timeout - no result in time");
        }
    }
    
    public static void main(String[] args) throws Exception {
        shutdownOnSuccess();
        shutdownOnFailure();
        realExampleSuccess();
        realExampleFailure();
        combiningWithTimeout();
    }
}
```

## 4. Exception Handling

### Managing Errors in Concurrent Tasks

```java
/**
 * EXCEPTION HANDLING
 * 
 * Error management in structured concurrency
 */

import java.util.concurrent.*;
import java.util.concurrent.StructuredTaskScope.*;

public class ExceptionHandling {
    
    /**
     * Basic exception handling
     */
    public static void basicExceptionHandling() {
        System.out.println("=== Basic Exception Handling ===");
        
        try (var scope = new StructuredTaskScope.ShutdownOnFailure()) {
            
            scope.fork(() -> {
                throw new RuntimeException("Task failed!");
            });
            
            scope.fork(() -> {
                return "Task succeeded";
            });
            
            scope.join();
            scope.throwIfFailed();
            
        } catch (InterruptedException e) {
            System.out.println("Interrupted: " + e);
        } catch (ExecutionException e) {
            System.out.println("Execution failed: " + e.getCause().getMessage());
            
            /**
             * EXCEPTION HANDLING:
             * 
             * throwIfFailed():
             * - Throws ExecutionException
             * - Cause is the original exception
             * - Happens after all tasks cancelled
             */
        }
    }
    
    /**
     * Multiple exceptions
     */
    public static void multipleExceptions() {
        System.out.println("\n=== Multiple Exceptions ===");
        
        try (var scope = new StructuredTaskScope.ShutdownOnFailure()) {
            
            scope.fork(() -> {
                throw new RuntimeException("Error 1");
            });
            
            scope.fork(() -> {
                Thread.sleep(100);  // Slightly delayed
                throw new RuntimeException("Error 2");
            });
            
            scope.join();
            scope.throwIfFailed();
            
        } catch (InterruptedException e) {
            e.printStackTrace();
        } catch (ExecutionException e) {
            System.out.println("First failure: " + e.getCause().getMessage());
            
            /**
             * MULTIPLE FAILURES:
             * 
             * - Only first exception thrown
             * - Others are suppressed
             * - Available via getSuppressed()
             * 
             * This is fail-fast behavior
             */
            
            Throwable[] suppressed = e.getSuppressed();
            for (Throwable t : suppressed) {
                System.out.println("Suppressed: " + t.getMessage());
            }
        }
    }
    
    /**
     * Collecting all exceptions
     */
    public static void collectingAllExceptions() throws InterruptedException {
        System.out.println("\n=== Collecting All Exceptions ===");
        
        // Custom policy to collect all exceptions
        class CollectAllFailures<T> extends StructuredTaskScope<T> {
            private final ConcurrentLinkedQueue<Throwable> exceptions = 
                new ConcurrentLinkedQueue<>();
            
            @Override
            protected void handleComplete(Subtask<? extends T> subtask) {
                if (subtask.state() == Subtask.State.FAILED) {
                    exceptions.add(subtask.exception());
                }
            }
            
            public void throwIfAnyFailed() throws ExecutionException {
                if (!exceptions.isEmpty()) {
                    ExecutionException ex = new ExecutionException(
                        "Multiple tasks failed",
                        exceptions.poll()
                    );
                    exceptions.forEach(ex::addSuppressed);
                    throw ex;
                }
            }
        }
        
        try (var scope = new CollectAllFailures<String>()) {
            
            scope.fork(() -> {
                throw new RuntimeException("Error 1");
            });
            
            scope.fork(() -> {
                throw new RuntimeException("Error 2");
            });
            
            scope.fork(() -> {
                throw new RuntimeException("Error 3");
            });
            
            scope.join();
            scope.throwIfAnyFailed();
            
        } catch (ExecutionException e) {
            System.out.println("Main: " + e.getCause().getMessage());
            System.out.println("Suppressed: " + e.getSuppressed().length);
            for (Throwable t : e.getSuppressed()) {
                System.out.println("  - " + t.getMessage());
            }
        }
    }
    
    /**
     * Partial success handling
     */
    public static void partialSuccessHandling() throws InterruptedException {
        System.out.println("\n=== Partial Success ===");
        
        try (var scope = new StructuredTaskScope<String>()) {
            
            Subtask<String> task1 = scope.fork(() -> "Success 1");
            Subtask<String> task2 = scope.fork(() -> {
                throw new RuntimeException("Failed!");
            });
            Subtask<String> task3 = scope.fork(() -> "Success 3");
            
            scope.join();
            
            // Check each task individually
            if (task1.state() == Subtask.State.SUCCESS) {
                System.out.println("Task 1: " + task1.get());
            }
            
            if (task2.state() == Subtask.State.FAILED) {
                System.out.println("Task 2 failed: " + task2.exception().getMessage());
            }
            
            if (task3.state() == Subtask.State.SUCCESS) {
                System.out.println("Task 3: " + task3.get());
            }
            
            /**
             * PARTIAL SUCCESS:
             * 
             * - Check each Subtask individually
             * - Handle successes and failures separately
             * - Useful when some failures acceptable
             */
        } catch (ExecutionException e) {
            e.printStackTrace();
        }
    }
    
    /**
     * Retry on failure
     */
    public static void retryOnFailure() throws InterruptedException, ExecutionException {
        System.out.println("\n=== Retry on Failure ===");
        
        String result = withRetry(() -> unreliableOperation(), 3);
        System.out.println("Result: " + result);
    }
    
    private static int attempt = 0;
    
    private static String unreliableOperation() throws Exception {
        attempt++;
        if (attempt < 3) {
            throw new RuntimeException("Temporary failure");
        }
        return "Success on attempt " + attempt;
    }
    
    private static <T> T withRetry(
            Callable<T> task, 
            int maxAttempts
    ) throws InterruptedException, ExecutionException {
        
        for (int i = 0; i < maxAttempts; i++) {
            try (var scope = new StructuredTaskScope.ShutdownOnFailure()) {
                scope.fork(task);
                scope.join();
                scope.throwIfFailed();
                // Success!
                return (T) scope.fork(task).get();
                
            } catch (ExecutionException e) {
                if (i == maxAttempts - 1) {
                    throw e;  // Last attempt failed
                }
                System.out.println("Attempt " + (i + 1) + " failed, retrying...");
                Thread.sleep(1000 * (i + 1));  // Exponential backoff
            }
        }
        
        throw new RuntimeException("Unreachable");
    }
    
    public static void main(String[] args) throws Exception {
        basicExceptionHandling();
        multipleExceptions();
        collectingAllExceptions();
        partialSuccessHandling();
        retryOnFailure();
    }
```
