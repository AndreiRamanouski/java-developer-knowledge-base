# String Optimization

## Overview

Strings are ubiquitous in Java applications—from user input to configuration files, log messages to database queries—and improper string handling is one of the most common performance bottlenecks. While String's immutability provides safety and enables caching, it also means every concatenation creates a new object, potentially generating millions of short-lived garbage objects. This guide covers string optimization techniques: StringBuilder vs StringBuffer vs the + operator (when the compiler optimizes and when it doesn't), String.intern() for sharing duplicate strings (saving memory but with caveats), string deduplication (JVM automatically detecting and eliminating duplicate strings), compact strings in Java 9+ (Latin1 encoding for ASCII strings saves 50% memory), avoiding concatenation in loops (the #1 string performance mistake), and regular expression compilation (compile once, use many times). Understanding these optimizations is critical—a single `str += x` in a loop can turn O(n) code into O(n²), while proper use of StringBuilder makes it O(n). Whether you're processing logs, building SQL queries, or formatting output, these techniques can make your code 10-1000x faster while reducing memory pressure and GC overhead.

## 1. StringBuilder vs StringBuffer vs +

### The String Concatenation Showdown

```java
/**
 * STRING CONCATENATION METHODS
 */

import org.openjdk.jmh.annotations.*;
import java.util.concurrent.TimeUnit;

public class StringConcatenationComparison {
    
    /**
     * Understanding String immutability
     */
    public static void stringImmutability() {
        /**
         * STRING IMMUTABILITY:
         * 
         * String is IMMUTABLE
         * Every operation creates new String
         * 
         * String s1 = "Hello";
         * String s2 = s1 + " World";
         * 
         * What happens:
         * 1. Create new char array
         * 2. Copy s1 contents
         * 3. Copy " World" contents
         * 4. Create new String object
         * 5. s1 unchanged, s2 is new object
         * 
         * 
         * IMPLICATIONS:
         * 
         * String s = "";
         * for (int i = 0; i < 1000; i++) {
         *   s = s + i;  // 1000 new String objects!
         * }
         * 
         * Each iteration:
         * - Allocates new String
         * - Copies all previous content
         * - Adds new content
         * 
         * O(n²) time complexity!
         * 1000 iterations = 500,000 char copies!
         */
    }
    
    /**
     * The + operator
     */
    @Benchmark
    @BenchmarkMode(Mode.AverageTime)
    @OutputTimeUnit(TimeUnit.MICROSECONDS)
    public String plusOperator() {
        String result = "";
        for (int i = 0; i < 100; i++) {
            result = result + i;  // BAD!
        }
        return result;
    }
    
    /**
     * WHAT COMPILER DOES:
     * 
     * Modern Java compilers optimize:
     * String s = "a" + "b" + "c";
     * 
     * Becomes:
     * String s = "abc";  // Compile-time constant
     * 
     * 
     * But in loops:
     * for (int i = 0; i < 100; i++) {
     *   s = s + i;
     * }
     * 
     * Each iteration becomes:
     * s = new StringBuilder(s).append(i).toString();
     * 
     * 100 StringBuilder allocations!
     * Still inefficient!
     */
    
    /**
     * StringBuilder - mutable, not thread-safe
     */
    @Benchmark
    @BenchmarkMode(Mode.AverageTime)
    @OutputTimeUnit(TimeUnit.MICROSECONDS)
    public String stringBuilder() {
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < 100; i++) {
            sb.append(i);
        }
        return sb.toString();
    }
    
    /**
     * STRINGBUILDER:
     * 
     * - Mutable character array
     * - append() adds to existing buffer
     * - Resizes when needed (doubles capacity)
     * - NOT thread-safe
     * - Fast!
     * 
     * Time complexity: O(n)
     * Only final toString() creates String
     */
    
    /**
     * StringBuffer - mutable, thread-safe
     */
    @Benchmark
    @BenchmarkMode(Mode.AverageTime)
    @OutputTimeUnit(TimeUnit.MICROSECONDS)
    public String stringBuffer() {
        StringBuffer sb = new StringBuffer();
        for (int i = 0; i < 100; i++) {
            sb.append(i);
        }
        return sb.toString();
    }
    
    /**
     * STRINGBUFFER:
     * 
     * - Same as StringBuilder
     * - BUT synchronized methods
     * - Thread-safe
     * - Slower (synchronization overhead)
     * - Rarely needed (use StringBuilder + external sync)
     */
    
    /**
     * BENCHMARK RESULTS (100 iterations):
     * 
     * + operator:     5000 μs  (O(n²))
     * StringBuilder:    50 μs  (O(n)) - 100x faster!
     * StringBuffer:     80 μs  (O(n)) - synchronized overhead
     * 
     * For 1000 iterations:
     * + operator:     500,000 μs  (500ms!)
     * StringBuilder:       500 μs  (0.5ms)
     * 
     * 1000x difference!
     */
    
    /**
     * When compiler DOES optimize
     */
    public static void compilerOptimization() {
        // OPTIMIZED by compiler
        String s1 = "Hello" + " " + "World";
        // Becomes: String s1 = "Hello World";
        
        // OPTIMIZED by compiler
        String s2 = "Value: " + 42;
        // Becomes: String s2 = "Value: 42";
        
        // OPTIMIZED by compiler (Java 9+)
        String s3 = method1() + method2();
        // Uses invokedynamic + StringConcatFactory
        // Efficient!
        
        /**
         * JAVA 9+ OPTIMIZATION:
         * 
         * Uses invokedynamic
         * StringConcatFactory creates optimized code
         * No explicit StringBuilder in bytecode
         * JVM optimizes at runtime
         * 
         * But still O(n²) in loops!
         */
    }
    
    /**
     * When compiler DOESN'T optimize
     */
    public static void noOptimization() {
        // NOT OPTIMIZED - loop
        String s = "";
        for (int i = 0; i < 100; i++) {
            s += i;  // BAD! O(n²)
        }
        
        // NOT OPTIMIZED - unknown length
        String[] parts = getParts();
        String result = "";
        for (String part : parts) {
            result += part;  // BAD! O(n²)
        }
    }
    
    /**
     * Real-world example: Building SQL
     */
    public static void buildingSqlQueries() {
        // BAD: String concatenation
        String sql = "SELECT * FROM users WHERE ";
        sql += "age > 18";
        sql += " AND ";
        sql += "active = true";
        sql += " ORDER BY name";
        // Multiple String objects created
        
        // GOOD: StringBuilder
        StringBuilder sql2 = new StringBuilder("SELECT * FROM users WHERE ");
        sql2.append("age > 18");
        sql2.append(" AND ");
        sql2.append("active = true");
        sql2.append(" ORDER BY name");
        String result = sql2.toString();
        // Single final String created
        
        /**
         * REAL DIFFERENCE:
         * 
         * Building query with 20 parts:
         * + operator: 20 String allocations
         * StringBuilder: 1 String allocation
         * 
         * 20x less garbage!
         */
    }
    
    /**
     * StringBuilder capacity
     */
    public static void stringBuilderCapacity() {
        // Default capacity: 16
        StringBuilder sb1 = new StringBuilder();
        
        // Specified capacity
        StringBuilder sb2 = new StringBuilder(100);
        
        /**
         * CAPACITY MATTERS:
         * 
         * If result will be ~100 chars:
         * 
         * Default capacity (16):
         * - Resize at 16 chars
         * - Resize at 34 chars
         * - Resize at 70 chars
         * - Resize at 142 chars
         * 4 allocations + copies!
         * 
         * With capacity (100):
         * - No resize needed
         * 1 allocation!
         * 
         * 
         * BEST PRACTICE:
         * 
         * If you know approximate size:
         * new StringBuilder(estimatedSize)
         * 
         * Saves allocations and copies
         */
    }
    
    private static String method1() { return "a"; }
    private static String method2() { return "b"; }
    private static String[] getParts() { return new String[]{"a", "b"}; }
}
```

## 2. String.intern() Use Cases

### Sharing String Instances

```java
/**
 * STRING INTERNING
 */

import java.util.*;

public class StringInterning {
    
    /**
     * What is String.intern()?
     */
    public static void whatIsIntern() {
        /**
         * STRING INTERNING:
         * 
         * String pool in memory
         * Stores unique String values
         * Multiple references to same object
         * 
         * String s1 = "hello";        // In pool
         * String s2 = "hello";        // Same object!
         * s1 == s2  // true
         * 
         * String s3 = new String("hello");  // Heap
         * String s4 = s3.intern();          // Pool
         * s4 == s1  // true
         * 
         * 
         * MEMORY SAVINGS:
         * 
         * Without interning:
         * 1000 "hello" strings = 1000 * 40 bytes = 40 KB
         * 
         * With interning:
         * 1000 references = 1 string * 40 bytes + 1000 * 8 bytes
         *                 = 40 bytes + 8 KB = ~8 KB
         * 
         * 5x memory savings!
         */
    }
    
    /**
     * Automatic interning of literals
     */
    public static void literalInterning() {
        String s1 = "hello";
        String s2 = "hello";
        String s3 = "hel" + "lo";  // Compile-time constant
        
        System.out.println(s1 == s2);  // true
        System.out.println(s1 == s3);  // true
        
        /**
         * ALL string literals automatically interned
         * Compile-time constants interned
         * Same object in memory
         */
    }
    
    /**
     * Manual interning
     */
    public static void manualInterning() {
        String s1 = new String("hello");  // Heap
        String s2 = new String("hello");  // Different object
        
        System.out.println(s1 == s2);  // false (different objects)
        
        String i1 = s1.intern();  // Pool
        String i2 = s2.intern();  // Same pool object
        
        System.out.println(i1 == i2);  // true (same object)
        
        /**
         * intern() METHOD:
         * 
         * If string in pool:
         * - Return pool reference
         * 
         * If string not in pool:
         * - Add to pool
         * - Return pool reference
         * 
         * 
         * PERFORMANCE:
         * 
         * intern() is SLOW:
         * - Hash map lookup
         * - String comparison
         * - Synchronized
         * - ~100-1000 ns
         * 
         * But == is FAST:
         * - Reference comparison
         * - ~1 ns
         * 
         * Trade-off: Slow intern vs fast comparison
         */
    }
    
    /**
     * Good use case: Configuration keys
     */
    public static void configurationKeys() {
        Map<String, String> config = new HashMap<>();
        
        // Loading config file
        String[] lines = readConfigFile();
        for (String line : lines) {
            String[] parts = line.split("=");
            String key = parts[0].trim().intern();    // Intern key!
            String value = parts[1].trim();
            config.put(key, value);
        }
        
        /**
         * WHY INTERN KEYS:
         * 
         * Keys repeat: database.host, database.port, etc.
         * 
         * Without interning:
         * - "database.host" appears 10 times
         * - 10 String objects
         * - 10 * 40 bytes = 400 bytes
         * 
         * With interning:
         * - 1 String object
         * - 40 bytes
         * 
         * 10x memory savings!
         * 
         * Plus: Fast comparison with ==
         */
    }
    
    /**
     * Good use case: Enum-like values
     */
    public static void enumLikeValues() {
        // Parsing status codes
        Set<String> knownStatuses = new HashSet<>(Arrays.asList(
            "SUCCESS", "FAILURE", "PENDING", "CANCELLED"
        ));
        
        // Processing records
        List<Record> records = loadRecords();
        for (Record record : records) {
            String status = parseStatus(record).intern();  // Intern!
            
            if (status == "SUCCESS") {  // Can use ==
                processSuccess(record);
            }
        }
        
        /**
         * BENEFITS:
         * 
         * 1. Memory savings (many duplicate statuses)
         * 2. Fast comparison (== instead of equals())
         * 3. Thread-safe (interned strings immutable)
         */
    }
    
    /**
     * Bad use case: User input
     */
    public static void userInputBad() {
        // DON'T DO THIS!
        List<String> usernames = new ArrayList<>();
        
        while (true) {
            String input = readUserInput();
            String username = input.intern();  // BAD!
            usernames.add(username);
        }
        
        /**
         * PROBLEMS:
         * 
         * 1. Strings rarely duplicate
         *    - Each username likely unique
         *    - No memory savings
         * 
         * 2. String pool grows forever
         *    - Pool in PermGen (Java 7)
         *    - Can cause OutOfMemoryError
         *    - Even in Java 8+, wastes space
         * 
         * 3. intern() overhead
         *    - Hash lookup cost
         *    - No benefit gained
         * 
         * 4. Potential DoS attack
         *    - Malicious user sends many unique strings
         *    - Fills string pool
         *    - Application crashes
         */
    }
    
    /**
     * When to use intern()
     */
    public static void whenToUseIntern() {
        /**
         * USE INTERN WHEN:
         * 
         * ✓ Many duplicate strings expected
         *   - Configuration keys
         *   - Status codes
         *   - Country codes
         *   - Category names
         * 
         * ✓ Strings are long-lived
         *   - Held in caches
         *   - Stored in collections
         *   - Application lifetime
         * 
         * ✓ Frequent equality checks
         *   - Can use == instead of equals()
         *   - 10-100x faster comparison
         * 
         * ✓ Bounded set of values
         *   - Known set (like enum)
         *   - Won't grow indefinitely
         * 
         * 
         * DON'T USE WHEN:
         * 
         * ✗ Unique strings
         *   - User input
         *   - Generated IDs
         *   - Random data
         * 
         * ✗ Large strings
         *   - File contents
         *   - Log messages
         *   - Large documents
         * 
         * ✗ Short-lived strings
         *   - Temporary variables
         *   - Local processing
         *   - Soon garbage collected
         * 
         * ✗ Unbounded growth
         *   - User-generated content
         *   - External data
         *   - Unpredictable input
         */
    }
    
    private static String[] readConfigFile() { return new String[]{"key=value"}; }
    private static List<Record> loadRecords() { return new ArrayList<>(); }
    private static String parseStatus(Record r) { return "SUCCESS"; }
    private static void processSuccess(Record r) { }
    private static String readUserInput() { return "input"; }
    
    static class Record { }
}
```

## 3. String Deduplication

### JVM Feature for Memory Savings

```java
/**
 * STRING DEDUPLICATION
 */

public class StringDeduplication {
    
    /**
     * What is string deduplication?
     */
    public static void whatIsDeduplication() {
        /**
         * STRING DEDUPLICATION:
         * 
         * JVM feature (Java 8u20+)
         * Automatically detects duplicate strings
         * Shares underlying char arrays
         * 
         * 
         * HOW IT WORKS:
         * 
         * String s1 = new String("hello");  // char[] = ['h','e','l','l','o']
         * String s2 = new String("hello");  // char[] = ['h','e','l','l','o']
         * 
         * Without deduplication:
         * s1.value → char[5] = "hello"
         * s2.value → char[5] = "hello"  (different array)
         * 
         * With deduplication:
         * s1.value → char[5] = "hello"
         * s2.value → same char[5]       (shared array!)
         * 
         * 
         * MEMORY SAVINGS:
         * 
         * 1000 duplicate "hello" strings:
         * Without: 1000 * 5 chars * 2 bytes = 10 KB
         * With: 1 * 5 chars * 2 bytes = 10 bytes
         * 
         * 1000x reduction!
         */
    }
    
    /**
     * Enabling string deduplication
     */
    public static void enablingDeduplication() {
        /**
         * JVM FLAGS:
         * 
         * Enable deduplication:
         * -XX:+UseStringDeduplication
         * 
         * Requires G1 GC:
         * -XX:+UseG1GC
         * 
         * Age threshold (default 3):
         * -XX:StringDeduplicationAgeThreshold=3
         * 
         * 
         * HOW IT WORKS:
         * 
         * 1. During GC, young gen strings promoted to old gen
         * 2. If string age >= threshold
         * 3. Add to deduplication queue
         * 4. Background thread processes queue
         * 5. Compares strings for duplicates
         * 6. Shares char arrays
         * 
         * 
         * OVERHEAD:
         * 
         * CPU: ~1-2% (background thread)
         * Memory: Hash table for tracking
         * 
         * Usually worth it!
         */
    }
    
    /**
     * When it helps

    public static void whenItHelps() {
        /**

         *    - Log messages
         *    - JSON/XML keys
         *    - Database records
         * 
         * 2. Large heap sizes
         *    - More strings → more duplicates
         *    - Bigger savings
         * 
         * 3. Long-running applications
         *    - Strings accumulate
         *    - Deduplication pays off
         * 
         * 
         * EXAMPLE SCENARIOS:
         * 
         * JSON parsing:
         * - Field names repeat: "id", "name", "email"
         * - Many records with same fields
         * - 20-40% heap reduction!
         * 
         * Logging:
         * - Log patterns repeat

         * - 10-30% heap reduction
         * 
         * Database caches:
         * - Column names repeat
         * - Status values repeat
         * - 30-50% heap reduction!
         */
```

## 4. Compact Strings

### Latin1 Encoding for Memory Savings

```java
/**
 * COMPACT STRINGS
 */

public class CompactStrings {
    
    /**
     * What are compact strings?
     */
    public static void whatAreCompactStrings() {
        /**
         * COMPACT STRINGS (Java 9+):
         * 
         * Before Java 9:
         * - String backed by char[]
         * - Each char = 2 bytes (UTF-16)
         * - "hello" = 10 bytes
         * 
         * Java 9+:
         * - String backed by byte[]
         * - Latin1 (1 byte) or UTF-16 (2 bytes)
         * - Encoding flag (coder field)
         * - "hello" = 5 bytes (Latin1!)
         * 
         * 
         * MEMORY SAVINGS:
         * 
         * ASCII strings:
         * Before: 2 bytes per char
         * After: 1 byte per char
         * 50% reduction!
         * 
         * 
         * AUTOMATIC:
         * 
         * JVM chooses encoding automatically
         * Latin1 if all chars < 256
         * UTF-16 if any char >= 256
         * 
         * No code changes needed!
         */
    }
    
    /**
     * String internal structure
     */
    public static void stringStructure() {
        /**
         * JAVA 8:
         * 
         * class String {
         *   char[] value;  // Always UTF-16
         *   int hash;
         * }
         * 
         * "hello" storage:
         * value = ['h','e','l','l','o']  // 10 bytes
         * 
         * 
         * JAVA 9+:
         * 
         * class String {
         *   byte[] value;  // Latin1 or UTF-16
         *   byte coder;    // LATIN1 or UTF16
         *   int hash;
         * }
         * 
         * "hello" storage (ASCII):
         * value = [104,101,108,108,111]  // 5 bytes
         * coder = LATIN1
         * 
         * "héllo" storage (non-ASCII):
         * value = [0,104,0,233,0,108,0,108,0,111]  // 10 bytes
         * coder = UTF16
         */
    }
    
    /**
     * Performance characteristics
     */
    public static void performanceCharacteristics() {
        /**
         * LATIN1 STRINGS:
         * 
         * Pros:
         * ✓ 50% less memory
         * ✓ Better cache locality

         * 
         * Cons:
         * ✗ Extra coder check

         * 
         * BENCHMARK:
         * 
         * equals() on ASCII strings:
         * Java 8:  100 ns
         * Java 9+: 60 ns  (40% faster!)
         * 
         * substring() on ASCII:
         * Java 8:  50 ns
         * Java 9+: 30 ns  (40% faster!)
         * 
         * 
         * OVERALL:
         * 
         * Most strings in typical apps are ASCII
         * → 20-30% heap reduction
         * → 10-20% performance improvement

    }
    
    /**
     * Disabling compact strings
     */
    public static void disablingCompactStrings() {
        /**
         * JVM FLAG:
         * 
         * Disable compact strings:
         * -XX:-CompactStrings
         * 
         * (Enabled by default)
         * 
         * 
         * WHEN TO DISABLE:
         * 
         * ✗ Almost never!
         * 
         * Rare cases:
         * - Mostly non-ASCII strings
         * - Coder checks hurt performance
         * - Profile first!
         * 
         * 
         * RECOMMENDATION:
         * 
         * Leave enabled (default)
         * Huge benefit for most apps
         */
    }
    
    /**
     * Real-world impact
     */
    public static void realWorldImpact() {
        /**
         * CASE STUDY: Web application
         * 
         * String types:
         * - URLs: ASCII
         * - User names: mostly ASCII
         * - Log messages: ASCII
         * - Configuration: ASCII
```

## 5. Avoiding String Concatenation in Loops

### The #1 String Performance Mistake

```java
/**
 * STRING CONCATENATION IN LOOPS
 */

import org.openjdk.jmh.annotations.*;
import java.util.concurrent.TimeUnit;
import java.util.stream.Collectors;
import java.util.stream.IntStream;

public class LoopConcatenation {
    
    /**
     * The classic mistake
     */
    @Benchmark
    @BenchmarkMode(Mode.AverageTime)
    @OutputTimeUnit(TimeUnit.MILLISECONDS)
    public String concatenationInLoop() {
        String result = "";
        for (int i = 0; i < 1000; i++) {
            result += i + ",";  // TERRIBLE!
        }
        return result;
    }
    
    /**
     * WHAT'S WRONG:
     * 
     * Iteration 1: result = "" + "0,"           (1 char copy)
     * Iteration 2: result = "0," + "1,"         (3 char copies)
     * Iteration 3: result = "0,1," + "2,"       (6 char copies)
     * ...
     * Iteration 1000: result = "0,1,...999," + "1000,"  (huge copy!)
     * 
     * Total copies: 1 + 3 + 6 + ... + 3000 = ~1,500,000 char copies!
     * 
     * Time complexity: O(n²)
     * 1000 iterations → 500,000x work!
     */
    
    /**
     * The correct way
     */
    @Benchmark
    @BenchmarkMode(Mode.AverageTime)
    @OutputTimeUnit(TimeUnit.MILLISECONDS)
    public String stringBuilderInLoop() {
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < 1000; i++) {
            sb.append(i).append(",");
        }
        return sb.toString();
    }
    
    /**
     * WHAT'S RIGHT:
     * 
     * Each iteration:
     * - Append to buffer (amortized O(1))
     * - No copying previous content
     * 
     * Final toString():
     * - Single copy to create String
     * 
     * Time complexity: O(n)
     * 1000 iterations → 1000x work
     * 
     * 500x faster!
     */
    
    /**
     * BENCHMARK RESULTS:
     *

        String bad = "";

     * Stream joining
     */
    public static void streamJoining() {

        String bad = IntStream.range(0, 1000)
            .mapToObj(String::valueOf)
            .reduce("", (a, b) -> a + b);  // O(n²)!

        String good = IntStream.range(0, 1000)
            .mapToObj(String::valueOf)

         * Collectors.joining():

            sqlBad += id + ",";  // O(n²)
        }
        sqlBad += ")";
        
        // GOOD
        StringBuilder sqlGood = new StringBuilder("SELECT * FROM users WHERE id IN (");
        for (int i = 0; i < ids.size(); i++) {
            if (i > 0) sqlGood.append(",");
            sqlGood.append(ids.get(i));
        }
        sqlGood.append(")");

        String sqlBetter = "SELECT * FROM users WHERE id IN (" +
            ids.stream()
               .map(String::valueOf)
               .collect(Collectors.joining(",")) +
            ")";
    }

/**
```

```java
import java.util.regex.*;
import org.openjdk.jmh.annotations.*;
import java.util.concurrent.TimeUnit;

public class RegexCompilation {
    
    /**
     * The expensive operation
     */
    public static void regexCompilation() {
        /**
         * REGEX COMPILATION:
         * 
         * Pattern.compile(regex) is EXPENSIVE
         * 
         * Steps:
         * 1. Parse regex syntax
         * 2. Build NFA (Non-deterministic Finite Automaton)
         * 3. Optimize state machine
         * 4. Generate bytecode
         * 
         * Time: ~1-10 microseconds
         * 
         * Matching is FAST (once compiled)
         * Time: ~10-100 nanoseconds
         * 
         * 100-1000x difference!
         */
    }
    
    /**
     * Naive approach - BAD
     */
    @Benchmark
    @BenchmarkMode(Mode.AverageTime)
    @OutputTimeUnit(TimeUnit.MICROSECONDS)
    public boolean naiveRegex() {
        String input = "test@example.com";
        return input.matches("^[A-Za-z0-9+_.-]+@[A-Za-z0-9.-]+$");
    }
    
    /**
     * PROBLEM:
     * 
     * matches() compiles Pattern every call!
     * 
     * Each call:
     * 1. Compile regex (~1000 ns)
     * 2. Match (~100 ns)
     * Total: ~1100 ns
     * 
     * Compilation is 90% of time!
     */
    
    /**
     * Pre-compiled Pattern - GOOD
     */
    private static final Pattern EMAIL_PATTERN = 
        Pattern.compile("^[A-Za-z0-9+_.-]+@[A-Za-z0-9.-]+$");
    
    @Benchmark
    @BenchmarkMode(Mode.AverageTime)
    @OutputTimeUnit(TimeUnit.MICROSECONDS)
    public boolean precompiledRegex() {
        String input = "test@example.com";
        return EMAIL_PATTERN.matcher(input).matches();
    }
    
    /**
     * Each call:
     * 1. Create matcher (~10 ns)
     * 2. Match (~100 ns)
     * Total: ~110 ns
     * 
     * 10x faster!
     */
    
    /**
     * BENCHMARK RESULTS:
     * 
     * naiveRegex:        1.1 μs
     * precompiledRegex:  0.11 μs
     * 
     * 10x faster!
     * 
     * For 1,000,000 validations:
     * Naive: 1100 seconds (18 minutes!)
     * Compiled: 110 seconds (2 minutes)
     */

     * Real-world example: Email validation
     */
    public static class EmailValidator {
        // BAD: Compile every time
        public static boolean isValidBad(String email) {

            Pattern.compile("^[A-Za-z0-9+_.-]+@[A-Za-z0-9.-]+$");

        }
    }

    }
```
