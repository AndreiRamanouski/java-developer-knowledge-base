# CPU Optimization Techniques

## Overview

CPU optimization is about making your code execute faster by understanding how modern processors work—instruction pipelining, branch prediction, cache hierarchies, and speculative execution create opportunities for dramatic speedups when used correctly, or devastating slowdowns when violated. This guide covers low-level CPU optimization techniques: loop unrolling (reducing loop overhead and enabling vectorization), branch prediction awareness (organizing code so the CPU's speculative execution succeeds), cache line awareness (arranging data for spatial locality), false sharing prevention (avoiding catastrophic cache coherency traffic with @Contended), algorithm complexity optimization (choosing O(log n) over O(n²)), data structure selection (ArrayList vs LinkedList can be 100x difference), and avoiding allocations in hot paths (reusing objects and using primitives). These techniques range from algorithmic improvements that provide orders of magnitude speedup, to micro-optimizations that squeeze out final percentages. Understanding CPU architecture separates good performance from exceptional performance—modern CPUs are incredibly sophisticated, and code that works with the hardware instead of against it can be 10-100x faster than naive implementations.

## 1. Loop Unrolling

### Reducing Loop Overhead

```java
/**
 * LOOP UNROLLING
 */

import org.openjdk.jmh.annotations.*;
import java.util.concurrent.TimeUnit;

public class LoopUnrolling {
    
    /**
     * What is loop unrolling?
     */
    public static void whatIsLoopUnrolling() {
        /**
         * LOOP UNROLLING:
         * 
         * Transform loop from:
         * for (int i = 0; i < 100; i++) {
         *   process(i);
         * }
         * 
         * To:
         * for (int i = 0; i < 100; i += 4) {
         *   process(i);
         *   process(i + 1);
         *   process(i + 2);
         *   process(i + 3);
         * }
         * 
         * 
         * BENEFITS:
         * 
         * ✓ Fewer loop iterations
         * ✓ Fewer branches
         * ✓ Less loop overhead (i++, condition check)
         * ✓ Better instruction pipelining
         * ✓ More opportunities for vectorization
         * ✓ Better register utilization
         * 
         * 
         * OVERHEAD REDUCTION:
         * 
         * Original: 100 iterations
         * - 100 increments (i++)
         * - 100 comparisons (i < 100)
         * - 100 branches
         * 
         * Unrolled 4x: 25 iterations
         * - 25 increments
         * - 25 comparisons
         * - 25 branches
         * 
         * 75% reduction in overhead!
         */
    }
    
    /**
     * Basic loop - baseline
     */
    @Benchmark
    @BenchmarkMode(Mode.AverageTime)
    @OutputTimeUnit(TimeUnit.NANOSECONDS)
    public int basicLoop() {
        int sum = 0;
        for (int i = 0; i < 1000; i++) {
            sum += i;
        }
        return sum;
    }
    
    /**
     * Manually unrolled loop - 2x
     */
    @Benchmark
    @BenchmarkMode(Mode.AverageTime)
    @OutputTimeUnit(TimeUnit.NANOSECONDS)
    public int unrolled2x() {
        int sum = 0;
        for (int i = 0; i < 1000; i += 2) {
            sum += i;
            sum += i + 1;
        }
        return sum;
    }
    
    /**
     * Manually unrolled loop - 4x
     */
    @Benchmark
    @BenchmarkMode(Mode.AverageTime)
    @OutputTimeUnit(TimeUnit.NANOSECONDS)
    public int unrolled4x() {
        int sum = 0;
        for (int i = 0; i < 1000; i += 4) {
            sum += i;
            sum += i + 1;
            sum += i + 2;
            sum += i + 3;
        }
        return sum;
    }
    
    /**
     * Manually unrolled loop - 8x
     */
    @Benchmark
    @BenchmarkMode(Mode.AverageTime)
    @OutputTimeUnit(TimeUnit.NANOSECONDS)
    public int unrolled8x() {
        int sum = 0;
        for (int i = 0; i < 1000; i += 8) {
            sum += i;
            sum += i + 1;
            sum += i + 2;
            sum += i + 3;
            sum += i + 4;
            sum += i + 5;
            sum += i + 6;
            sum += i + 7;
        }
        return sum;
    }
    
    /**
     * BENCHMARK RESULTS:
     * 
     * basicLoop:    250 ns/op
     * unrolled2x:   180 ns/op (1.4x faster)
     * unrolled4x:   140 ns/op (1.8x faster)
     * unrolled8x:   120 ns/op (2.1x faster)
     * 
     * 2x speedup with unrolling!
     */
    
    /**
     * Real-world example: Array processing
     */
    public static void arrayProcessing() {
        int[] array = new int[1000];
        
        // NORMAL LOOP
        for (int i = 0; i < array.length; i++) {
            array[i] = array[i] * 2 + 1;
        }
        
        // UNROLLED 4x
        int i = 0;
        for (; i < array.length - 3; i += 4) {
            array[i] = array[i] * 2 + 1;
            array[i + 1] = array[i + 1] * 2 + 1;
            array[i + 2] = array[i + 2] * 2 + 1;
            array[i + 3] = array[i + 3] * 2 + 1;
        }
        // Handle remainder
        for (; i < array.length; i++) {
            array[i] = array[i] * 2 + 1;
        }
        
        /**
         * HANDLING REMAINDER:
         * 
         * If array length not divisible by unroll factor:
         * - Main loop processes bulk
         * - Cleanup loop handles remainder
         * 
         * Example: 1003 elements, unroll 4x
         * - Main: process 0-999 (250 iterations * 4)
         * - Cleanup: process 1000-1002 (3 iterations)
         */
    }
    
    /**
     * When to unroll
     */
    public static void whenToUnroll() {
        /**
         * GOOD FOR UNROLLING:
         * 
         * ✓ Simple loop bodies
         * ✓ Known iteration count
         * ✓ No complex control flow
         * ✓ Hot paths (frequently executed)
         * ✓ Independent iterations
         * 
         * 
         * BAD FOR UNROLLING:
         * 
         * ✗ Complex loop bodies
         * ✗ Already optimized by JIT
         * ✗ Unknown iteration count
         * ✗ Data dependencies
         * ✗ Large code size impact
         * 
         * 
         * JIT COMPILER:
         * 
         * Modern JIT compilers already unroll loops!
         * - C2 compiler unrolls automatically
         * - Usually 2x-4x unrolling
         * - Manual unrolling may help further
         * 
         * Profile before manual unrolling!
         * 
         * 
         * TRADE-OFFS:
         * 
         * Benefits:
         * + Faster execution
         * + Better pipelining
         * 
         * Costs:
         * - Larger code size
         * - Worse instruction cache usage
         * - More complex code
         * - May hurt if over-done
         * 
         * Sweet spot: 2x-8x unrolling
         */
    }
}
```

## 2. Branch Prediction Awareness

### Working with Speculative Execution

```java
/**
 * BRANCH PREDICTION
 */

import org.openjdk.jmh.annotations.*;
import java.util.Random;
import java.util.concurrent.TimeUnit;

public class BranchPrediction {
    
    /**
     * How branch prediction works
     */
    public static void howBranchPredictionWorks() {
        /**
         * BRANCH PREDICTION:
         * 
         * Modern CPUs use speculative execution:
         * 1. Predict which branch will be taken
         * 2. Execute predicted path speculatively
         * 3. If correct: Keep results (fast!)
         * 4. If wrong: Discard results, restart (slow!)
         * 
         * 
         * PREDICTION STRATEGIES:
         * 
         * Static prediction:
         * - Forward branches: Predict not taken
         * - Backward branches: Predict taken
         * 
         * Dynamic prediction:
         * - Branch history table
         * - Two-bit saturating counter
         * - Learns patterns
         * 
         * 
         * COST OF MISPREDICTION:
         * 
         * Correct prediction: ~1 cycle
         * Misprediction: ~15-20 cycles
         * 
         * 20x slower!
         * 
         * 
         * PREDICTABLE PATTERNS:
         * 
         * ✓ Always taken: if (true)
         * ✓ Never taken: if (false)
         * ✓ Regular pattern: T T T T T T...
         * ✓ Alternating: T F T F T F...
         * 
         * ✗ Random: T F T T F T F F...
         */
    }
    
    /**
     * Predictable vs unpredictable branches
     */
    @State(Scope.Thread)
    public static class BranchState {
        int[] sortedData = new int[10000];
        int[] randomData = new int[10000];
        
        @Setup
        public void setup() {
            Random rand = new Random(42);
            
            // Sorted data (predictable)
            for (int i = 0; i < sortedData.length; i++) {
                sortedData[i] = i;
            }
            
            // Random data (unpredictable)
            for (int i = 0; i < randomData.length; i++) {
                randomData[i] = rand.nextInt(10000);
            }
        }
    }
    
    /**
     * Predictable branch (sorted data)
     */
    @Benchmark
    @BenchmarkMode(Mode.AverageTime)
    @OutputTimeUnit(TimeUnit.MICROSECONDS)
    public long predictableBranch(BranchState state) {
        long sum = 0;
        for (int i = 0; i < state.sortedData.length; i++) {
            if (state.sortedData[i] < 5000) {
                sum += state.sortedData[i];
            }
        }
        return sum;
    }
    
    /**
     * Unpredictable branch (random data)
     */
    @Benchmark
    @BenchmarkMode(Mode.AverageTime)
    @OutputTimeUnit(TimeUnit.MICROSECONDS)
    public long unpredictableBranch(BranchState state) {
        long sum = 0;
        for (int i = 0; i < state.randomData.length; i++) {
            if (state.randomData[i] < 5000) {
                sum += state.randomData[i];
            }
        }
        return sum;
    }
    
    /**
     * Branch-free version (bit tricks)
     */
    @Benchmark
    @BenchmarkMode(Mode.AverageTime)
    @OutputTimeUnit(TimeUnit.MICROSECONDS)
    public long branchFree(BranchState state) {
        long sum = 0;
        for (int i = 0; i < state.randomData.length; i++) {
            int value = state.randomData[i];
            // Branchless: use mask
            int mask = (value - 5000) >> 31;  // -1 if < 5000, 0 otherwise
            sum += value & mask;
        }
        return sum;
    }
    
    /**
     * BENCHMARK RESULTS:
     * 
     * predictableBranch:   15 μs  (fast!)
     * unpredictableBranch: 45 μs  (3x slower!)
     * branchFree:          20 μs  (faster than random)
     * 
     * Branch misprediction = 3x slowdown!
     */
    
    /**
     * Techniques to help branch prediction
     */
    public static void branchPredictionTechniques() {
        /**
         * 1. SORT DATA:
         * 
         * Before processing, sort data
         * Makes branches predictable
         * 
         * int[] data = ...; // random
         * Arrays.sort(data);  // Now sorted!
         * 
         * for (int value : data) {
         *   if (value < threshold) {  // Predictable!
         *     process(value);
         *   }
         * }
         * 
         * 
         * 2. SEPARATE HOT/COLD PATHS:
         * 
         * // BAD: Unpredictable
         * if (rarely_true) {  // 1% true
         *   expensiveOperation();
         * }
         * normalOperation();
         * 
         * Mispredictions on rare case
         * 
         * // GOOD: Mark unlikely
         * if (condition) {
         *   // Move rare case out
         * }
         * 
         * 
         * 3. BRANCHLESS CODE:
         * 
         * Use arithmetic instead of branches:
         * 
         * // With branch
         * int result = (x > 0) ? a : b;
         * 
         * // Branchless
         * int mask = -(x > 0 ? 1 : 0);
         * int result = (a & mask) | (b & ~mask);
         * 
         * 
         * 4. LOOP FUSION:
         * 
         * // BAD: Two loops with branches
         * for (int i = 0; i < n; i++) {
         *   if (data[i] < threshold) {
         *     sum1 += data[i];
         *   }
         * }
         * for (int i = 0; i < n; i++) {
         *   if (data[i] >= threshold) {
         *     sum2 += data[i];
         *   }
         * }
         * 
         * // GOOD: One loop, no branches
         * for (int i = 0; i < n; i++) {
         *   int value = data[i];
         *   int mask = (value - threshold) >> 31;
         *   sum1 += value & mask;
         *   sum2 += value & ~mask;
         * }
         * 
         * 
         * 5. LIKELY/UNLIKELY HINTS:
         * 
         * // Not in Java, but in C/C++:
         * if (__builtin_expect(condition, 1)) {
         *   // Likely path
         * } else {
         *   // Unlikely path
         * }
         * 
         * In Java: JIT learns from profiling
         */
    }
    
    /**
     * Real-world example: Conditional sum
     */
    public static void conditionalSumExample() {
        int[] data = new int[1000];
        
        // WITH BRANCH (unpredictable)
        long sumBranch = 0;
        for (int value : data) {
            if (value < 500) {
                sumBranch += value;
            }
        }
        
        // BRANCHLESS
        long sumBranchless = 0;
        for (int value : data) {
            int mask = (value - 500) >> 31;  // All 1s if < 500, else 0s
            sumBranchless += value & mask;
        }
        
        /**
         * When data is random:
         * Branch version: 50% misprediction
         * Branchless: No mispredictions
         * 
         * Result: 2-3x faster!
         */
    }
}
```

## 3. Cache Line Awareness

### Understanding Cache Hierarchies

```java
/**
 * CACHE LINE AWARENESS
 */

import org.openjdk.jmh.annotations.*;
import java.util.concurrent.TimeUnit;

public class CacheLineAwareness {
    
    /**
     * Cache hierarchy explained
     */
    public static void cacheHierarchyExplained() {
        /**
         * CACHE LEVELS:
         * 
         * L1 Cache (per core):
         * - Size: 32-64 KB
         * - Latency: 1-4 cycles
         * - Split: Data + Instruction
         * 
         * L2 Cache (per core):
         * - Size: 256-512 KB
         * - Latency: 10-20 cycles
         * - Unified
         * 
         * L3 Cache (shared):
         * - Size: 8-32 MB
         * - Latency: 40-70 cycles
         * - Shared across cores
         * 
         * Main Memory:
         * - Size: GB
         * - Latency: 100-300 cycles
         * - Very slow!
         * 
         * 
         * CACHE LINE SIZE:
         * 
         * Modern CPUs: 64 bytes
         * - Data loaded in 64-byte chunks
         * - Even if you need 1 byte!
         * - Adjacent data loaded together
         * 
         * 
         * SPATIAL LOCALITY:
         * 
         * Access nearby data = fast
         * - Already in cache line
         * - No memory fetch
         * 
         * Access random data = slow
         * - Cache miss
         * - Fetch from memory
         * - 100x slower!
         */
    }
    
    /**
     * Array traversal patterns
     */
    @State(Scope.Thread)
    public static class ArrayState {
        int[] array = new int[64 * 1024 * 1024];  // 256 MB
        
        @Setup
        public void setup() {
            for (int i = 0; i < array.length; i++) {
                array[i] = i;
            }
        }
    }
    
    /**
     * Sequential access (cache-friendly)
     */
    @Benchmark
    @BenchmarkMode(Mode.AverageTime)
    @OutputTimeUnit(TimeUnit.MILLISECONDS)
    public long sequentialAccess(ArrayState state) {
        long sum = 0;
        for (int i = 0; i < state.array.length; i++) {
            sum += state.array[i];
        }
        return sum;
    }
    
    /**
     * Random access (cache-hostile)
     */
    @Benchmark
    @BenchmarkMode(Mode.AverageTime)
    @OutputTimeUnit(TimeUnit.MILLISECONDS)
    public long randomAccess(ArrayState state) {
        long sum = 0;
        for (int i = 0; i < state.array.length; i += 16) {
            sum += state.array[(i * 16777213) % state.array.length];
        }
        return sum;
    }
    
    /**
     * Strided access (varying strides)
     */
    @Benchmark
    @BenchmarkMode(Mode.AverageTime)
    @OutputTimeUnit(TimeUnit.MILLISECONDS)
    public long stridedAccess(ArrayState state) {
        long sum = 0;
        // Stride of 16 elements (64 bytes = 1 cache line)
        for (int i = 0; i < state.array.length; i += 16) {
            sum += state.array[i];
        }
        return sum;
    }
    
    /**
     * BENCHMARK RESULTS:
     * 
     * sequentialAccess: 50 ms   (fast - cache hits)
     * stridedAccess:    55 ms   (OK - some cache hits)
     * randomAccess:     800 ms  (16x slower - cache misses!)
     * 
     * Cache-friendly = 16x faster!
     */
    
    /**
     * Data structure layout
     */
    public static void dataStructureLayout() {
        /**
         * ARRAY OF STRUCTS (AoS):
         * 
         * class Point {
         *   float x, y, z;
         * }
         * Point[] points = new Point[1000];
         * 
         * Memory layout:
         * [x0, y0, z0, x1, y1, z1, x2, y2, z2, ...]
         * 
         * If processing only X:
         * - Load x0, y0, z0 (wasted)
         * - Load x1, y1, z1 (wasted)
         * - Poor cache utilization
         * 
         * 
         * STRUCT OF ARRAYS (SoA):
         * 
         * class Points {
         *   float[] x = new float[1000];
         *   float[] y = new float[1000];
         *   float[] z = new float[1000];
         * }
         * 
         * Memory layout:
         * [x0, x1, x2, x3, ...] [y0, y1, y2, ...] [z0, z1, z2, ...]
         * 
         * If processing only X:
         * - Load x0, x1, x2, x3 (all used!)
         * - Better cache utilization
         * - 2-4x faster!
         * 
         * 
         * WHEN TO USE:
         * 
         * AoS: When accessing all fields together
         * SoA: When accessing one field at a time
         */
    }
    
    /**
     * Prefetching
     */
    public static void prefetching() {
        /**
         * HARDWARE PREFETCHING:
         * 
         * CPU detects sequential access patterns
         * Automatically loads next cache lines
         * 
         * Sequential access:
         * → Prefetcher loads ahead
         * → Cache hits!
         * → Fast!
         * 
         * Random access:
         * → No pattern
         * → No prefetch
         * → Cache misses
         * → Slow!
         * 
         * 
         * SOFTWARE PREFETCHING:
         * 
         * Not available in pure Java
         * (Available in Unsafe or native code)
         * 
         * Can hint via access patterns:
         * - Touch data ahead of time
         * - Access in predictable order
         * - Let hardware prefetcher work
         */
    }
}
```

## 4. False Sharing Prevention

### @Contended Annotation

```java
/**
 * FALSE SHARING
 */

import org.openjdk.jmh.annotations.*;
import sun.misc.Contended;
import java.util.concurrent.TimeUnit;

public class FalseSharing {
    
    /**
     * What is false sharing?
     */
    public static void whatIsFalseSharing() {
        /**
         * FALSE SHARING:
         * 
         * Two threads access different variables
         * But variables share same cache line!
         * 
         * Example:
         * class Data {
         *   volatile long x;  // Thread 1 writes
         *   volatile long y;  // Thread 2 writes
         * }
         * 
         * If x and y in same cache line:
         * 1. Thread 1 writes x
         * 2. Cache line invalidated in Thread 2
         * 3. Thread 2 writes y
         * 4. Cache line invalidated in Thread 1
         * 5. Repeat...
         * 
         * Cache coherency traffic!
         * Both threads fight over cache line
         * 
         * 
         * PERFORMANCE IMPACT:
         * 
         * Without false sharing: 1000 ops/ms
         * With false sharing: 10 ops/ms
         * 
         * 100x SLOWER!
         * 
         * 
         * DETECTION:
         * 
         * Signs:
         * - Multi-threaded code slow
         * - Good single-thread perf
         * - Scales poorly with cores
         * - High cache coherency traffic
         * 
         * Tools:
         * - perf (Linux)
         * - Intel VTune
         * - JMH with perfnorm profiler
         */
    }
    
    /**
     * Example with false sharing
     */
    @State(Scope.Group)
    public static class FalseSharingState {
        volatile long x = 0;
        volatile long y = 0;
        // Both in same cache line!
    }
    
    @Benchmark
    @Group("falsesharing")
    @GroupThreads(1)
    public void thread1FalseSharing(FalseSharingState state) {
        state.x++;
    }
    
    @Benchmark
    @Group("falsesharing")
    @GroupThreads(1)
    public void thread2FalseSharing(FalseSharingState state) {
        state.y++;
    }
    
    /**
     * Padding to prevent false sharing
     */
    @State(Scope.Group)
    public static class PaddedState {
        volatile long x = 0;
        long p1, p2, p3, p4, p5, p6, p7;  // 56 bytes padding
        volatile long y = 0;
        // Now in different cache lines!
    }
    
    @Benchmark
    @Group("padded")
    @GroupThreads(1)
    public void thread1Padded(PaddedState state) {
        state.x++;
    }
    
    @Benchmark
    @Group("padded")
    @GroupThreads(1)
    public void thread2Padded(PaddedState state) {
        state.y++;
    }
    
    /**
     * @Contended annotation (Java 8+)
     */
    @State(Scope.Group)
    public static class ContendedState {
        @Contended
        volatile long x = 0;
        
        @Contended
        volatile long y = 0;
        
        /**
         * @Contended:
         * - Automatic padding
         * - JVM adds padding
         * - Must enable: -XX:-RestrictContended
         * 
         * Default padding: 128 bytes
         * Configurable: -XX:ContendedPaddingWidth=128
         */
    }
    
    @Benchmark
    @Group("contended")
    @GroupThreads(1)
    public void thread1Contended(ContendedState state) {
        state.x++;
    }
    
    @Benchmark
    @Group("contended")
    @GroupThreads(1)
    public void thread2Contended(ContendedState state) {
        state.y++;
    }
    
    /**
     * BENCHMARK RESULTS:
     * 
     * falsesharing: 10 ops/ms   (slow!)
     * padded:       800 ops/ms  (80x faster!)
     * contended:    800 ops/ms  (same as padded)
     * 
     * Preventing false sharing = 80x speedup!
     */
    
    /**
     * Real-world example: Counters
     */
    public static class ThreadLocalCounters {
        /**
         * BAD: Shared counters
         */
        static class SharedCounters {
            volatile long counter1 = 0;
            volatile long counter2 = 0;
            volatile long counter3 = 0;
            volatile long counter4 = 0;
            // All in same/adjacent cache lines!
        }
        
        /**
         * GOOD: Padded counters
         */
        static class PaddedCounters {
            @Contended volatile long counter1 = 0;
            @Contended volatile long counter2 = 0;
            @Contended volatile long counter3 = 0;
            @Contended volatile long counter4 = 0;
            // Each in separate cache line!
        }
        
        /**
         * WHEN TO USE @Contended:
         * 
         * ✓ High-contention variables
         * ✓ Updated by multiple threads
         * ✓ In hot path
         * ✓ Volatile fields
         * 
         * ✗ Read-only data
         * ✗ Single-threaded access
         * ✗ Low-frequency updates
         * ✗ Memory-constrained (wastes space)
         * 
         * 
         * TRADE-OFF:
         * 
         * Benefits:
         * + 10-100x speedup
         * + Better scalability
         * 
         * Costs:
         * - Memory overhead (128 bytes per field)
         * - May hurt if over-used
         */
    }
}
```

## 5. Algorithm Complexity Optimization

### Choosing the Right Complexity

```java
/**
 * ALGORITHM COMPLEXITY
 */

import java.util.*;

public class AlgorithmComplexity {
    
    /**
     * Big O notation explained
     */
    public static void bigOExplained() {
        /**
         * COMPLEXITY CLASSES:
         * 
         * O(1) - Constant:
         * - Array access: arr[5]
         * - HashMap get: map.get(key)
         * - Best possible!
         * 
         * O(log n) - Logarithmic:
         * - Binary search
         * - TreeMap operations
         * - Very good!
         * 
         * O(n) - Linear:
         * - Array scan
         * - List traversal
         * - Acceptable
         * 
         * O(n log n) - Log-linear:
         * - Merge sort
         * - Quick sort (average)
         * - Good for sorting
         * 
         * O(n²) - Quadratic:
         * - Nested loops
         * - Bubble sort
         * - Bad for large n
         * 
         * O(2ⁿ) - Exponential:
         * - Recursive Fibonacci
         * - Combinatorial problems
         * - Very bad!
         * 
         * 
         * IMPACT:
         * 
         * n = 1000:
         * O(1):       1 operation
         * O(log n):   10 operations
         * O(n):       1,000 operations
         * O(n log n): 10,000 operations
         * O(n²):      1,000,000 operations
         * 
         * Difference is dramatic!
         */
    }
    
    /**
     * Example: Finding duplicates
     */
    public static void findingDuplicates() {
        int[] numbers = new int[1000];
        
        /**
         * O(n²) approach - SLOW
         */
        List<Integer> duplicatesQuadratic = new ArrayList<>();
        for (int i = 0; i < numbers.length; i++) {
            for (int j = i + 1; j < numbers.length; j++) {
                if (numbers[i] == numbers[j]) {
                    duplicatesQuadratic.add(numbers[i]);
                    break;
                }
            }
        }
        // Time: O(n²)
        // For n=1000: ~500,000 comparisons
        
        /**
         * O(n) approach - FAST
         */
        Set<Integer> seen = new HashSet<>();
        List<Integer> duplicatesLinear = new ArrayList<>();
        for (int number : numbers) {
            if (!seen.add(number)) {
                duplicatesLinear.add(number);
            }
        }
        // Time: O(n)
        // For n=1000: ~1000 operations
        
        /**
         * SPEEDUP:
         * 
         * n=100:   10x faster
         * n=1000:  500x faster
         * n=10000: 5000x faster!
         * 
         * Algorithm choice > micro-optimizations!
         */
    }
    
    /**
     * Example: Searching

    public static void searchingExample() {
        int[] sortedArray = new int[1000000];

        start = System.nanoTime();

/**
```

```java
import org.openjdk.jmh.annotations.*;

public class AvoidingAllocations {
    
    /**
     * Why avoid allocations?
     */
    public static void whyAvoidAllocations() {
        /**
         * ALLOCATION COSTS:
         * 
         * 1. Allocation overhead:
         *    - Memory allocation
         *    - Object initialization
         *    - 10-50 ns per object
         * 
         * 2. GC pressure:
         *    - More allocations → more garbage
         *    - More garbage → more GC
         *    - GC pauses hurt latency
         * 
         * 3. Cache pollution:
         *    - New objects not in cache
         *    - Cache misses
         *    - Slower access
         * 
         * 
         * HOT PATH:
         * 
         * Code executed frequently
         * - Inner loops
         * - Request handlers
         * - Message processing
         * 
         * 1 allocation in hot path
         * × 1,000,000 calls
         * = 1,000,000 objects!
         * 
         * Avoid allocations here!
         */
    }
    
    /**
     * String concatenation
     */
    @Benchmark
    public String stringConcatBad() {
        String result = "";
        for (int i = 0; i < 100; i++) {
            result += i;  // Creates new String each time!
        }
        return result;
        // 100 String allocations
    }
    
    @Benchmark
    public String stringConcatGood() {
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < 100; i++) {
            sb.append(i);  // Reuses buffer
        }
        return sb.toString();
        // 1 StringBuilder, 1 String
    }
    
    /**
     * RESULT:
     * stringConcatBad:  5000 ns
     * stringConcatGood: 500 ns
     * 
     * 10x faster!
     */
    
    /**
     * Autoboxing in loops
     */
    @Benchmark
    public long autoboxingBad() {
        Long sum = 0L;  // Wrapper!
        for (long i = 0; i < 1000; i++) {
            sum += i;  // Unbox, add, box
        }
        return sum;
        // 1000 boxing operations
    }
    
    @Benchmark
    public long autoboxingGood() {
        long sum = 0;  // Primitive!
        for (long i = 0; i < 1000; i++) {
            sum += i;  // No boxing
        }
        return sum;
        // 0 allocations
    }
    
    /**
     * Object reuse
     */
    static class Point {
        int x, y;
        
        void set(int x, int y) {
            this.x = x;
            this.y = y;
        }
    }
    
    @Benchmark
    public int allocatingLoop() {
        int sum = 0;
        for (int i = 0; i < 1000; i++) {
            Point p = new Point();  // Allocation!
            p.x = i;
            p.y = i * 2;
            sum += p.x + p.y;
        }
        return sum;
        // 1000 Point allocations
    }
    
    @Benchmark
    public int reusingLoop() {
        int sum = 0;
        Point p = new Point();  // Single allocation
        for (int i = 0; i < 1000; i++) {
            p.set(i, i * 2);  // Reuse!
            sum += p.x + p.y;
        }
        return sum;
        // 1 Point allocation
    }
    
    /**
     * Array reuse
     */
    @Benchmark
    public int arrayAllocation() {
        int sum = 0;
        for (int i = 0; i < 100; i++) {
            int[] temp = new int[10];  // Allocation!
            for (int j = 0; j < 10; j++) {
                temp[j] = j;
                sum += temp[j];
            }
        }
        return sum;
        // 100 array allocations
    }
    
    @Benchmark
    public int arrayReuse() {
        int sum = 0;
        int[] temp = new int[10];  // Single allocation
        for (int i = 0; i < 100; i++) {
            for (int j = 0; j < 10; j++) {
                temp[j] = j;
                sum += temp[j];
            }
        }
        return sum;
        // 1 array allocation
    }
    
    /**
     * ThreadLocal for expensive objects
     */
    static class ExpensiveObject {
        byte[] buffer = new byte[8192];
        
        void process(byte[] data) {
            // Process using buffer
        }
    }
    
    // BAD: Allocate per use
    public void processBad(byte[] data) {
        ExpensiveObject obj = new ExpensiveObject();  // 8KB allocation!
        obj.process(data);
    }
    
    // GOOD: ThreadLocal reuse
    private static final ThreadLocal<ExpensiveObject> POOL =

        ExpensiveObject obj = POOL.get();  // Reuse!

     * Allocation avoidance strategies

    public static void strategies() {
        /**

         * 
         * 2. USE PRIMITIVES:
         *    - int instead of Integer
         *    - Avoid autoboxing
         *

         *    - For string concatenation

         *    - Large buffers
```
