# Proxy Pattern and Dynamic Proxies

## Overview

The Proxy pattern provides a surrogate or placeholder for another object to control access to it. Java's dynamic proxies allow creating proxy instances at runtime without code generation. This guide covers JDK dynamic proxies, InvocationHandler, proxy limitations, CGLIB for class proxies, ByteBuddy for advanced bytecode manipulation, Spring AOP mechanisms, and performance considerations essential for building frameworks and AOP systems.

## 1. java.lang.reflect.Proxy

### JDK Dynamic Proxy Basics

```java
/**
 * JAVA.LANG.REFLECT.PROXY
 * 
 * JDK dynamic proxy creation
 */

import java.lang.reflect.*;

public class JDKDynamicProxy {
    
    /**
     * Simple interface for proxying
     */
    interface Calculator {
        int add(int a, int b);
        int subtract(int a, int b);
        int multiply(int a, int b);
    }
    
    /**
     * Real implementation
     */
    static class CalculatorImpl implements Calculator {
        @Override
        public int add(int a, int b) {
            return a + b;
        }
        
        @Override
        public int subtract(int a, int b) {
            return a - b;
        }
        
        @Override
        public int multiply(int a, int b) {
            return a * b;
        }
    }
    
    /**
     * Basic InvocationHandler
     */
    static class LoggingHandler implements InvocationHandler {
        private final Object target;
        
        public LoggingHandler(Object target) {
            this.target = target;
        }
        
        @Override
        public Object invoke(Object proxy, Method method, Object[] args) 
                throws Throwable {
            System.out.println("Before: " + method.getName());
            
            try {
                Object result = method.invoke(target, args);
                System.out.println("After: " + method.getName() + " = " + result);
                return result;
            } catch (InvocationTargetException e) {
                System.out.println("Exception in: " + method.getName());
                throw e.getCause();
            }
        }
    }
    
    /**
     * Creating a proxy
     */
    public static void basicProxyCreation() {
        Calculator calculator = new CalculatorImpl();
        
        // Create proxy
        Calculator proxy = (Calculator) Proxy.newProxyInstance(
            Calculator.class.getClassLoader(),
            new Class<?>[] { Calculator.class },
            new LoggingHandler(calculator)
        );
        
        System.out.println("=== Basic Proxy ===");
        int result = proxy.add(5, 3);
        System.out.println("Client received: " + result);
        
        /**
         * PROXY.NEWPROXYINSTANCE():
         * 
         * Parameters:
         * 1. ClassLoader - loads proxy class
         * 2. Class<?>[] - interfaces to implement
         * 3. InvocationHandler - method interceptor
         * 
         * Returns:
         * - Proxy instance implementing interfaces
         * - All methods route through InvocationHandler
         * 
         * Generated class name:
         * $Proxy0, $Proxy1, etc.
         */
    }
    
    /**
     * Proxy class inspection
     */
    public static void proxyClassInspection() {
        Calculator calculator = new CalculatorImpl();
        
        Calculator proxy = (Calculator) Proxy.newProxyInstance(
            Calculator.class.getClassLoader(),
            new Class<?>[] { Calculator.class },
            new LoggingHandler(calculator)
        );
        
        System.out.println("\n=== Proxy Inspection ===");
        
        // Check if proxy
        System.out.println("Is proxy: " + Proxy.isProxyClass(proxy.getClass()));
        System.out.println("Proxy class: " + proxy.getClass().getName());
        
        // Get InvocationHandler
        InvocationHandler handler = Proxy.getInvocationHandler(proxy);
        System.out.println("Handler: " + handler.getClass().getName());
        
        // Get interfaces
        Class<?>[] interfaces = proxy.getClass().getInterfaces();
        System.out.println("Interfaces: ");
        for (Class<?> iface : interfaces) {
            System.out.println("  " + iface.getName());
        }
        
        /**
         * PROXY UTILITY METHODS:
         * 
         * Proxy.isProxyClass(Class):
         * - Check if class is proxy
         * 
         * Proxy.getInvocationHandler(Object):
         * - Get handler from proxy
         * - Throws IllegalArgumentException if not proxy
         * 
         * Proxy.getProxyClass(ClassLoader, Class<?>...):
         * - Get proxy Class without instance
         */
    }
    
    /**
     * Multiple interfaces
     */
    interface Auditable {
        void audit();
    }
    
    static class AuditableCalculatorImpl extends CalculatorImpl implements Auditable {
        @Override
        public void audit() {
            System.out.println("Auditing...");
        }
    }
    
    public static void multipleInterfacesProxy() {
        AuditableCalculatorImpl calculator = new AuditableCalculatorImpl();
        
        // Proxy implementing multiple interfaces
        Object proxy = Proxy.newProxyInstance(
            Calculator.class.getClassLoader(),
            new Class<?>[] { Calculator.class, Auditable.class },
            new LoggingHandler(calculator)
        );
        
        System.out.println("\n=== Multiple Interfaces ===");
        
        // Use as Calculator
        Calculator calcProxy = (Calculator) proxy;
        calcProxy.add(1, 2);
        
        // Use as Auditable
        Auditable auditProxy = (Auditable) proxy;
        auditProxy.audit();
        
        /**
         * MULTIPLE INTERFACES:
         * 
         * Proxy can implement multiple interfaces
         * Single InvocationHandler for all
         * Cast to any implemented interface
         * 
         * Useful for:
         * - Combining concerns
         * - Adapter pattern
         * - Facade pattern
         */
    }
    
    /**
     * Proxy without target (virtual proxy)
     */
    static class VirtualProxyHandler implements InvocationHandler {
        private Object target;
        private final Class<?> targetClass;
        
        public VirtualProxyHandler(Class<?> targetClass) {
            this.targetClass = targetClass;
        }
        
        private synchronized Object getTarget() {
            if (target == null) {
                System.out.println("Lazy initialization of target");
                try {
                    target = targetClass.getDeclaredConstructor().newInstance();
                } catch (Exception e) {
                    throw new RuntimeException(e);
                }
            }
            return target;
        }
        
        @Override
        public Object invoke(Object proxy, Method method, Object[] args) 
                throws Throwable {
            return method.invoke(getTarget(), args);
        }
    }
    
    public static void virtualProxy() {
        System.out.println("\n=== Virtual Proxy ===");
        
        Calculator proxy = (Calculator) Proxy.newProxyInstance(
            Calculator.class.getClassLoader(),
            new Class<?>[] { Calculator.class },
            new VirtualProxyHandler(CalculatorImpl.class)
        );
        
        System.out.println("Proxy created, target not initialized");
        System.out.println("Calling add...");
        proxy.add(1, 2);
        
        /**
         * VIRTUAL PROXY:
         * 
         * Delays object creation
         * Creates on first use
         * Lazy initialization
         * 
         * Use cases:
         * - Expensive object creation
         * - Resource loading
         * - Remote proxies
         */
    }
    
    public static void main(String[] args) {
        basicProxyCreation();
        proxyClassInspection();
        multipleInterfacesProxy();
        virtualProxy();
    }
}
```

## 2. InvocationHandler Implementation

### Method Interception Patterns

```java
/**
 * INVOCATIONHANDLER IMPLEMENTATION
 * 
 * Various InvocationHandler patterns
 */

import java.lang.reflect.*;
import java.util.*;
import java.util.concurrent.*;

public class InvocationHandlerPatterns {
    
    interface UserService {
        void createUser(String name);
        String getUser(int id);
        void deleteUser(int id);
    }
    
    static class UserServiceImpl implements UserService {
        @Override
        public void createUser(String name) {
            System.out.println("Creating user: " + name);
        }
        
        @Override
        public String getUser(int id) {
            return "User-" + id;
        }
        
        @Override
        public void deleteUser(int id) {
            System.out.println("Deleting user: " + id);
        }
    }
    
    /**
     * Performance monitoring handler
     */
    static class PerformanceMonitorHandler implements InvocationHandler {
        private final Object target;
        private final Map<String, Long> methodTimes = new ConcurrentHashMap<>();
        
        public PerformanceMonitorHandler(Object target) {
            this.target = target;
        }
        
        @Override
        public Object invoke(Object proxy, Method method, Object[] args) 
                throws Throwable {
            long start = System.nanoTime();
            
            try {
                return method.invoke(target, args);
            } finally {
                long elapsed = System.nanoTime() - start;
                methodTimes.merge(method.getName(), elapsed, Long::sum);
            }
        }
        
        public void printStats() {
            System.out.println("\n=== Performance Stats ===");
            methodTimes.forEach((method, time) -> 
                System.out.println(method + ": " + time / 1_000_000 + " ms"));
        }
    }
    
    /**
     * Caching handler
     */
    static class CachingHandler implements InvocationHandler {
        private final Object target;
        private final Map<String, Object> cache = new ConcurrentHashMap<>();
        
        public CachingHandler(Object target) {
            this.target = target;
        }
        
        @Override
        public Object invoke(Object proxy, Method method, Object[] args) 
                throws Throwable {
            // Only cache if method has no parameters (simple example)
            if (args == null || args.length == 0) {
                String cacheKey = method.getName();
                
                return cache.computeIfAbsent(cacheKey, k -> {
                    try {
                        System.out.println("Cache miss: " + method.getName());
                        return method.invoke(target, args);
                    } catch (Exception e) {
                        throw new RuntimeException(e);
                    }
                });
            }
            
            // Not cacheable - call directly
            return method.invoke(target, args);
        }
    }
    
    /**
     * Security handler
     */
    @interface RequiresPermission {
        String value();
    }
    
    interface SecureService {
        void publicMethod();
        
        @RequiresPermission("ADMIN")
        void adminMethod();
    }
    
    static class SecureServiceImpl implements SecureService {
        @Override
        public void publicMethod() {
            System.out.println("Public method executed");
        }
        
        @Override
        public void adminMethod() {
            System.out.println("Admin method executed");
        }
    }
    
    static class SecurityHandler implements InvocationHandler {
        private final Object target;
        private final Set<String> permissions;
        
        public SecurityHandler(Object target, Set<String> permissions) {
            this.target = target;
            this.permissions = permissions;
        }
        
        @Override
        public Object invoke(Object proxy, Method method, Object[] args) 
                throws Throwable {
            // Check permission
            RequiresPermission annotation = method.getAnnotation(RequiresPermission.class);
            
            if (annotation != null) {
                String required = annotation.value();
                if (!permissions.contains(required)) {
                    throw new SecurityException(
                        "Permission denied: " + required + " required");
                }
            }
            
            return method.invoke(target, args);
        }
    }
    
    public static void securityHandlerExample() {
        System.out.println("=== Security Handler ===");
        
        SecureService service = new SecureServiceImpl();
        
        // User without ADMIN permission
        SecureService userProxy = (SecureService) Proxy.newProxyInstance(
            SecureService.class.getClassLoader(),
            new Class<?>[] { SecureService.class },
            new SecurityHandler(service, Set.of("USER"))
        );
        
        userProxy.publicMethod();  // OK
        
        try {
            userProxy.adminMethod();  // Should fail
        } catch (SecurityException e) {
            System.out.println("Caught: " + e.getMessage());
        }
        
        // Admin user
        SecureService adminProxy = (SecureService) Proxy.newProxyInstance(
            SecureService.class.getClassLoader(),
            new Class<?>[] { SecureService.class },
            new SecurityHandler(service, Set.of("USER", "ADMIN"))
        );
        
        adminProxy.adminMethod();  // OK
    }
    
    /**
     * Transaction handler
     */
    @interface Transactional {}
    
    interface TransactionalService {
        void normalMethod();
        
        @Transactional
        void transactionalMethod();
    }
    
    static class TransactionalServiceImpl implements TransactionalService {
        @Override
        public void normalMethod() {
            System.out.println("Normal execution");
        }
        
        @Override
        public void transactionalMethod() {
            System.out.println("Transactional execution");
        }
    }
    
    static class TransactionHandler implements InvocationHandler {
        private final Object target;
        
        public TransactionHandler(Object target) {
            this.target = target;
        }
        
        @Override
        public Object invoke(Object proxy, Method method, Object[] args) 
                throws Throwable {
            boolean transactional = method.isAnnotationPresent(Transactional.class);
            
            if (transactional) {
                System.out.println("  BEGIN TRANSACTION");
            }
            
            try {
                Object result = method.invoke(target, args);
                
                if (transactional) {
                    System.out.println("  COMMIT TRANSACTION");
                }
                
                return result;
            } catch (Exception e) {
                if (transactional) {
                    System.out.println("  ROLLBACK TRANSACTION");
                }
                throw e;
            }
        }
    }
    
    public static void transactionHandlerExample() {
        System.out.println("\n=== Transaction Handler ===");
        
        TransactionalService service = new TransactionalServiceImpl();
        TransactionalService proxy = (TransactionalService) Proxy.newProxyInstance(
            TransactionalService.class.getClassLoader(),
            new Class<?>[] { TransactionalService.class },
            new TransactionHandler(service)
        );
        
        proxy.normalMethod();
        proxy.transactionalMethod();
    }
    
    /**
     * Chaining handlers (decorator pattern)
     */
    static class ChainedHandler implements InvocationHandler {
        private final InvocationHandler next;
        
        public ChainedHandler(InvocationHandler next) {
            this.next = next;
        }
        
        @Override
        public Object invoke(Object proxy, Method method, Object[] args) 
                throws Throwable {
            // Pre-processing
            System.out.println("  [Chain] Before");
            
            try {
                // Delegate to next handler
                return next.invoke(proxy, method, args);
            } finally {
                // Post-processing
                System.out.println("  [Chain] After");
            }
        }
    }
    
    /**
     * Method dispatch optimization
     */
    static class OptimizedHandler implements InvocationHandler {
        private final Object target;
        private final Map<Method, Method> methodCache = new ConcurrentHashMap<>();
        
        public OptimizedHandler(Object target) {
            this.target = target;
        }
        
        @Override
        public Object invoke(Object proxy, Method method, Object[] args) 
                throws Throwable {
            // Cache method lookup
            Method targetMethod = methodCache.computeIfAbsent(method, m -> {
                try {
                    return target.getClass().getMethod(m.getName(), m.getParameterTypes());
                } catch (NoSuchMethodException e) {
                    throw new RuntimeException(e);
                }
            });
            
            return targetMethod.invoke(target, args);
        }
    }
    
    /**
     * Handling equals/hashCode/toString
     */
    static class ProperHandler implements InvocationHandler {
        private final Object target;
        
        public ProperHandler(Object target) {
            this.target = target;
        }
        
        @Override
        public Object invoke(Object proxy, Method method, Object[] args) 
                throws Throwable {
            String methodName = method.getName();
            
            // Handle Object methods specially
            if ("equals".equals(methodName) && args.length == 1) {
                return proxyEquals(proxy, args[0]);
            }
            
            if ("hashCode".equals(methodName) && args.length == 0) {
                return proxyHashCode(proxy);
            }
            
            if ("toString".equals(methodName) && args.length == 0) {
                return proxyToString(proxy);
            }
            
            // Delegate to target
            return method.invoke(target, args);
        }
        
        private boolean proxyEquals(Object proxy, Object other) {
            if (proxy == other) return true;
            if (!Proxy.isProxyClass(other.getClass())) return false;
            
            InvocationHandler otherHandler = Proxy.getInvocationHandler(other);
            if (!(otherHandler instanceof ProperHandler)) return false;
            
            return target.equals(((ProperHandler) otherHandler).target);
        }
        
        private int proxyHashCode(Object proxy) {
            return target.hashCode();
        }
        
        private String proxyToString(Object proxy) {
            return "Proxy[" + target + "]";
        }
        
        /**
         * HANDLING OBJECT METHODS:
         * 
         * equals/hashCode/toString from Object
         * Go through InvocationHandler too!
         * 
         * Options:
         * 1. Delegate to target
         * 2. Implement on proxy
         * 3. Special handling
         * 
         * Be consistent with proxy semantics
         */
    }
    
    public static void main(String[] args) {
        securityHandlerExample();
        transactionHandlerExample();
    }
}
```

## 3. Proxy Limitations (Interfaces Only)

### Understanding JDK Proxy Constraints

```java
/**
 * PROXY LIMITATIONS
 * 
 * JDK dynamic proxy constraints
 */

import java.lang.reflect.*;

public class ProxyLimitations {
    
    /**
     * Interface requirement
     */
    static class ConcreteClass {
        public void method() {
            System.out.println("Concrete method");
        }
    }
    
    public static void interfaceRequirement() {
        System.out.println("=== Interface Requirement ===");
        
        try {
            // This FAILS - no interfaces
            ConcreteClass proxy = (ConcreteClass) Proxy.newProxyInstance(
                ConcreteClass.class.getClassLoader(),
                new Class<?>[] { ConcreteClass.class },
                (p, m, args) -> null
            );
        } catch (IllegalArgumentException e) {
            System.out.println("Cannot proxy class: " + e.getMessage());
        }
        
        /**
         * JDK PROXY LIMITATIONS:
         * 
         * 1. INTERFACES ONLY:
         *    Cannot proxy classes directly
         *    Must have interface
         * 
         * 2. PUBLIC INTERFACES:
         *    Non-public interfaces have restrictions
         * 
         * 3. FINAL METHODS:
         *    Can't override final methods
         *    (Not relevant for interfaces)
         * 
         * 4. OBJECT METHODS:
         *    equals/hashCode/toString go through handler
         *    Be careful with implementations
         */
    }
    
    /**
     * Non-public interface restriction
     */
    interface PublicInterface {
        void publicMethod();
    }
    
    // Package-private interface
    interface PackageInterface {
        void packageMethod();
    }
    
    public static void nonPublicInterfaceRestriction() {
        System.out.println("\n=== Non-Public Interface ===");
        
        // Public interface - OK
        PublicInterface proxy1 = (PublicInterface) Proxy.newProxyInstance(
            PublicInterface.class.getClassLoader(),
            new Class<?>[] { PublicInterface.class },
            (p, m, args) -> null
        );
        
        // Package interface - OK in same package
        // But restrictions on ClassLoader
        try {
            PackageInterface proxy2 = (PackageInterface) Proxy.newProxyInstance(
                PackageInterface.class.getClassLoader(),
                new Class<?>[] { PackageInterface.class },
                (p, m, args) -> null
            );
            System.out.println("Package interface proxy OK");
        } catch (IllegalArgumentException e) {
            System.out.println("Package interface issue: " + e.getMessage());
        }
        
        /**
         * NON-PUBLIC INTERFACES:
         * 
         * Must be in same package as proxy
         * ClassLoader restrictions apply
         * 
         * Rule: All non-public interfaces must be
         * in same package
         */
    }
    
    /**
     * ClassLoader restrictions
     */
    public static void classLoaderRestrictions() {
        /**
         * CLASSLOADER RULES:
         * 
         * 1. All interfaces visible to ClassLoader
         * 2. Non-public interfaces in same package
         * 3. No duplicate methods (different interfaces)
         * 4. Return types must be compatible
         * 
         * 
         * CHOOSING CLASSLOADER:
         * 
         * - Interface ClassLoader if public
         * - Custom ClassLoader for non-public
         * - System ClassLoader as fallback
         * 
         * 
         * COMMON ISSUE:
         * 
         * Different ClassLoaders for interfaces
         * → IllegalArgumentException
         * 
         * Solution:
         * Use common parent ClassLoader
         */
    }
    
    /**
     * Performance overhead
     */
    static class DirectCall {
        public int add(int a, int b) {
            return a + b;
        }
    }
    
    interface CalculatorInterface {
        int add(int a, int b);
    }
    
    static class CalculatorImpl implements CalculatorInterface {
        @Override
        public int add(int a, int b) {
            return a + b;
        }
    }
    
    public static void performanceOverhead() {
        int iterations = 10_000_000;
        
        // Direct call
        DirectCall direct = new DirectCall();
        long start = System.nanoTime();
        for (int i = 0; i < iterations; i++) {
            direct.add(1, 2);
        }
        long directTime = System.nanoTime() - start;
        
        // Interface call (no proxy)
        CalculatorInterface impl = new CalculatorImpl();
        start = System.nanoTime();
        for (int i = 0; i < iterations; i++) {
            impl.add(1, 2);
        }
        long interfaceTime = System.nanoTime() - start;
        
        // Proxy call
        CalculatorInterface proxy = (CalculatorInterface) Proxy.newProxyInstance(
            CalculatorInterface.class.getClassLoader(),
            new Class<?>[] { CalculatorInterface.class },
            (p, m, args) -> m.invoke(impl, args)
        );
        
        start = System.nanoTime();
        for (int i = 0; i < iterations; i++) {
            proxy.add(1, 2);
        }
        long proxyTime = System.nanoTime() - start;
        
        System.out.println("\n=== Performance (" + iterations + " iterations) ===");
        System.out.println("Direct call:     " + directTime / 1_000_000 + " ms");
        System.out.println("Interface call:  " + interfaceTime / 1_000_000 + " ms");
        System.out.println("Proxy call:      " + proxyTime / 1_000_000 + " ms");
        System.out.println("Proxy overhead:  " + (proxyTime / (double) directTime) + "x");
        
        /**
         * TYPICAL RESULTS:
         * 
         * Direct:    ~10 ms
         * Interface: ~12 ms (virtual call)
         * Proxy:     ~200 ms (reflection)
         * 
         * Overhead: 10-20x slower
         * 
         * Acceptable for:
         * - Framework code
         * - Infrequent calls
         * - Setup/initialization
         * 
         * Not for:
         * - Hot loops
         * - Performance-critical paths
         * - High-frequency calls
         */
    }
    
    /**
     * Workarounds for limitations
     */
    public static void workarounds() {
        /**
         * WORKAROUNDS FOR LIMITATIONS:
         * 
         * 1. EXTRACT INTERFACE:
         *    Create interface from class
         *    Refactor to use interface
         *    Use JDK proxy
         * 
         * 2. USE CGLIB:
         *    Proxy classes directly
         *    Bytecode generation
         *    See next section
         * 
         * 3. USE BYTEBUDDY:
         *    Modern bytecode library
         *    More powerful
         *    Better API
         * 
         * 4. ADAPTER PATTERN:
         *    Wrap class in interface
         *    Proxy the interface
         *    Delegate to wrapped class
         * 
         * 5. COMPILE-TIME GENERATION:
         *    Annotation processing
         *    Generate proxy at compile-time
         *    No runtime overhead
         */
    }
    
    public static void main(String[] args) {
        interfaceRequirement();
        nonPublicInterfaceRestriction();
        performanceOverhead();
    }
}
```

## 4. CGLIB for Class Proxies

### Proxying Classes Without Interfaces

```java
/**
 * CGLIB FOR CLASS PROXIES
 * 
 * Proxying classes using CGLIB
 * 
 * Dependencies (Maven):
 * <dependency>
 *     <groupId>cglib</groupId>
 *     <artifactId>cglib</artifactId>
 *     <version>3.3.0</version>
 * </dependency>
 */

import net.sf.cglib.proxy.*;
import java.lang.reflect.Method;

public class CGLIBClassProxies {
    
    /**
     * Class without interface
     */
    static class UserService {
        public void createUser(String name) {
            System.out.println("Creating user: " + name);
        }
        
        public String getUser(int id) {
            return "User-" + id;
        }
        
        // Final methods cannot be proxied
        public final void finalMethod() {
            System.out.println("Final method");
        }
    }
    
    /**
     * CGLIB MethodInterceptor
     */
    static class LoggingInterceptor implements MethodInterceptor {
        @Override
        public Object intercept(Object obj, Method method, Object[] args,
                               MethodProxy proxy) throws Throwable {
            System.out.println("Before: " + method.getName());
            
            // Use proxy.invokeSuper() not method.invoke()
            Object result = proxy.invokeSuper(obj, args);
            
            System.out.println("After: " + method.getName());
            return result;
        }
    }
    
    /**
     * Creating CGLIB proxy
     */
    public static void basicCGLIBProxy() {
        System.out.println("=== Basic CGLIB Proxy ===");
        
        Enhancer enhancer = new Enhancer();
        enhancer.setSuperclass(UserService.class);
        enhancer.setCallback(new LoggingInterceptor());
        
        UserService proxy = (UserService) enhancer.create();
        
        proxy.createUser("Alice");
        String user = proxy.getUser(1);
        System.out.println("Got: " + user);
        
        /**
         * CGLIB BASICS:
         * 
         * Enhancer - creates proxy
         * setSuperclass() - class to proxy
         * setCallback() - method interceptor
         * create() - creates instance
         * 
         * Generated class:
         * - Extends target class
         * - Overrides non-final methods
         * - Bytecode generation
         * 
         * Use invokeSuper() not invoke()!
         */
    }
    
    /**
     * Callback filters (selective interception)
     */
    static class SelectiveInterceptor implements MethodInterceptor {
        @Override
        public Object intercept(Object obj, Method method, Object[] args,
                               MethodProxy proxy) throws Throwable {
            System.out.println("  [Intercepted: " + method.getName() + "]");
            return proxy.invokeSuper(obj, args);
        }
    }
    
    static class NoOpInterceptor implements MethodInterceptor {
        @Override
        public Object intercept(Object obj, Method method, Object[] args,
                               MethodProxy proxy) throws Throwable {
            // No logging
            return proxy.invokeSuper(obj, args);
        }
    }
    
    public static void callbackFilters() {
```

```java
import java.lang.reflect.*;

public class SpringAOPMechanisms {
    
    /**
     * Spring proxy decision tree
     */
    public static void springProxyDecision() {
        /**
         * SPRING PROXY DECISION:
         * 
         * Does class implement interface?
         * ├─ YES
         * │  └─ Use JDK dynamic proxy
         * │     (unless forced to CGLIB)
         * │
         * └─ NO
         *    └─ Use CGLIB proxy
         * 
         * 
         * FORCING CGLIB:
         * 
         * @Configuration
         * @EnableAspectJAutoProxy(proxyTargetClass = true)
         * 
         * Or XML:
         * <aop:config proxy-target-class="true"/>
         * 
         * 
         * SPRING BOOT 2.0+:
         * 
         * Default: CGLIB for all
         * Even with interfaces
         * 
         * Reason: More consistent behavior
         * No interface extraction needed
         */
    }
    
    /**
     * Spring AOP example (conceptual)
     */
    public static void springAOPExample() {
        /**
         * SPRING AOP:
         * 
         * @Aspect
         * public class LoggingAspect {
         *   
         *   @Before("execution(* com.example.service.*.*(..))")
         *   public void logBefore(JoinPoint joinPoint) {
         *     System.out.println("Before: " + joinPoint.getSignature());
         *   }
         *   
         *   @Around("@annotation(Transactional)")
         *   public Object handleTransaction(ProceedingJoinPoint pjp) 
         *       throws Throwable {
         *     // Begin transaction
         *     try {
         *       Object result = pjp.proceed();
         *       // Commit
         *       return result;
         *     } catch (Exception e) {
         *       // Rollback
         *       throw e;
         *     }
         *   }
         * }
         * 
         * 
         * ADVICE TYPES:
         * 
         * @Before - before method
         * @After - after method (finally)
         * @AfterReturning - after successful return
         * @AfterThrowing - after exception
         * @Around - wrap method (most powerful)
         * 
         * 
         * POINTCUT EXPRESSIONS:
         * 
         * execution(* com.example.*.*(..))
         * - * = any return type
         * - com.example.* = package
         * - * = any class
         * - *(..) = any method, any args
         * 
         * @annotation(Transactional)
         * - Methods with @Transactional
         * 
         * within(com.example.service.*)
         * - All methods in package
         */
    }
    
    /**
     * Spring-style proxy creation (simplified)
     */
    interface UserService {
        void createUser(String name);
    }
    
    static class UserServiceImpl implements UserService {
        @Override
        public void createUser(String name) {
            System.out.println("Creating: " + name);

    static class SimpleSpringProxy {
        public static <T> T createProxy(T target, Class<T> interfaceClass) {
            return (T) Proxy.newProxyInstance(
                interfaceClass.getClassLoader(),
                new Class<?>[] { interfaceClass },
                new InvocationHandler() {
                    @Override
                    public Object invoke(Object proxy, Method method, Object[] args) 
                            throws Throwable {
                        // Before advice
                        System.out.println("  [Before] " + method.getName());
                        
                        try {
                            // Invoke target
                            Object result = method.invoke(target, args);
                            
                            // After returning
                            System.out.println("  [After Returning]");
                            return result;
                            
                        } catch (InvocationTargetException e) {
                            // After throwing
                            System.out.println("  [After Throwing]");
                            throw e.getCause();
                        } finally {
                            // After (finally)
                            System.out.println("  [After]");
                        }
                    }
                }
            );
        }
    }
    
    public static void springStyleProxy() {
        System.out.println("=== Spring-Style Proxy ===");
        
        UserService target = new UserServiceImpl();
        UserService proxy = SimpleSpringProxy.createProxy(target, UserService.class);
        
        proxy.createUser("Alice");
    }

     */
    @interface Transactional {}
    
    interface AccountService {
        @Transactional
        void transfer(int from, int to, double amount);
        
        double getBalance(int account);
    }
    
    static class AccountServiceImpl implements AccountService {
        @Override
        public void transfer(int from, int to, double amount) {
            System.out.println("    Transferring " + amount);
        }
        
        @Override
        public double getBalance(int account) {
            return 1000.0;
        }
    }
    
    static class TransactionProxy {
        public static <T> T createProxy(T target, Class<T> interfaceClass) {
            return (T) Proxy.newProxyInstance(
                interfaceClass.getClassLoader(),
                new Class<?>[] { interfaceClass },
                (proxy, method, args) -> {
                    // Check if transactional
                    boolean transactional = 
                        method.isAnnotationPresent(Transactional.class);
                    
                    if (transactional) {
                        System.out.println("  BEGIN TRANSACTION");
                    }
                    
                    try {
                        Object result = method.invoke(target, args);
                        
                        if (transactional) {
                            System.out.println("  COMMIT TRANSACTION");
                        }
                        
                        return result;
                        
                    } catch (Exception e) {
                        if (transactional) {
                            System.out.println("  ROLLBACK TRANSACTION");
                        }
                        throw e;
                    }
                }
            );
        }
    }
    
    public static void transactionProxy() {
        System.out.println("\n=== Transaction Proxy ===");
        
        AccountService service = new AccountServiceImpl();
        AccountService proxy = TransactionProxy.createProxy(service, AccountService.class);
        
        proxy.transfer(1, 2, 100.0);

    }

         *      public void methodA() {
         *        methodB();  // No transaction!

         *      }
         *    }

         *    Solution: Inject self or use AspectJ

         * 
         * 2. FINAL METHODS:
         *

         *    Advice not applied

         * 
         * 
         * 3. PRIVATE METHODS:
         *

         *    Advice not applied
         * 
         * 
         * 4. CONSTRUCTOR CALLS:
         *    
         *    Proxy created after construction

         * 
         * 
         * 5. EQUALS/HASHCODE:

         *    Proxy != target

         *    Be careful with proxies in Sets/Maps

    }
```

```java
/**

import java.lang.reflect.*;
import java.util.*;
import java.util.concurrent.*;

public class ProxyPerformance {
    
    interface Calculator {
        int add(int a, int b);
    }
    
    static class CalculatorImpl implements Calculator {
        @Override
        public int add(int a, int b) {
            return a + b;
        }
    }
    
    /**
     * Direct vs Proxy benchmark
     */
    public static void directVsProxyBenchmark() {
        int iterations = 10_000_000;
        
        // Direct call
        Calculator direct = new CalculatorImpl();
        long start = System.nanoTime();
        for (int i = 0; i < iterations; i++) {
            direct.add(1, 2);
        }
        long directTime = System.nanoTime() - start;
        
        // Simple proxy
        Calculator proxy = (Calculator) Proxy.newProxyInstance(
            Calculator.class.getClassLoader(),
            new Class<?>[] { Calculator.class },
            (p, method, args) -> method.invoke(direct, args)
        );
        
        start = System.nanoTime();
        for (int i = 0; i < iterations; i++) {
            proxy.add(1, 2);
        }
        long proxyTime = System.nanoTime() - start;
        
        System.out.println("=== Performance (" + iterations + " iterations) ===");
        System.out.println("Direct:  " + directTime / 1_000_000 + " ms");
        System.out.println("Proxy:   " + proxyTime / 1_000_000 + " ms");
        System.out.println("Overhead: " + 
            String.format("%.1fx", proxyTime / (double) directTime));
        
        /**
         * TYPICAL RESULTS:
         * 
         * Direct:   10 ms
         * Proxy:    150 ms
         * Overhead: 15x slower
         * 
         * Overhead from:
         * - Reflection (method.invoke)
         * - Handler logic
         * - Boxing/unboxing
         */
    }
    
    /**
     * Optimization: Method caching
     */
    static class CachedHandler implements InvocationHandler {
        private final Object target;
        private final Map<Method, Method> methodCache = new ConcurrentHashMap<>();
        
        public CachedHandler(Object target) {
            this.target = target;
        }
        
        @Override
        public Object invoke(Object proxy, Method method, Object[] args) 
                throws Throwable {
            Method targetMethod = methodCache.computeIfAbsent(method, m -> {
                try {
                    return target.getClass().getMethod(m.getName(), m.getParameterTypes());
                } catch (NoSuchMethodException e) {
                    throw new RuntimeException(e);
                }
            });
            
            return targetMethod.invoke(target, args);
        }
    }
    
    /**
     * Best practices
     */
    public static void performanceBestPractices() {
        /**
         * PERFORMANCE BEST PRACTICES:
         * 
         * 1. CACHE PROXIES:
         *    Create once, reuse
         *    Singleton pattern
         * 
         * 2. MINIMIZE HANDLER LOGIC:
         *    Keep invoke() fast
         *    Avoid expensive operations
         * 
         * 3. CACHE METHOD LOOKUPS:
         *    Method objects are expensive
         *    Cache in Map
         * 
         * 4. USE JDK PROXY WHEN POSSIBLE:
         *    Faster than CGLIB

         *    Measure actual impact
         *    Profile hotspots
         * 
         * 6. CONSIDER ALTERNATIVES:
         *    Compile-time generation

    }
```

- Forget about self-invocation issues
