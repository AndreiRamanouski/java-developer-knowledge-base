# Method Handles and invokedynamic

## Overview

Method handles provide a modern, performant alternative to reflection for dynamic method invocation. The invokedynamic bytecode instruction enables efficient implementation of dynamic languages and language features like lambdas. This guide covers MethodHandle vs Reflection, MethodHandles.Lookup API, invokedynamic mechanics, how lambdas use invokedynamic, performance characteristics, practical use cases, and JVM optimization strategies.

---

## 1. MethodHandle vs Reflection

### Modern Alternative to Reflection

```java
/**
 * METHODHANDLE VS REFLECTION
 * 
 * Comparing traditional reflection with method handles
 */

import java.lang.invoke.*;
import java.lang.reflect.*;

public class MethodHandleVsReflection {
    
    /**
     * Example class for comparison
     */
    static class Calculator {
        private int value = 0;
        
        public int add(int a, int b) {
            return a + b;
        }
        
        public void setValue(int value) {
            this.value = value;
        }
        
        public int getValue() {
            return value;
        }
    }
    
    /**
     * Using Reflection
     */
    public static void usingReflection() throws Exception {
        Calculator calc = new Calculator();
        
        // Get method
        Method addMethod = Calculator.class.getMethod("add", int.class, int.class);
        
        // Invoke method
        Integer result = (Integer) addMethod.invoke(calc, 5, 3);
        System.out.println("Reflection result: " + result);
        
        /**
         * REFLECTION CHARACTERISTICS:
         * 
         * - Slower (10-20x overhead)
         * - Type erasure (Object return)
         * - Checked exceptions
         * - Security checks
         * - Not JIT-friendly
         * - Method.invoke() always polymorphic
         */
    }
    
    /**
     * Using MethodHandle
     */
    public static void usingMethodHandle() throws Throwable {
        Calculator calc = new Calculator();
        
        // Get MethodHandles.Lookup
        MethodHandles.Lookup lookup = MethodHandles.lookup();
        
        // Get MethodHandle
        MethodType methodType = MethodType.methodType(int.class, int.class, int.class);
        MethodHandle addHandle = lookup.findVirtual(
            Calculator.class, 
            "add", 
            methodType
        );
        
        // Invoke - note: throws Throwable
        int result = (int) addHandle.invoke(calc, 5, 3);
        System.out.println("MethodHandle result: " + result);
        
        /**
         * METHODHANDLE CHARACTERISTICS:
         * 
         * - Faster (2-5x overhead vs 10-20x for reflection)
         * - Type-safe (with exact invoke)
         * - Throws Throwable
         * - Fewer security checks
         * - JIT-friendly
         * - Can be optimized by JVM
         */
    }
    
    /**
     * Performance comparison
     */
    public static void performanceComparison() throws Throwable {
        int iterations = 10_000_000;
        Calculator calc = new Calculator();
        
        // Direct call baseline
        long start = System.nanoTime();
        for (int i = 0; i < iterations; i++) {
            calc.add(1, 2);
        }
        long directTime = System.nanoTime() - start;
        
        // Reflection
        Method method = Calculator.class.getMethod("add", int.class, int.class);
        start = System.nanoTime();
        for (int i = 0; i < iterations; i++) {
            method.invoke(calc, 1, 2);
        }
        long reflectionTime = System.nanoTime() - start;
        
        // MethodHandle with invoke()
        MethodHandles.Lookup lookup = MethodHandles.lookup();
        MethodType mt = MethodType.methodType(int.class, int.class, int.class);
        MethodHandle mh = lookup.findVirtual(Calculator.class, "add", mt);
        
        start = System.nanoTime();
        for (int i = 0; i < iterations; i++) {
            mh.invoke(calc, 1, 2);
        }
        long mhInvokeTime = System.nanoTime() - start;
        
        // MethodHandle with invokeExact()
        start = System.nanoTime();
        for (int i = 0; i < iterations; i++) {
            int result = (int) mh.invokeExact(calc, 1, 2);
        }
        long mhExactTime = System.nanoTime() - start;
        
        System.out.println("=== Performance (" + iterations + " iterations) ===");
        System.out.println("Direct:            " + directTime / 1_000_000 + " ms");
        System.out.println("Reflection:        " + reflectionTime / 1_000_000 + " ms");
        System.out.println("MH invoke():       " + mhInvokeTime / 1_000_000 + " ms");
        System.out.println("MH invokeExact():  " + mhExactTime / 1_000_000 + " ms");
        System.out.println();
        System.out.println("Reflection overhead:  " + 
            String.format("%.1fx", reflectionTime / (double) directTime));
        System.out.println("MH invoke overhead:   " + 
            String.format("%.1fx", mhInvokeTime / (double) directTime));
        System.out.println("MH exact overhead:    " + 
            String.format("%.1fx", mhExactTime / (double) directTime));
        
        /**
         * TYPICAL RESULTS:
         * 
         * Direct:           10 ms   (baseline)
         * Reflection:       200 ms  (20x slower)
         * MH invoke():      50 ms   (5x slower)
         * MH invokeExact(): 30 ms   (3x slower)
         * 
         * MethodHandle is 4-7x faster than reflection!
         * invokeExact() is faster than invoke()
         */
    }
    
    /**
     * Type safety comparison
     */
    public static void typeSafetyComparison() throws Throwable {
        Calculator calc = new Calculator();
        
        // Reflection - no compile-time safety
        Method method = Calculator.class.getMethod("add", int.class, int.class);
        Object result1 = method.invoke(calc, 1, 2);  // Returns Object
        Integer result2 = (Integer) result1;          // Cast required
        
        // MethodHandle with invoke() - runtime type conversion
        MethodHandles.Lookup lookup = MethodHandles.lookup();
        MethodType mt = MethodType.methodType(int.class, int.class, int.class);
        MethodHandle mh = lookup.findVirtual(Calculator.class, "add", mt);
        
        Object result3 = mh.invoke(calc, 1, 2);       // Can return Object
        int result4 = (int) mh.invoke(calc, 1, 2);    // Or cast to int
        
        // MethodHandle with invokeExact() - exact signature required
        int result5 = (int) mh.invokeExact(calc, 1, 2);  // Exact match
        
        // This would fail at runtime:
        // long result6 = (long) mh.invokeExact(calc, 1, 2);  // WrongMethodTypeException
        
        /**
         * TYPE SAFETY:
         * 
         * Reflection:
         * - No type safety
         * - Always returns Object
         * - Runtime type errors
         * 
         * MethodHandle.invoke():
         * - Runtime type conversion
         * - Adapts arguments
         * - More forgiving
         * 
         * MethodHandle.invokeExact():
         * - Compile-time + runtime checking
         * - Exact signature match
         * - Fastest, but strictest
         * - WrongMethodTypeException if mismatch
         */
    }
    
    /**
     * invoke() vs invokeExact()
     */
    public static void invokeVsInvokeExact() throws Throwable {
        Calculator calc = new Calculator();
        MethodHandles.Lookup lookup = MethodHandles.lookup();
        MethodType mt = MethodType.methodType(int.class, int.class, int.class);
        MethodHandle mh = lookup.findVirtual(Calculator.class, "add", mt);
        
        System.out.println("\n=== invoke() vs invokeExact() ===");
        
        // invoke() - automatic type conversion
        Object result1 = mh.invoke(calc, 1, 2);           // int -> Object (OK)
        int result2 = (int) mh.invoke(calc, 1, 2);        // int (OK)
        Integer result3 = (Integer) mh.invoke(calc, 1, 2); // int -> Integer (OK)
        
        System.out.println("invoke() allows conversions");
        
        // invokeExact() - exact match required
        int result4 = (int) mh.invokeExact(calc, 1, 2);   // Exact match (OK)
        
        try {
            Object result5 = mh.invokeExact(calc, 1, 2);  // Wrong! Expects int
        } catch (WrongMethodTypeException e) {
            System.out.println("invokeExact() requires exact types: " + e.getMessage());
        }
        
        /**
         * INVOKE() VS INVOKEEXACT():
         * 
         * invoke():
         * - Type conversions allowed
         * - Widening/narrowing
         * - Boxing/unboxing
         * - More convenient
         * - Slightly slower
         * 
         * invokeExact():
         * - No conversions
         * - Signature must match exactly
         * - Faster
         * - More strict
         * - Better for performance
         */
    }
    
    /**
     * Key differences summary
     */
    public static void keyDifferences() {
        /**
         * METHODHANDLE VS REFLECTION:
         * 
         * PERFORMANCE:
         * Reflection:     20x slower than direct
         * MH invoke():    5x slower
         * MH invokeExact(): 3x slower
         * Winner: MethodHandle
         * 
         * 
         * TYPE SAFETY:
         * Reflection:     None (Object return)
         * MH invoke():    Runtime conversion
         * MH invokeExact(): Compile + runtime checking
         * Winner: MethodHandle (invokeExact)
         * 
         * 
         * EXCEPTION HANDLING:
         * Reflection:     InvocationTargetException wrapping
         * MethodHandle:   Direct exception throw (Throwable)
         * Winner: MethodHandle (cleaner)
         * 
         * 
         * JIT OPTIMIZATION:
         * Reflection:     Limited optimization
         * MethodHandle:   Inlineable, optimizable
         * Winner: MethodHandle
         * 
         * 
         * EASE OF USE:
         * Reflection:     More familiar, simpler API
         * MethodHandle:   More complex, learning curve
         * Winner: Reflection
         * 
         * 
         * COMPATIBILITY:
         * Reflection:     Java 1.1+
         * MethodHandle:   Java 7+
         * Winner: Reflection
         * 
         * 
         * USE CASES:
         * Reflection:     Rare calls, exploration, tools
         * MethodHandle:   Performance-critical, frameworks, DSLs
         */
    }
    
    public static void main(String[] args) throws Throwable {
        System.out.println("=== Using Reflection ===");
        usingReflection();
        
        System.out.println("\n=== Using MethodHandle ===");
        usingMethodHandle();
        
        performanceComparison();
        typeSafetyComparison();
        invokeVsInvokeExact();
    }
}
```

## 2. MethodHandles.Lookup

### Finding and Creating Method Handles

```java
/**
 * METHODHANDLES.LOOKUP
 * 
 * Various ways to obtain method handles
 */

import java.lang.invoke.*;
import java.lang.reflect.*;

public class MethodHandlesLookupAPI {
    
    /**
     * Example class
     */
    static class Person {
        private String name;
        private int age;
        
        public Person(String name, int age) {
            this.name = name;
            this.age = age;
        }
        
        public String getName() { return name; }
        public void setName(String name) { this.name = name; }
        public int getAge() { return age; }
        
        public static Person create(String name, int age) {
            return new Person(name, age);
        }
    }
    
    /**
     * Getting a Lookup
     */
    public static void gettingLookup() {
        // Public lookup - limited access
        MethodHandles.Lookup publicLookup = MethodHandles.publicLookup();
        
        // Caller lookup - full access from calling class
        MethodHandles.Lookup lookup = MethodHandles.lookup();
        
        // Lookup from specific class
        try {
            MethodHandles.Lookup personLookup = MethodHandles.privateLookupIn(
                Person.class, 
                MethodHandles.lookup()
            );
        } catch (IllegalAccessException e) {
            e.printStackTrace();
        }
        
        /**
         * LOOKUP TYPES:
         * 
         * publicLookup():
         * - Access to public members only
         * - No private/protected access
         * - Safe for untrusted code
         * 
         * lookup():
         * - Full access from calling context
         * - Private members accessible
         * - Same as reflection access
         * 
         * privateLookupIn(Class, Lookup):
         * - Java 9+ feature
         * - Access to private members of target class
         * - Requires module access
         */
    }
    
    /**
     * Finding virtual methods
     */
    public static void findingVirtualMethods() throws Throwable {
        MethodHandles.Lookup lookup = MethodHandles.lookup();
        
        // Find virtual method (instance method)
        MethodType getterType = MethodType.methodType(String.class);
        MethodHandle getNameHandle = lookup.findVirtual(
            Person.class,
            "getName",
            getterType
        );
        
        Person person = new Person("Alice", 30);
        String name = (String) getNameHandle.invoke(person);
        System.out.println("Name: " + name);
        
        // Find setter
        MethodType setterType = MethodType.methodType(void.class, String.class);
        MethodHandle setNameHandle = lookup.findVirtual(
            Person.class,
            "setName",
            setterType
        );
        
        setNameHandle.invoke(person, "Bob");
        System.out.println("New name: " + person.getName());
        
        /**
         * FINDVIRTUAL:
         * 
         * lookup.findVirtual(Class, String, MethodType)
         * 
         * - Instance methods
         * - Polymorphic (respects overriding)
         * - First parameter is receiver (this)
         * - MethodType doesn't include receiver
         */
    }
    
    /**
     * Finding static methods
     */
    public static void findingStaticMethods() throws Throwable {
        MethodHandles.Lookup lookup = MethodHandles.lookup();
        
        // Find static method
        MethodType factoryType = MethodType.methodType(
            Person.class, 
            String.class, 
            int.class
        );
        MethodHandle createHandle = lookup.findStatic(
            Person.class,
            "create",
            factoryType
        );
        
        Person person = (Person) createHandle.invoke("Charlie", 25);
        System.out.println("Created: " + person.getName());
        
        /**
         * FINDSTATIC:
         * 
         * lookup.findStatic(Class, String, MethodType)
         * 
         * - Static methods
         * - No receiver parameter
         * - MethodType includes all parameters
         */
    }
    
    /**
     * Finding constructors
     */
    public static void findingConstructors() throws Throwable {
        MethodHandles.Lookup lookup = MethodHandles.lookup();
        
        // Find constructor
        MethodType constructorType = MethodType.methodType(
            void.class,  // Constructor "returns" void
            String.class, 
            int.class
        );
        MethodHandle constructor = lookup.findConstructor(
            Person.class,
            constructorType
        );
        
        Person person = (Person) constructor.invoke("Diana", 28);
        System.out.println("Constructed: " + person.getName());
        
        /**
         * FINDCONSTRUCTOR:
         * 
         * lookup.findConstructor(Class, MethodType)
         * 
         * - Constructor invocation
         * - Return type is void in MethodType
         * - Actual return is the constructed instance
         */
    }
    
    /**
     * Finding fields
     */
    public static void findingFields() throws Throwable {
        MethodHandles.Lookup lookup = MethodHandles.lookup();
        
        // Getter for instance field
        MethodHandle nameGetter = lookup.findGetter(
            Person.class,
            "name",
            String.class
        );
        
        // Setter for instance field
        MethodHandle nameSetter = lookup.findSetter(
            Person.class,
            "name",
            String.class
        );
        
        Person person = new Person("Eve", 35);
        
        String name = (String) nameGetter.invoke(person);
        System.out.println("Field value: " + name);
        
        nameSetter.invoke(person, "Frank");
        System.out.println("New value: " + nameGetter.invoke(person));
        
        /**
         * FIELD ACCESS:
         * 
         * findGetter(Class, String, Class):
         * - Read field value
         * - Signature: (Object) -> FieldType
         * 
         * findSetter(Class, String, Class):
         * - Write field value
         * - Signature: (Object, FieldType) -> void
         * 
         * findStaticGetter/findStaticSetter:
         * - For static fields
         * - No receiver parameter
         */
    }
    
    /**
     * Special methods
     */
    public static void specialMethods() throws Throwable {
        MethodHandles.Lookup lookup = MethodHandles.lookup();
        
        /**
         * FINDSPECIAL:
         * 
         * Invoke specific implementation
         * Bypass polymorphism
         * Like super.method() call
         * 
         * MethodHandle mh = lookup.findSpecial(
         *   TargetClass.class,
         *   "method",
         *   methodType,
         *   CallerClass.class
         * );
         * 
         * Use cases:
         * - Call super implementation
         * - Bypass overriding
         * - Default interface methods
         */
    }
    
    /**
     * MethodType creation
     */
    public static void methodTypeCreation() {
        // Basic method type
        MethodType mt1 = MethodType.methodType(int.class, int.class, int.class);
        System.out.println("Basic: " + mt1);  // (int,int)int
        
        // Void return
        MethodType mt2 = MethodType.methodType(void.class, String.class);
        System.out.println("Void: " + mt2);   // (String)void
        
        // No parameters
        MethodType mt3 = MethodType.methodType(String.class);
        System.out.println("No params: " + mt3);  // ()String
        
        // Generic return
        MethodType mt4 = MethodType.genericMethodType(2);
        System.out.println("Generic: " + mt4);  // (Object,Object)Object
        
        // From method descriptor
        MethodType mt5 = MethodType.fromMethodDescriptorString(
            "(Ljava/lang/String;I)V",
            Person.class.getClassLoader()
        );
        System.out.println("From descriptor: " + mt5);
        
        /**
         * METHODTYPE CREATION:
         * 
         * methodType(returnType, paramTypes...):
         * - Explicit types
         * 
         * genericMethodType(paramCount):
         * - All Object types
         * 
         * fromMethodDescriptorString(descriptor, loader):
         * - Parse bytecode descriptor
         * 
         * MethodType is immutable
         * Cached by JVM
         * Cheap to create/compare
         */
    }
    
    /**
     * Access control
     */
    public static void accessControl() {
        /**
         * ACCESS MODES:
         * 
         * PUBLIC:
         * - Public members only
         * - publicLookup()
         * 
         * PACKAGE:
         * - Package-private members
         * - Same package
         * 
         * PROTECTED:
         * - Protected members
         * - Subclass access
         * 
         * PRIVATE:
         * - Private members
         * - Same class
         * - lookup() from class
         * 
         * 
         * CHECKING ACCESS:
         * 
         * Lookup has access modes
         * Checked when finding handles
         * IllegalAccessException if denied
         * 
         * 
         * SECURITY:
         * 
         * Lookup objects are capabilities
         * Don't share across trust boundaries
         * Can't escalate privileges
         */
    }
    
    /**
     * Best practices
     */
    public static void bestPractices() {
        /**
         * LOOKUP BEST PRACTICES:
         * 
         * 1. CACHE METHODHANDLES:
         *    Lookup is expensive
         *    MethodHandle is cheap to invoke
         *    Store in static final fields
         * 
         * 2. USE APPROPRIATE LOOKUP:
         *    publicLookup() for untrusted
         *    lookup() for full access
         *    Don't share Lookup objects
         * 
         * 3. HANDLE EXCEPTIONS:
         *    NoSuchMethodException
         *    IllegalAccessException
         *    Fail fast at initialization
         * 
         * 4. USE METHODTYPE CORRECTLY:
         *    Match exact signature
         *    Cache MethodType instances
         *    Descriptors for persistence
         * 
         * 5. PREFER INVOKEEXACT:
         *    Faster than invoke()
         *    Better for known types
         *    Use invoke() for flexibility
         */
    }
    
    public static void main(String[] args) throws Throwable {
        System.out.println("=== Finding Virtual Methods ===");
        findingVirtualMethods();
        
        System.out.println("\n=== Finding Static Methods ===");
        findingStaticMethods();
        
        System.out.println("\n=== Finding Constructors ===");
        findingConstructors();
        
        System.out.println("\n=== Finding Fields ===");
        findingFields();
        
        System.out.println("\n=== MethodType Creation ===");
        methodTypeCreation();
    }
}
```

## 3. invokedynamic Bytecode Instruction

### Understanding the invokedynamic Instruction

```java
/**
 * INVOKEDYNAMIC BYTECODE INSTRUCTION
 * 
 * Understanding how invokedynamic works
 */

import java.lang.invoke.*;

public class InvokedynamicInstruction {
    
    /**
     * What is invokedynamic?
     */
    public static void whatIsInvokedynamic() {
        /**
         * INVOKEDYNAMIC INSTRUCTION:
         * 
         * New bytecode instruction (Java 7)
         * Dynamic method invocation
         * 
         * Traditional:
         * - invokevirtual (instance methods)
         * - invokestatic (static methods)
         * - invokeinterface (interface methods)
         * - invokespecial (constructors, private, super)
         * 
         * invokedynamic:
         * - Dynamic call site
         * - Resolved at runtime
         * - User-defined resolution
         * 
         * 
         * PURPOSE:
         * 
         * Enable efficient implementation of:
         * - Dynamic languages (JRuby, Groovy, JavaScript)
         * - Language features (lambdas, method references)
         * - Runtime code generation
         * - String concatenation (Java 9+)
         * 
         * 
         * ADVANTAGES:
         * 
         * - JVM optimizes call sites
         * - Inlining possible
         * - Better than reflection
         * - Flexible linking
         * - One-time overhead
         */
    }
    
    /**
     * Call site resolution
     */
    public static void callSiteResolution() {
        /**
         * CALL SITE RESOLUTION:
         * 
         * 1. First call to invokedynamic:
         *    - JVM calls bootstrap method
         *    - Bootstrap returns CallSite
         *    - CallSite contains target MethodHandle
         * 
         * 2. Subsequent calls:
         *    - Use cached CallSite
         *    - Invoke target MethodHandle
         *    - No bootstrap overhead
         * 
         * 3. CallSite types:
         *    - ConstantCallSite (immutable)
         *    - MutableCallSite (can change)
         *    - VolatileCallSite (thread-safe changes)
         * 
         * 
         * BOOTSTRAP METHOD:
         * 
         * public static CallSite bootstrap(
         *   MethodHandles.Lookup caller,
         *   String name,
         *   MethodType type,
         *   ...additional args
         * ) {
         *   MethodHandle target = ...;
         *   return new ConstantCallSite(target);
         * }
         * 
         * 
         * INVOCATION:
         * 
         * invokedynamic #1  // Bootstrap method index
         * 
         * Bytecode refers to:
         * - Bootstrap method
         * - Method name
         * - Method type
         * - Static arguments
         */
    }
    
    /**
     * CallSite example
     */
    static class SimpleCallSite {
        // Bootstrap method
        public static CallSite bootstrap(
                MethodHandles.Lookup caller,
                String name,
                MethodType type) throws Throwable {
            
            System.out.println("Bootstrap called!");
            System.out.println("  Lookup: " + caller);
            System.out.println("  Name: " + name);
            System.out.println("  Type: " + type);
            
            // Find target method
            MethodHandle target = caller.findStatic(
                SimpleCallSite.class,
                "actualMethod",
                type
            );
            
            // Return constant call site
            return new ConstantCallSite(target);
        }
        
        public static String actualMethod(String arg) {
            return "Hello, " + arg;
        }
        
        /**
         * CALLSITE TYPES:
         * 
         * ConstantCallSite:
         * - Immutable target
         * - Best performance
         * - JVM can inline
         * - Use for lambdas
         * 
         * MutableCallSite:
         * - Target can change
         * - setTarget() method
         * - Invalidates JIT optimizations
         * - Use for dynamic languages
         * 
         * VolatileCallSite:
         * - Thread-safe target changes
         * - Memory barrier
         * - Slower than Mutable
         * - Use for concurrent updates
         */
    }
    
    /**
     * Dynamic invocation example
     */
    public static void dynamicInvocationExample() {
        /**
         * MANUAL INVOKEDYNAMIC (conceptual):
         * 
         * Can't write invokedynamic in Java source
         * Only in bytecode
         * 
         * But can simulate with MethodHandle:
         * 
         * // Bootstrap creates CallSite
         * CallSite cs = bootstrap(lookup, "method", type);
         * 
         * // Get target
         * MethodHandle target = cs.getTarget();
         * 
         * // Invoke (like invokedynamic would)
         * target.invoke(args);
         * 
         * 
         * REAL INVOKEDYNAMIC:
         * 
         * Bytecode:
         * 0: ldc "World"
         * 2: invokedynamic #1 // Method bootstrap
         * 7: areturn
         * 
         * Constant pool:
         * #1 = InvokeDynamic #0:#2
         * #2 = NameAndType method:(String)String
         * #0 = BootstrapMethod bootstrap
         */
    }
    
    /**
     * Use in dynamic languages
     */
    public static void dynamicLanguagesUsage() {
        /**
         * DYNAMIC LANGUAGES:
         * 
         * Problem with JVM:
         * - Static typing
         * - Fixed method dispatch
         * - Hard to implement dynamic features
         * 
         * Before invokedynamic:
         * - Reflection (slow)
         * - Generated bytecode (complex)
         * - Inline caching (limited)
         * 
         * With invokedynamic:
         * - Fast method dispatch
         * - JVM optimizations
         * - Clean implementation
         * 
         * 
         * EXAMPLE: JRuby method call
         * 
         * Ruby: obj.method(arg)
         * 
         * invokedynamic bootstrap:
         * 1. Check object type
         * 2. Find method
         * 3. Create MethodHandle
         * 4. Return CallSite
         * 
         * Subsequent calls:
         * - Fast MethodHandle invoke
         * - JVM can inline
         * - Type profiling
         * 
         * 
         * LANGUAGES USING INVOKEDYNAMIC:
         * 
         * - JRuby
         * - Groovy
         * - Nashorn (JavaScript)
         * - Kotlin (some features)
         * - Scala (some features)
         * - Clojure
         */
    }
    
    /**
     * String concatenation (Java 9+)
     */
    public static void stringConcatenation() {
        /**
         * STRING CONCATENATION EVOLUTION:
         * 
         * Java 8 and earlier:
         * String s = "a" + b + "c";
         * 
         * Compiled to:
         * new StringBuilder()
```

```java
        private static final Map<Class<?>, List<MethodHandle>> GETTERS = 
            new HashMap<>();

            List<MethodHandle> getters = new ArrayList<>();
            MethodHandles.Lookup lookup = MethodHandles.lookup();
            
            for (var field : clazz.getDeclaredFields()) {
                String getterName = "get" + 
                    Character.toUpperCase(field.getName().charAt(0)) +

                        clazz, getterName, mt);
                    getters.add(getter);
                } catch (NoSuchMethodException e) {
                    // No getter, skip
                }

            Map<String, Object> data = new HashMap<>();

                for (MethodHandle getter : getters) {
                    Object value = getter.invoke(obj);
```

```java
         *    static final MethodHandle MH = ...;
         * 
         * 2. Call site is hot
         *    Executed many times
         * 
         * 3. Method small enough
         *    < 35 bytecodes (default)
         * 
         * 4. Use invokeExact()
         *    invoke() harder to inline
         * 
         * 
         * BENEFITS:
         * 
         * - Eliminates call overhead
         * - Enables further optimizations
         * - Escape analysis
         * - Dead code elimination
         * - Constant folding
         */
    }

    public static void typeProfiling() {
        /**
         * TYPE PROFILING:
         * 
         * JVM tracks types at call sites

         * Example:
         * MethodHandle mh = ...;
         * for (Object obj : objects) {
         *   mh.invoke(obj);
         * }
         * 
         * If 99% of objects are String:
         * 1. JVM generates optimized code for String

         * if (obj instanceof String) {
         *   // Optimized path

         */
    }
    
    /**
     * Polymorphic inline cache
     */
    public static void polymorphicInlineCache() {
        /**
         * POLYMORPHIC INLINE CACHE (PIC):
         * 
         * Handle multiple types efficiently
         * 
         * Monomorphic (1 type):
         * if (type == String.class) {
         *   return stringHandler();
         * }
         * 
         * Bimorphic (2 types):
         * if (type == String.class) {
         *   return stringHandler();

         * Polymorphic (3+ types):

         * Megamorphic (many types):

         * - Falls back to generic
         * - Performance degradation
         * 
         * 
         * OPTIMIZATION STRATEGY:
         * 
         * Keep call sites monomorphic
         * - Use different MethodHandles
         * - Separate code paths
         * - Avoid type pollution
         */
    }
    
    /**
     * Escape analysis

    public static void escapeAnalysis() {
        /**
         * ESCAPE ANALYSIS:
         * 
         * Determines if object escapes method

         * With inlining:
         * 
         * Integer boxed = 42;  // Boxing
         * mh.invokeExact(boxed);

         * If mh inlined and Integer doesn't escape:
         * - No allocation

         * 
         * SCALAR REPLACEMENT:
         * 
         * Object fields → local variables

         * Example:
         * Point p = new Point(x, y);

         * 
         * Optimized to:
         * int px = x;
         * int py = y;

         * 
         * 
         * METHODHANDLE BENEFIT:
         * 
         * Inlining enables escape analysis
         * Better than reflection
         * Can eliminate boxing
         */
    }
    
    /**
     * Constant folding
     */
    public static void constantFolding() {
        /**
         * CONSTANT FOLDING:
         * 
         * Evaluate constants at compile time
         * 
         * static final MethodHandle ADD = ...;
         * 
         * int result = (int) ADD.invokeExact(calculator, 2, 3);
         * 
         * If ADD is constant and inlined:
         * - JIT knows exact method
         * - Can constant-fold result
         * - Dead code elimination
         * 
         * 
         * LOOP UNROLLING:
         * 
         * Combined with constant folding:
         * 
         * for (int i = 0; i < 4; i++) {
         *   result += (int) ADD.invokeExact(calc, i, i);
         * }
         * 
         * Can unroll and evaluate:
         * result += 0;  // 0 + 0
         * result += 2;  // 1 + 1
         * result += 4;  // 2 + 2

    }
```
