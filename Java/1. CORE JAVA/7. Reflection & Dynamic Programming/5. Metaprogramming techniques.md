# Metaprogramming Techniques

## Overview

Metaprogramming is writing code that generates, manipulates, or analyzes other code. Java supports metaprogramming through code generation (source and bytecode), bytecode manipulation, annotation processing, and runtime class creation. This guide covers tools like JavaPoet and ByteBuddy for code generation, ASM and Javassist for bytecode manipulation, compile-time vs runtime generation strategies, real-world use cases in frameworks like Hibernate and Mockito, and critical trade-offs including performance, maintainability, and security risks.

## 1. Code Generation with JavaPoet

### Type-Safe Source Code Generation

```java
/**
 * CODE GENERATION WITH JAVAPOET
 * 
 * Generate Java source code programmatically
 * 
 * Dependencies (Maven):
 * <dependency>
 *     <groupId>com.squareup</groupId>
 *     <artifactId>javapoet</artifactId>
 *     <version>1.13.0</version>
 * </dependency>
 */

import com.squareup.javapoet.*;
import javax.lang.model.element.Modifier;
import java.io.IOException;

public class JavaPoetGeneration {
    
    /**
     * Basic class generation
     */
    public static void generateSimpleClass() throws IOException {
        // Generate a simple class
        TypeSpec helloWorld = TypeSpec.classBuilder("HelloWorld")
            .addModifiers(Modifier.PUBLIC, Modifier.FINAL)
            .addMethod(MethodSpec.methodBuilder("main")
                .addModifiers(Modifier.PUBLIC, Modifier.STATIC)
                .returns(void.class)
                .addParameter(String[].class, "args")
                .addStatement("$T.out.println($S)", System.class, "Hello, JavaPoet!")
                .build())
            .build();
        
        JavaFile javaFile = JavaFile.builder("com.example", helloWorld)
            .build();
        
        // Print to console
        javaFile.writeTo(System.out);
        
        /**
         * GENERATED CODE:
         * 
         * package com.example;
         * 
         * import java.lang.String;
         * import java.lang.System;
         * 
         * public final class HelloWorld {
         *   public static void main(String[] args) {
         *     System.out.println("Hello, JavaPoet!");
         *   }
         * }
         */
    }
    
    /**
     * Generating a builder pattern
     */
    public static void generateBuilder() throws IOException {
        // Field specifications
        FieldSpec name = FieldSpec.builder(String.class, "name")
            .addModifiers(Modifier.PRIVATE, Modifier.FINAL)
            .build();
        
        FieldSpec age = FieldSpec.builder(int.class, "age")
            .addModifiers(Modifier.PRIVATE, Modifier.FINAL)
            .build();
        
        // Constructor
        MethodSpec constructor = MethodSpec.constructorBuilder()
            .addModifiers(Modifier.PRIVATE)
            .addParameter(String.class, "name")
            .addParameter(int.class, "age")
            .addStatement("this.$N = $N", "name", "name")
            .addStatement("this.$N = $N", "age", "age")
            .build();
        
        // Getters
        MethodSpec getName = MethodSpec.methodBuilder("getName")
            .addModifiers(Modifier.PUBLIC)
            .returns(String.class)
            .addStatement("return $N", name)
            .build();
        
        MethodSpec getAge = MethodSpec.methodBuilder("getAge")
            .addModifiers(Modifier.PUBLIC)
            .returns(int.class)
            .addStatement("return $N", age)
            .build();
        
        // Builder class
        TypeSpec builder = TypeSpec.classBuilder("Builder")
            .addModifiers(Modifier.PUBLIC, Modifier.STATIC)
            .addField(String.class, "name", Modifier.PRIVATE)
            .addField(int.class, "age", Modifier.PRIVATE)
            .addMethod(MethodSpec.methodBuilder("name")
                .addModifiers(Modifier.PUBLIC)
                .returns(ClassName.get("", "Builder"))
                .addParameter(String.class, "name")
                .addStatement("this.$N = $N", "name", "name")
                .addStatement("return this")
                .build())
            .addMethod(MethodSpec.methodBuilder("age")
                .addModifiers(Modifier.PUBLIC)
                .returns(ClassName.get("", "Builder"))
                .addParameter(int.class, "age")
                .addStatement("this.$N = $N", "age", "age")
                .addStatement("return this")
                .build())
            .addMethod(MethodSpec.methodBuilder("build")
                .addModifiers(Modifier.PUBLIC)
                .returns(ClassName.get("", "Person"))
                .addStatement("return new $T($N, $N)", 
                    ClassName.get("", "Person"), "name", "age")
                .build())
            .build();
        
        // Main class
        TypeSpec person = TypeSpec.classBuilder("Person")
            .addModifiers(Modifier.PUBLIC, Modifier.FINAL)
            .addField(name)
            .addField(age)
            .addMethod(constructor)
            .addMethod(getName)
            .addMethod(getAge)
            .addType(builder)
            .addMethod(MethodSpec.methodBuilder("builder")
                .addModifiers(Modifier.PUBLIC, Modifier.STATIC)
                .returns(ClassName.get("", "Builder"))
                .addStatement("return new $T()", ClassName.get("", "Builder"))
                .build())
            .build();
        
        JavaFile javaFile = JavaFile.builder("com.example", person)
            .build();
        
        javaFile.writeTo(System.out);
        
        /**
         * JAVAPOET FEATURES:
         * 
         * TypeSpec - Class/Interface/Enum
         * MethodSpec - Method
         * FieldSpec - Field
         * ParameterSpec - Parameter
         * AnnotationSpec - Annotation
         * 
         * Format strings:
         * $T - Type (auto-imports)
         * $N - Name (no imports)
         * $S - String literal
         * $L - Literal (no escaping)
         */
    }
    
    /**
     * Generating interfaces with generics
     */
    public static void generateGenericInterface() throws IOException {
        // Type variable
        TypeVariableName t = TypeVariableName.get("T");
        
        // Interface
        TypeSpec repository = TypeSpec.interfaceBuilder("Repository")
            .addModifiers(Modifier.PUBLIC)
            .addTypeVariable(t)
            .addMethod(MethodSpec.methodBuilder("findById")
                .addModifiers(Modifier.PUBLIC, Modifier.ABSTRACT)
                .returns(t)
                .addParameter(long.class, "id")
                .build())
            .addMethod(MethodSpec.methodBuilder("save")
                .addModifiers(Modifier.PUBLIC, Modifier.ABSTRACT)
                .returns(void.class)
                .addParameter(t, "entity")
                .build())
            .addMethod(MethodSpec.methodBuilder("findAll")
                .addModifiers(Modifier.PUBLIC, Modifier.ABSTRACT)
                .returns(ParameterizedTypeName.get(
                    ClassName.get("java.util", "List"),
                    t
                ))
                .build())
            .build();
        
        JavaFile javaFile = JavaFile.builder("com.example", repository)
            .build();
        
        javaFile.writeTo(System.out);
        
        /**
         * GENERICS IN JAVAPOET:
         * 
         * TypeVariableName - <T>, <E extends X>
         * ParameterizedTypeName - List<T>, Map<K,V>
         * WildcardTypeName - ? extends X, ? super Y
         */
    }
    
    /**
     * Annotations and Javadoc
     */
    public static void generateAnnotatedClass() throws IOException {
        MethodSpec toString = MethodSpec.methodBuilder("toString")
            .addAnnotation(Override.class)
            .addModifiers(Modifier.PUBLIC)
            .returns(String.class)
            .addStatement("return $S + $N + $S + $N", 
                "User{name='", "name", "', email='", "email") 
            .addStatement("return result + $S", "'}")
            .build();
        
        TypeSpec user = TypeSpec.classBuilder("User")
            .addModifiers(Modifier.PUBLIC)
            .addJavadoc("Represents a user in the system.\n")
            .addJavadoc("\n")
            .addJavadoc("@author Generated\n")
            .addAnnotation(AnnotationSpec.builder(SuppressWarnings.class)
                .addMember("value", "$S", "unused")
                .build())
            .addField(FieldSpec.builder(String.class, "name")
                .addModifiers(Modifier.PRIVATE)
                .addJavadoc("The user's name\n")
                .build())
            .addField(FieldSpec.builder(String.class, "email")
                .addModifiers(Modifier.PRIVATE)
                .addJavadoc("The user's email address\n")
                .build())
            .addMethod(toString)
            .build();
        
        JavaFile javaFile = JavaFile.builder("com.example", user)
            .build();
        
        javaFile.writeTo(System.out);
    }
    
    /**
     * JavaPoet use cases
     */
    public static void javaPoetUseCases() {
        /**
         * JAVAPOET USE CASES:
         * 
         * 1. ANNOTATION PROCESSORS:
         *    - Generate code at compile-time
         *    - Lombok alternative
         *    - AutoValue, Immutables
         * 
         * 2. BUILDER GENERATION:
         *    - Type-safe builders
         *    - Fluent APIs
         *    - Configuration objects
         * 
         * 3. DTO/ENTITY GENERATION:
         *    - Database entities
         *    - API models
         *    - Protocol buffers
         * 
         * 4. FACTORY CLASSES:
         *    - Service locators
         *    - Dependency injection
         *    - Plugin systems
         * 
         * 5. TEST CODE:
         *    - Mock generators
         *    - Test fixtures
         *    - Data builders
         * 
         * 
         * ADVANTAGES:
         * 
         * ✓ Type-safe code generation
         * ✓ Readable generated code
         * ✓ Auto-import management
         * ✓ Fluent API
         * ✓ Well-maintained
         * 
         * 
         * LIMITATIONS:
         * 
         * ✗ Only generates source code
         * ✗ Requires compilation
         * ✗ Cannot modify existing classes
         * ✗ Verbose for simple cases
         */
    }
    
    public static void main(String[] args) throws IOException {
        System.out.println("=== Simple Class Generation ===");
        generateSimpleClass();
        
        System.out.println("\n=== Builder Pattern ===");
        generateBuilder();
        
        System.out.println("\n=== Generic Interface ===");
        generateGenericInterface();
    }
}
```

## 2. Code Generation with ByteBuddy

### Runtime Bytecode Generation

```java
/**
 * CODE GENERATION WITH BYTEBUDDY
 * 
 * Generate classes at runtime via bytecode
 * 
 * Dependencies (Maven):
 * <dependency>
 *     <groupId>net.bytebuddy</groupId>
 *     <artifactId>byte-buddy</artifactId>
 *     <version>1.14.10</version>
 * </dependency>
 */

import net.bytebuddy.ByteBuddy;
import net.bytebuddy.implementation.*;
import net.bytebuddy.matcher.ElementMatchers;
import net.bytebuddy.dynamic.loading.ClassLoadingStrategy;

public class ByteBuddyGeneration {
    
    /**
     * Creating a simple class
     */
    public static void createSimpleClass() throws Exception {
        Class<?> dynamicType = new ByteBuddy()
            .subclass(Object.class)
            .name("com.example.DynamicClass")
            .make()
            .load(ByteBuddyGeneration.class.getClassLoader())
            .getLoaded();
        
        Object instance = dynamicType.getDeclaredConstructor().newInstance();
        System.out.println("Created: " + instance.getClass().getName());
        
        /**
         * BYTEBUDDY BASICS:
         * 
         * ByteBuddy() - Builder
         * subclass() - Extend class
         * implement() - Implement interface
         * make() - Generate bytecode
         * load() - Load into JVM
         */
    }
    
    /**
     * Implementing interface
     */
    interface Greeting {
        String greet(String name);
    }
    
    public static void implementInterface() throws Exception {
        Class<?> dynamicType = new ByteBuddy()
            .subclass(Object.class)
            .implement(Greeting.class)
            .method(ElementMatchers.named("greet"))
            .intercept(FixedValue.value("Hello, World!"))
            .make()
            .load(ByteBuddyGeneration.class.getClassLoader())
            .getLoaded();
        
        Greeting greeting = (Greeting) dynamicType.getDeclaredConstructor().newInstance();
        System.out.println(greeting.greet("Alice"));  // Hello, World!
        
        /**
         * IMPLEMENTATION STRATEGIES:
         * 
         * FixedValue - Return constant
         * FieldAccessor - Access field
         * MethodCall - Call method
         * MethodDelegation - Delegate to another method
         * SuperMethodCall - Call super
         * StubMethod - Return default value
         */
    }
    
    /**
     * Field and method generation
     */
    public static void generateFieldsAndMethods() throws Exception {
        Class<?> personClass = new ByteBuddy()
            .subclass(Object.class)
            .name("Person")
            .defineField("name", String.class, 
                net.bytebuddy.description.modifier.Visibility.PRIVATE)
            .defineField("age", int.class,
                net.bytebuddy.description.modifier.Visibility.PRIVATE)
            .defineConstructor(
                net.bytebuddy.description.modifier.Visibility.PUBLIC)
            .withParameters(String.class, int.class)
            .intercept(MethodCall.invoke(Object.class.getConstructor())
                .andThen(FieldAccessor.ofField("name").setsArgumentAt(0))
                .andThen(FieldAccessor.ofField("age").setsArgumentAt(1)))
            .defineMethod("getName", String.class,
                net.bytebuddy.description.modifier.Visibility.PUBLIC)
            .intercept(FieldAccessor.ofField("name"))
            .defineMethod("getAge", int.class,
                net.bytebuddy.description.modifier.Visibility.PUBLIC)
            .intercept(FieldAccessor.ofField("age"))
            .make()
            .load(ByteBuddyGeneration.class.getClassLoader())
            .getLoaded();
        
        Object person = personClass
            .getConstructor(String.class, int.class)
            .newInstance("Alice", 30);
        
        System.out.println("Name: " + 
            personClass.getMethod("getName").invoke(person));
        System.out.println("Age: " + 
            personClass.getMethod("getAge").invoke(person));
    }
    
    /**
     * Method delegation
     */
    static class Interceptor {
        public static String intercept(String name) {
            return "Hello, " + name + "!";
        }
    }
    
    public static void methodDelegation() throws Exception {
        Class<?> dynamicType = new ByteBuddy()
            .subclass(Object.class)
            .implement(Greeting.class)
            .method(ElementMatchers.named("greet"))
            .intercept(MethodDelegation.to(Interceptor.class))
            .make()
            .load(ByteBuddyGeneration.class.getClassLoader())
            .getLoaded();
        
        Greeting greeting = (Greeting) dynamicType.getDeclaredConstructor().newInstance();
        System.out.println(greeting.greet("Bob"));
        
        /**
         * METHOD DELEGATION:
         * 
         * Delegates to static or instance method
         * Matches by:
         * - Parameter types
         * - Return type
         * - Annotations
         * 
         * Annotations:
         * @Argument - Bind specific argument
         * @AllArguments - Bind all arguments
         * @This - Bind receiver
         * @Super - Bind super method
         * @Origin - Bind Method/Constructor
         */
    }
    
    /**
     * ByteBuddy use cases
     */
    public static void byteBuddyUseCases() {
        /**
         * BYTEBUDDY USE CASES:
         * 
         * 1. RUNTIME PROXIES:
         *    - AOP implementations
         *    - Lazy loading
         *    - Caching
         * 
         * 2. MOCKING FRAMEWORKS:
         *    - Mockito uses ByteBuddy
         *    - Dynamic mock generation
         *    - Spy objects
         * 
         * 3. ORM FRAMEWORKS:
         *    - Hibernate entity proxies
         *    - Lazy associations
         *    - Dirty tracking
         * 
         * 4. JAVA AGENTS:
         *    - Instrumentation
         *    - Profiling
         *    - APM tools
         * 
         * 5. CLASS RELOADING:
         *    - Hot deployment
         *    - Development tools
         *    - Test frameworks
         * 
         * 
         * ADVANTAGES:
         * 
         * ✓ Runtime generation
         * ✓ No compilation needed
         * ✓ Modify existing classes
         * ✓ Fast execution
         * ✓ Modern API
         * 
         * 
         * LIMITATIONS:
         * 
         * ✗ Generated bytecode not readable
         * ✗ Debugging harder
         * ✗ Learning curve
         * ✗ Runtime overhead
         */
    }
    
    public static void main(String[] args) throws Exception {
        System.out.println("=== Simple Class ===");
        createSimpleClass();
        
        System.out.println("\n=== Implement Interface ===");
        implementInterface();
        
        System.out.println("\n=== Fields and Methods ===");
        generateFieldsAndMethods();
        
        System.out.println("\n=== Method Delegation ===");
        methodDelegation();
    }
}
```

## 3. Bytecode Manipulation with ASM

### Low-Level Bytecode Operations

```java
/**
 * BYTECODE MANIPULATION WITH ASM
 * 
 * Direct bytecode generation and manipulation
 * 
 * Dependencies (Maven):
 * <dependency>
 *     <groupId>org.ow2.asm</groupId>
 *     <artifactId>asm</artifactId>
 *     <version>9.6</version>
 * </dependency>
 */

import org.objectweb.asm.*;
import static org.objectweb.asm.Opcodes.*;

public class ASMBytecodeManipulation {
    
    /**
     * Generating a simple class with ASM
     */
    public static byte[] generateSimpleClass() {
        ClassWriter cw = new ClassWriter(ClassWriter.COMPUTE_FRAMES);
        
        // Class declaration
        cw.visit(V11, ACC_PUBLIC, "com/example/Generated", 
                null, "java/lang/Object", null);
        
        // Constructor
        MethodVisitor mv = cw.visitMethod(ACC_PUBLIC, "<init>", 
                "()V", null, null);
        mv.visitCode();
        mv.visitVarInsn(ALOAD, 0);
        mv.visitMethodInsn(INVOKESPECIAL, "java/lang/Object", 
                "<init>", "()V", false);
        mv.visitInsn(RETURN);
        mv.visitMaxs(1, 1);
        mv.visitEnd();
        
        // Method: public String getMessage()
        mv = cw.visitMethod(ACC_PUBLIC, "getMessage", 
                "()Ljava/lang/String;", null, null);
        mv.visitCode();
        mv.visitLdcInsn("Hello from ASM!");
        mv.visitInsn(ARETURN);
        mv.visitMaxs(1, 1);
        mv.visitEnd();
        
        cw.visitEnd();
        
        return cw.toByteArray();
        
        /**
         * ASM VISITOR PATTERN:
         * 
         * ClassWriter - Writes class bytecode
         * ClassVisitor - Visits class elements
         * MethodVisitor - Visits method bytecode
         * FieldVisitor - Visits field
         * AnnotationVisitor - Visits annotation
         * 
         * 
         * OPCODES:
         * 
         * ALOAD - Load reference
         * ILOAD - Load int
         * ASTORE - Store reference
         * ISTORE - Store int
         * INVOKESPECIAL - Call constructor/private
         * INVOKEVIRTUAL - Call instance method
         * INVOKESTATIC - Call static method
         * RETURN - Return void
         * ARETURN - Return reference
         * IRETURN - Return int
         */
    }
    
    /**
     * Loading generated class
     */
    public static void loadGeneratedClass() throws Exception {
        byte[] bytecode = generateSimpleClass();
        
        // Custom ClassLoader
        ClassLoader loader = new ClassLoader() {
            public Class<?> defineClass(String name, byte[] b) {
                return defineClass(name, b, 0, b.length);
            }
        };
        
        Class<?> clazz = loader.defineClass("com.example.Generated", bytecode);
        Object instance = clazz.getDeclaredConstructor().newInstance();
        
        Object result = clazz.getMethod("getMessage").invoke(instance);
        System.out.println("Result: " + result);
    }
    
    /**
     * Class transformation
     */
    public static class LoggingClassVisitor extends ClassVisitor {
        public LoggingClassVisitor(ClassVisitor cv) {
            super(ASM9, cv);
        }
        
        @Override
        public MethodVisitor visitMethod(int access, String name, 
                String descriptor, String signature, String[] exceptions) {
            MethodVisitor mv = super.visitMethod(access, name, 
                    descriptor, signature, exceptions);
            
            // Add logging to every method
            return new MethodVisitor(ASM9, mv) {
                @Override
                public void visitCode() {
                    super.visitCode();
                    // Add: System.out.println("Entering: " + name);
                    mv.visitFieldInsn(GETSTATIC, "java/lang/System", 
                            "out", "Ljava/io/PrintStream;");
                    mv.visitLdcInsn("Entering: " + name);
                    mv.visitMethodInsn(INVOKEVIRTUAL, "java/io/PrintStream", 
                            "println", "(Ljava/lang/String;)V", false);
                }
            };
        }
    }
    
    /**
     * ASM characteristics
     */
    public static void asmCharacteristics() {
        /**
         * ASM CHARACTERISTICS:
         * 
         * ADVANTAGES:
         * 
         * ✓ Most powerful
         * ✓ Fastest bytecode library
         * ✓ Small footprint (~50KB)
         * ✓ Used by Java compiler
         * ✓ Complete control
         * 
         * 
         * LIMITATIONS:
         * 
         * ✗ Very low-level
         * ✗ Steep learning curve
         * ✗ Bytecode knowledge required
         * ✗ Error-prone
         * ✗ Verbose
         * 
         * 
         * WHEN TO USE:
         * 
         * - Maximum performance needed
         * - Building other tools
         * - Class transformation
         * - Instrumentation
         * - When higher-level tools insufficient
         * 
         * 
         * WHEN NOT TO USE:
         * 
         * - Application code
         * - Simple proxies
         * - First time with bytecode
         * - Rapid development
         */
    }
    
    public static void main(String[] args) throws Exception {
        System.out.println("=== ASM Class Generation ===");
        loadGeneratedClass();
    }
}
```

## 4. Bytecode Manipulation with Javassist

### High-Level Bytecode Manipulation

```java
/**
 * BYTECODE MANIPULATION WITH JAVASSIST
 * 
 * Source-level bytecode manipulation
 * 
 * Dependencies (Maven):
 * <dependency>
 *     <groupId>org.javassist</groupId>
 *     <artifactId>javassist</artifactId>
 *     <version>3.29.2-GA</version>
 * </dependency>
 */

import javassist.*;

public class JavassistManipulation {
    
    /**
     * Creating a simple class
     */
    public static void createSimpleClass() throws Exception {
        ClassPool pool = ClassPool.getDefault();
        
        // Create class
        CtClass cc = pool.makeClass("com.example.Person");
        
        // Add fields
        CtField nameField = new CtField(
            pool.get("java.lang.String"), "name", cc);
        nameField.setModifiers(Modifier.PRIVATE);
        cc.addField(nameField);
        
        CtField ageField = new CtField(CtClass.intType, "age", cc);
        ageField.setModifiers(Modifier.PRIVATE);
        cc.addField(ageField);
        
        // Add constructor
        CtConstructor constructor = new CtConstructor(
            new CtClass[] { pool.get("java.lang.String"), CtClass.intType },
            cc
        );
        constructor.setBody(
            "{ this.name = $1; this.age = $2; }"
        );
        cc.addConstructor(constructor);
        
        // Add getter
        CtMethod getName = new CtMethod(
            pool.get("java.lang.String"), "getName", new CtClass[0], cc);
        getName.setBody("{ return this.name; }");
        cc.addMethod(getName);
        
        // Load class
        Class<?> clazz = cc.toClass();
        Object person = clazz.getConstructor(String.class, int.class)
            .newInstance("Alice", 30);
        
        System.out.println("Name: " + clazz.getMethod("getName").invoke(person));
        
        /**
         * JAVASSIST FEATURES:
         * 
         * ClassPool - Manages classes
         * CtClass - Class representation
         * CtField - Field
         * CtMethod - Method
         * CtConstructor - Constructor
         * 
         * Source-level API:
         * - Write Java source code
         * - Javassist compiles to bytecode
         * - Easier than ASM
         */
    }
    
    /**
     * Modifying existing class
     */
    public static void modifyExistingClass() throws Exception {
        ClassPool pool = ClassPool.getDefault();
        
        // Get existing class
        CtClass cc = pool.get("java.util.ArrayList");
        
        // Make it not frozen (if needed)
        cc.defrost();
        
        // Add method
        CtMethod logMethod = CtNewMethod.make(
            "public void logSize() { " +
            "  System.out.println(\"Size: \" + this.size()); " +
            "}",
            cc
        );
        cc.addMethod(logMethod);
        
        // Load modified class
        Class<?> modifiedClass = cc.toClass();
        
        /**
         * CLASS MODIFICATION:
         * 
         * Can modify:
         * - Add methods
         * - Add fields
         * - Change method bodies
         * - Add interfaces
         * 
         * Cannot modify:
         * - Already loaded classes (normally)
         * - Frozen classes (call defrost())
         */
    }
    
    /**
     * Method instrumentation
     */
    public static void methodInstrumentation() throws Exception {
        ClassPool pool = ClassPool.getDefault();
        CtClass cc = pool.makeClass("com.example.Calculator");
        
        // Create method
        CtMethod addMethod = CtNewMethod.make(
            "public int add(int a, int b) { return a + b; }",
            cc
        );
        
        // Add before
        addMethod.insertBefore(
            "System.out.println(\"Calling add with: \" + $1 + \", \" + $2);"
        );
        
        // Add after
        addMethod.insertAfter(
            "System.out.println(\"Result: \" + $_ + \"\");",
            false  // false = don't include exception handler
        );
        
        cc.addMethod(addMethod);
        
        // Test
        Class<?> clazz = cc.toClass();
        Object calc = clazz.getDeclaredConstructor().newInstance();
        clazz.getMethod("add", int.class, int.class).invoke(calc, 5, 3);
        
        /**
         * SPECIAL VARIABLES:
         * 
         * $0, $1, $2, ... - Method parameters
         * $0 = this (for instance methods)
         * $1 = first parameter
         * $2 = second parameter
         * 
         * $_ - Return value
         * $r - Return type
         * $w - Wrapper type
         * $args - Array of all parameters
         * $class - Class object
         * $sig - Array of Class (parameter types)
         */
    }
    
    /**
     * Javassist characteristics
```

```java
         * Compile-time gen: 100%

         * Runtime gen (cold): 10-50%
         * Reflection: 5-10%
         * 
         * After JIT warmup, runtime generation
         * performs nearly identical to direct code
         * 
         * 
         * MEMORY USAGE:
         *

         * - Metaspace (class metadata)
         * - Heap (instances)
```
